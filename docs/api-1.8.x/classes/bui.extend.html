Example:

   bui.extend({
                                           name:"accordion",
                                           config: {
                                               id: "",
                                               handle: "dt",
                                               target: "dd",
                                               targetHeight: 0,
                                               single: false,
                                               autoinit: true,
                                               callback: null
                                           },
                                           callback:function(option){
                                               // 获取父级的对象,该对象有以下公共方法
                                               var that = this;
                                    
                                               that.widget
                                               that.destroy;
                                               that.on;
                                               that.off;
                                               that.trigger;
                                               
                                               // 控件的初始化
                                               function inited(){
                                    
                                                   // 链式调用
                                                   return that
                                               }
                                               // 抛出对应的方法, 不能为以下方法或属性: 
                                               init, option, config, widgets, widget, reinit, destroy, handle, on, off, trigger, initStatus, reInitStatus,  emitter, prefix, showLog
                                    
                                               that.inited = inited;
                                               
                                               // 执行一次初始化
                                               inited();
                                               
                                               // 返回才能拿到对应的方法
                                               return that;
                                           }
                                       })
