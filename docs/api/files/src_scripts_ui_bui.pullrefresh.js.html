                                 *  {{#crossLink "bui.pullrefresh/unlock"}}{{/crossLink}}: 解锁滑动,一般用于控件冲突 <br>
                                 *  {{#crossLink "bui.pullrefresh/setHeight"}}{{/crossLink}}: 设置内容滚动高度 <br>
                                 *   </div>
                                 *   <div class="span4"><a href="http://www.easybui.com/demo/index.html#pages/ui_controls/bui.list_news.html" target="_blank"><img src="http://www.easybui.com/static/images/controls/bui-list_low.gif" alt="控件预览"/></a></div>
                                 * </div>

                                 *  @class pullrefresh

                                 *  @param {object} [option]
                                 *     @param {string} [option.id] 控件的ID,需要满足固定的结构 
                                 *  @param {string} [option.uid] 1.6.6新增，实例的自定义唯一名字，用于命令式调用
                                 *     @param {boolean} [option.autoLoad] 是否第一次先执行加载 true | false 
                                 *     @param {boolean} [option.lastUpdated] 是否显示更新的时间提醒 
                                 *     @param {number} [option.distance] 默认100,下拉的距离超过才会触发事件

                                 *     @param {boolean} [option.stopPropagation] 1.5.5新增, 默认 false | true 阻止触发外部滑动

                                ui.pullrefresh = function (option) {
                            
                                    //一个刷新要用到的默认结构
                                    var childrenTop = "." + ui.prefix("scroll-head"),
                                        childrenMain = "." + ui.prefix("scroll-main"),
                                        childrenBottom = "." + ui.prefix("scroll-foot");
                            
                                    var config = {
                            
                                        id: "",
                                        uid: "",
                                        stopHandle: "",
                                        childrenTop: childrenTop,
                                        childrenMain: childrenMain,
                                        handleMove: "main",
                                        header: ".bui-page header",
                                        footer: ".bui-page footer",
                                        // 滚动的距离要大于这个数才会触发
                                        distance: 100,
                                        backDistance: 50,
                                        maxDistance: 0,
                            
                                        //默认自动加载数据
                                        autoLoad: true,
                                        lastUpdated: false,
                                        height: 0,
                                        cacheHeight: true,
                                        // 底下的文本提醒
                                        refreshTips: {
                                            start: "刷新中..",
                                            release: "松开刷新",
                                            end: "下拉刷新",
                                            fail: "刷新失败,请检查下网络再试试",
                                            success: "刷新成功"
                                        },
                                        loading: null,
                                        autoinit: true,
                                        stopPropagation: false,
                                        onBeforeTouchstart: null,
                                        onBeforeTouchmove: null,
                                        onBeforeTouchend: null,
                                        onBeforeInit: null,
                                        onBeforeRefresh: null,
                                        onInited: null,
                                        onRefresh: null
                                    }
                            
                                    //方法
                                    var module = {
                                        name: "pullrefresh",
                                        handle: {},
                                        on: on,
                                        off: off,
                                        reverse: reverse,
                                        refresh: refresh,
                                        reload: reload,
                                        setHeight: setHeight,
                                        resize: resize,
                                        fail: fail,
                                        lock: lock,
                                        status: status,
                                        unlock: unlock,
                                        destroy: destroy,
                                        widget: widget,
                                        option: options,
                                        config: param,
                                        init: init
                                    };
                                    //用于option方法的设置参数
                                    var param = module.config = $.extend(true, {}, config, ui.config.pullrefresh, option);
                            
                                    var id,
                                        $id,
                                        el,
                                        $children,
                                        $childrenTop,
                                        $childrenMain,
                                        $childrenBottom,
                                        moveY = 0,
                                        isLock = false, //锁定刷新
                                        initTime = new Date().getTime(), //计算更新的时间
                                        endText, //更新时间的文本
                                        loadUpFn,
                                        cacheHeight = 0,
                                        isLoad = false, //是否正在加载
                                        hasEventInit = false,
                                        isFail = false,
                                        loading = null;
                            
                                    //滑动兼容处理
                                    var isTouchPad = (/hp-tablet/gi).test(navigator.appVersion),
                                        hasTouch = 'ontouchstart' in window && !isTouchPad,
                                        direction = "",
                                        $moveTarget = null,
                                        lastY = 0,
                                        lastX = 0,
                                        isTouchstart = false, //解决PC一直触发mousemove事件
                                        isTouchmove = false,
                                        touch = {
                                            x1: 0,
                                            x2: 0,
                                            y1: 0,
                                            y2: 0,
                                            direction: ""
                                        };
                                    //初始化
                                    //
                                    param.autoinit && init(param);
                            
                            
                                    /**
                                     * 初始化方法,用于重新初始化结构,事件只初始化一次
                                     *  @method init
                                     *  @param {object} [option] 参数控件本身
                                     *  @chainable
                                     */
                                    function init(opt) {
                                        var option = $.extend(true, param, opt);
                            
                                        option.onBeforeInit && option.onBeforeInit.call(module, option);
                                        if (option.id) {
                                            $id = ui.obj(option.id);
                                        } else {
                                            ui.hint("pullrefresh id不能为空");
                                            return;
                                        }
                            
                                        //option获取新参数使用
                                        param = module.config = option;
                            
                                        el = $id[0];
                            
                                        //一个scroll固有的结构
                                        $childrenTop = $id.children(childrenTop);
                                        $childrenMain = $id.children(childrenMain);
                                        $childrenBottom = $id.children(childrenBottom);
                                        if (option.handleMove === "head") {
                                            $moveTarget = $childrenTop;
                                            $id.attr("handlemove", 'head');
                                        } else {
                                            $moveTarget = $childrenMain;
                                            $id.attr("handlemove", 'main');
                                        }
                                        //加载数据
                                        loadUpFn = load;
                            
                                        //获取更新时间的
                                        endText = option.refreshTips.end;
                            
                                        //创建刷新的loading
                                        if (!loading) {
                            
                                            status("end");
                            
                                            let loadingconfig = $.extend(true, {
                                                appendTo: $childrenTop,
                                                width: 20,
                                                height: 20,
                                                autoClose: false,
                                                text: endText,
                                                onlyText: true,
                                                display: "inline",
                                                autoTrigger: false,
                                                mask: false
                                            }, option.loading);
                            
                                            loading = ui.loading(loadingconfig);
                                        }
                            
                            
                                        //第一次是否先加载数据,还是等拉动刷新再加载
                                        if (option.autoLoad) {
                            
                                            isLoad = true;
                            
                                            status("start");
                            
                                            loading.start({
                                                text: option.refreshTips.start,
                                                onlyText: false
                                            })
                            
                                            loadUpFn();
                            
                                        }
                            
                                        if (!hasEventInit) {
                                            //拉动加载
                                            bind(option);
                                        }
                            
                                        //设置高度才会有滚动条
                                        setHeight(option.height);
                                        // el.addEventListener("scroll", canLoad);
                            
                            
                                        // 存储实例
                                        option.uid && ui.history.setUI && ui.history.setUI({
                                            uid: option.uid,
                                            ui: module
                                        })
                            
                                        option.onInited && option.onInited.call(module, option);
                                        return this;
                                    };
                            
                                    /**
                                     * 修改status属性
                                     *  @method status
                                     *  @since 1.8.0
                                     *  @chainable
                                     *  @example
                                        
                                        uiPullrefresh.status("start");
                            
                                     */
                                    function status(opt) {
                                        if (typeof opt === "string") {
                                            $id.attr("pullrefresh-status", opt);
                            
                                        } else {
                                            var sta = $id.attr("pullrefresh-status");
                                            return sta;
                                        }
                            
                                        return this;
                                    }
                            
                            
                                    //加载数据
                                    function load() {
                            
                                        //更新的时间
                                        initTime = new Date().getTime();
                            
                                        param.onBeforeRefresh && param.onBeforeRefresh.call(module);
                                        param.onRefresh && param.onRefresh.call(module);
                            
                                        trigger.call(module, "refresh");
                                    }
                                    //更新时间的状态提醒
                                    function timeTips() {
                            
                                        var nowTime = new Date().getTime();
                            
                                        var timeCount = nowTime - initTime;
                            
                                        var second = 1000,
                                            minute = 60 * second,
                                            hour = 60 * minute;
                            
                                        var hours = Math.floor(timeCount / hour);
                                        var minutes = Math.floor(timeCount / minute);
                                        var seconds = Math.floor(timeCount / second);
                                        var text;
                            
                                        if (hours <= 0 && minutes <= 0) {
                                            text = "刚刚更新";
                                        } else if (hours <= 0 && minutes > 0) {
                                            text = minutes + "分钟前更新";
                                        } else {
                                            text = "上次更新时间:" + (hours < 10 ? "0" + hours : hours) + ":" + (minutes < 10 ? "0" + minutes : minutes);
                                        }
                            
                                        return text;
                            
                                    }
                            
                                    //绑定事件
                                    function bind(option) {
                            
                                        if (!isLock) {
                            
                                            // 如果有传loadUp 则开启下拉刷新模式
                                            unlock();
                                        }
                            
                                        hasEventInit = true;
                                    };
                            
                                    //滚动条大于
                                    function canLoad(e) {
                            
                                        var tar = e.target;
                                        var scrollTop = tar.scrollTop || 0;
                            
                                        // 滚到底部了
                                        if (tar.scrollTop > 0) {
                                            isLoad = true;
                            
                                        } else {
                                            isLoad = false;
                                        }
                            
                                    }
                            
                                    var scrollTopNum = 0;
                            
                                    function onStart(e) {
                                        // jquery || zepto
                                        isTouchstart = true;
                                        var targetTouches = e.touches ? e.touches[0] : e;
                                        touch.x1 = targetTouches.clientX;
                                        touch.y1 = targetTouches.clientY;
                                        touch.direction = "";
                            
                                        scrollTopNum = $id.scrollTop();
                            
                                        // 位移的百分比
                                        // $id.attr("move", "0%");
                                        // 排除掉某个样式
                                        if (param.stopHandle && ui.unit.checkTargetInclude(e.target, param.stopHandle)) {
                                            isTouchstart = false;
                                            return;
                                        }
                                        //当屏幕有多个touch或者页面被缩放过，就不执行move操作
                                        if (e.touches && e.touches > 1 || e.scale && e.scale !== 1) {
                                            isTouchstart = false;
                                            return;
                                        }
                            
                                        var canTouchNext = param.onBeforeTouchstart && param.onBeforeTouchstart.call(module, e, touch);
                                        if (canTouchNext === false) {
                                            return false;
                                        }
                            
                                        endText = param.lastUpdated ? timeTips() : param.refreshTips.end;
                            
                                        trigger.call(module, "touchstart", e, touch);
                            
                                        param.handleMove === "head" && $childrenTop.css("zIndex", 10);
                            
                                        if ($(window).scrollTop() <= 0 && scrollTopNum <= 0 && param.onRefresh && !isLoad) {
                            
                                            isTouchstart = true;
                                            //  Ignored attempt to cancel a touchmove event with cancelable=false
                                            $(this).css("touch-action", "none");
                                        } else {
                                            isTouchstart = false;
                                            $childrenTop.css("zIndex", 0);
                                        }
                            
                                        // 阻止冒泡，外部的tab也不能左右滑动
                                        param.stopPropagation && e.stopPropagation();
                                    }
                            
                                    // 移动过程中只修改进度条一次
                                    var moreThanDistance = false,
                                        lessThanDistance = false;
                            
                                    var $ring = $id.find(".bui-scroll-head .bui-loading-cell");
                            
                                    function onMove(e) {
                                        // jquery || zepto
                            
                                        var targetTouches = e.touches ? e.touches[0] : e;
                                        if (!isTouchstart) {
                                            move();
                                            // $id.attr("move", "0%");
                                            return;
                                        }
                                        //当屏幕有多个touch或者页面被缩放过，就不执行move操作
                                        if (e.touches && e.touches.length > 1 || e.scale && e.scale !== 1) {
                                            move();
                                            // $id.attr("move", "0%");
                                            return;
                                        }
                                        touch.x2 = targetTouches.clientX;
                                        touch.y2 = targetTouches.clientY;
                                        var canTouchNext = param.onBeforeTouchmove && param.onBeforeTouchmove.call(module, e, touch);
                                        if (canTouchNext === false) {
                                            return false;
                                        }
                                        //第一次下拉的方向
                                        if (!touch.direction) {
                                            //方向
                                            touch.direction = ui.swipeDirection(touch.x1, touch.x2, touch.y1, touch.y2);
                            
                                            status("start");
                            
                                            //释放刷新提醒
                                            loading.show({
                                                text: param.refreshTips.start,
                                                onlyText: false
                                            }).pause();
                            
                                        }
                            
                                        trigger.call(module, "touchmove", e, touch);
                            
                                        //阻止默认事件,除了滚动条,解决红米不触发end问题
                                        if (touch.direction === "swipeleft" || touch.direction === "swiperight") {
                                            e.preventDefault();
                                        }
                            
                                        // 阻止微信的浏览器下拉事件
                                        if ((touch.y2 - touch.y1) > 0 && scrollTopNum <= 0) {
                                            e.preventDefault();
                                        }
                            
                            
                                        if (touch.direction == "swipedown") {
                                            // pull
                                            trigger.call(module, "movingdown", e, touch);
                            
                                            //移动的距离
                                            moveY = Math.abs(touch.y2 - touch.y1);
                            
                                            let movedistance = moveY / 2;
                            
                                            let percent = parseFloat(moveY / param.distance).toFixed(2) * 360;
                                            // $id.attr("move", percent > 100 ? "100%" : percent + "%");
                            
                                            if (!$ring.length) {
                                                $ring = $id.find(".bui-scroll-head .bui-loading-cell");
                                            }
                                            $ring.css("transform", `rotate(${percent > 360 ? 360 : percent}deg)`);
                            
                                            //计算移动的距离
                                            move(movedistance, false)
                            
                                            if (moveY >= param.distance) {
                            
                                                //下拉提醒
                                                // loading.text(endText)
                            
                                                if (!moreThanDistance) {
                                                    status("release");
                            
                                                    //释放刷新提醒
                                                    loading.show({
                                                        text: param.refreshTips.release,
                                                        onlyText: false
                                                    }).pause();
                            
                                                    moreThanDistance = true;
                                                    lessThanDistance = false;
                                                }
                            
                                            } else {
                            
                                                //下拉提醒
                                                // loading.text(endText)
                            
                                                if (!lessThanDistance) {
                                                    //下拉提醒
                                                    loading.show({
                                                        text: endText,
                                                        onlyText: false
                                                    }).pause();
                            
                                                    lessThanDistance = true;
                                                    moreThanDistance = false;
                                                }
                                            }
                                            // 可以触发touchend
                                            isTouchmove = true;
                            
                                            // 如果有设置最大值,并且滑动的距离超过最大值,
                                            if (param.maxDistance > param.distance && moveY >= param.maxDistance) {
                            
                                                status("release");
                            
                                                loading.show({
                                                    text: param.refreshTips.start,
                                                    onlyText: false
                                                }) //.start();
                            
                                                isLoad = true;
                                                //刷新加载下一页
                                                param.onRefresh && loadUpFn();
                            
                                                // 不触发touchend
                                                isTouchmove = false;
                            
                                                moreThanDistance = false;
                                                lessThanDistance = false;
                                                //清除掉这个数据
                                                touch = {};
                            
                                                //保存最后的位置
                                                lastY = lastY + (touch.y2 - touch.y1);
                                                lastX = lastX + (touch.x2 - touch.x1);
                            
                                                touch.lastX = lastX;
                                                touch.lastY = lastY;
                                                //清空滑动缓存
                                                touch = {
                                                    x1: 0,
                                                    x2: 0,
                                                    y1: 0,
                                                    y2: 0,
                                                    direction: ""
                                                };
                            
                                                isTouchstart = false;
                                                isTouchmove = false;
                                            }
                            
                                            //解决红米监听不到touchend事件
                                            e.preventDefault();
                                            e.stopPropagation();
                                        } else if (touch.direction == "swipeup") {
                                            // pull
                                            trigger.call(module, "movingup", e, touch);
                            
                                            isTouchmove = true;
                            
                                            e.stopPropagation();
                                        }
                            
                                        loading && loading.pause();
                            
                                    }
                            
                                    function onEnd(e) {
                            
                                        trigger.call(module, "touchend", e, touch);
                            
                                        var canTouchNext = param.onBeforeTouchend && param.onBeforeTouchend.call(module, e, touch);
                                        if (canTouchNext === false) {
                                            return false;
                                        }
                                        $(this).css("touch-action", "auto");
                            
                                        if (!isTouchmove) {
                                            //清空滑动缓存
                                            touch = {
                                                x1: 0,
                                                x2: 0,
                                                y1: 0,
                                                y2: 0,
                                                direction: ""
                                            };
                            
                                            isTouchstart = false;
                                            isTouchmove = false;
                                            return;
                                        }
                            
                            
                                        if (touch.direction == "swipedown") {
                                            trigger.call(module, touch.direction, e, touch);
                            
                                            if (moveY >= param.distance) {
                            
                                                move(param.backDistance);
                                                //计算移动的距离
                                                status("start");
                            
                                                loading.show({
                                                    text: param.refreshTips.start,
                                                    onlyText: false
                                                }).start();
                            
                                                isLoad = true;
                                                //刷新加载下一页
                                                param.onRefresh && loadUpFn();
                            
                                            } else {
                            
                                                move();
                            
                                            }
                            
                                            //解决红米监听不到touchend事件
                                            e.stopPropagation();
                            
                                            moreThanDistance = false;
                                            lessThanDistance = false;
                                            //清除掉这个数据
                                            touch = {};
                                        } else if (touch.direction == "swipeup") {
                                            // pull
                                            trigger.call(module, touch.direction, e, touch);
                            
                                            e.stopPropagation();
                                        }
                            
                                        //保存最后的位置
                                        lastY = lastY + (touch.y2 - touch.y1);
                                        lastX = lastX + (touch.x2 - touch.x1);
                            
                                        touch.lastX = lastX;
                                        touch.lastY = lastY;
                                        //清空滑动缓存
                                        touch = {
                                            x1: 0,
                                            x2: 0,
                                            y1: 0,
                                            y2: 0,
                                            direction: ""
                                        };
                            
                                        isTouchstart = false;
                                        isTouchmove = false;
                            
                                        $childrenTop.css("zIndex", 0);
                            
                            
                                    }
                            
                                    // 获取对应的PC事件
                                    function getEvent() {
                            
                                        // 事件类型
                                        var eventTouch = {
                                            touchstart: "touchstart",
                                            touchmove: "touchmove",
                                            touchend: "touchend",
                                            touchcancel: "touchcancel",
                                        }
                                        if (ui.platform.isPC()) {
                                            eventTouch = {
                                                touchstart: "mousedown",
                                                touchmove: "mousemove",
                                                touchend: "mouseup",
                                                touchcancel: "mouseleave",
                                            }
                                        }
                                        return eventTouch;
                                    }
                                    /**
                                     * 锁住拖动刷新,跟unlock配合使用
                                     *  @method lock
                                     *  @param {function} [callback] 回调
                                     *  @chainable
                                     *  @example
                                     *
                                        //uiPullRefresh基于顶部例子
                                        uiPullRefresh.lock();
                            
                                     */
                                    function lock(callback) {
                                        //锁定状态
                                        isLock = true;
                            
                                        var eventTouch = getEvent();
                                        // $id.off("touchstart", onStart)
                                        //     .off("touchmove", onMove)
                                        //     .off("touchend", onEnd)
                                        //     .off("touchcancel", move);
                                        $id.off(eventTouch.touchstart, onStart)
                                            .off(eventTouch.touchmove, onMove)
                                            .off(eventTouch.touchend, onEnd)
                                            .off(eventTouch.touchcancel);
                                        //还原位置
                                        move();
                            
                                        callback && callback.call(module);
                            
                                        trigger.call(module, "lock");
                            
                                        return this;
                                    }
                            
                                    /**
                                     * 绑定拖动刷新,跟lock配合使用
                                     *  @method unlock
                                     *  @param {function} [callback] 回调
                                     *  @chainable
                                     *  @example
                                     *
                                        //uiPullRefresh基于顶部例子
                                        uiPullRefresh.unlock();
                            
                                     */
                                    function unlock(callback) {
                            
                                        var eventTouch = getEvent();
                            
                                        isLock = false;
                            
                                        $id.on(eventTouch.touchstart, onStart)
                                            .on(eventTouch.touchmove, onMove)
                                            .on(eventTouch.touchend, onEnd)
                                            .on(eventTouch.touchcancel, function (e) {
                                                isTouchstart = false;
                                                isTouchmove = false;
                            
                                                !ui.platform.isPC() && move(e);
                                            });
                                        //继续绑定
                                        // $id.on("touchstart", onStart)
                                        //     .on("touchmove", onMove)
                                        //     .on("touchend", onEnd)
                                        //     .on("touchcancel", move);
                            
                                        callback && callback.call(this);
                            
                                        trigger.call(module, "unlock");
                            
                            
                                        return this;
                                    }

                                    function move(num, bool) {
                            
                                        bool = bool == false ? false : true;
                                        num = num || 0;
                            
                                        isLoad = false;
                            
                                        var transition = bool ? "all 300ms ease-out" : "none";
                            
                                        transform(0, num + "px", transition, $moveTarget);
                                        // 修正字体重复覆盖
                                        if ($childrenBottom.length) {
                                            param.handleMove === "head" ? $childrenBottom.find(".bui-loading-text").text("") : transform(0, num + "px", transition, $childrenBottom);
                                        }
                            
                            
                                        return this;
                                    }
                            
                                    // transform 会改变定位的层级, 跟dropdown使用时,需要设置为bui-scroll-main transform:none;
                                    function transform(x, y, transition, ele) {
                                        var transition,
                                            ele = ele || $id, //jquery 对象
                                            x = x || 0,
                                            y = y || 0,
                                            xRem = x,
                                            yRem = y,
                                            xVal = String(x).indexOf("%") > -1 ? String(x) : xRem,
                                            yVal = String(y).indexOf("%") > -1 ? String(y) : yRem;
                                        if (typeof transition === "number") {
                                            transition = "all " + transition + "ms";
                                        } else {
                                            transition = transition || "all 300ms";
                                        }
                            
                                        try {
                                            // translateZ(0) 会导致中兴z802t 的下拉一次,上拉滚动加载只请求一次失效
                                            ele.css({
                                                "-webkit-transition": transition,
                                                "transition": transition,
                                                "-webkit-transform": "translate(" + xVal + "," + yVal + ")",
                                                "transform": "translate(" + xVal + "," + yVal + ")"
                                            });
                            
                                            //清除动画过渡时间,下次拉动才不会有延迟现象
                                            ele.one('webkitTransitionEnd', function () {
                            
                                                ele.css({
                                                    "-webkit-transition": "none",
                                                    "transition": "none"
                                                });
                            
                                                // loading.hide();
                            
                                                // $childrenTop.css("zIndex", 0);
                            
                                            });
                                        } catch (e) {
                                            console.log(e.message)
                                        }
                                    }
