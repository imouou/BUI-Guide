                                    /*
                                     * 添加文件, 参数的配置全部针对原生选择
                                     *  @method add
                                     *  @param {object} [option] [对象]
                                     *  @param {number} [option.quality] [图片质量,默认40]
                                     *  @param {number} [option.width] [图片宽度,默认800, targetWidth和targetHeight设置为undefined 则添加原图]
                                     *  @param {number} [option.height] [图片高度,默认800, targetWidth和targetHeight设置为undefined 则添加原图]
                                     *  @param {string} [option.from] [ 图片来源, 默认来自 photo(相册) | camera(相机) | savephoto(来自保存的相册) ]
                                     *  @param {string} [option.destinationType] [ file (返回路径值) | data (返回base64位值) ]
                                     *  @param {function} [option.onSuccess] [ 成功的回调 ]
                                     *  @param {function} [option.onFail] [ 失败的回调 ]
                                     *  @chainable
                                     *  @example
                                     *
                                        uiFileSelect.add({
                                            onSuccess: function(file){
                                                // this 指向uiFileSelect
                                                // console.log(this)
                                            }
                                        })
                            
                                     */
                                    function add(param) {
                            
                                        uploadData = null;
                                        var _self = this;
                            
                                        var params = $.extend(true, { width: 800, height: 800 }, options, param);
                            
                                        try {
                                            addPicture.call(_self, params);;
                                        } catch (e) {
                                            bui.alert(e)
                                        }
                            
                                        return this;
                                    }
                            
                                    // 添加图片类型 添加的图片会自动修改为质量40,分辨率 800*800 的jpg文件
                                    // 要显示本地图片,需要使用destinationType 设置为 data; 返回 data:image/jpeg;base64,
                                    function addPicture(params) {
                            
                                        var _self = this;
                                        var config = {};
                            
                                        switch (params.from) {
                            
                                            case "camera":
                                                plus.camera.getCamera().captureImage(function(img) {
                                                    // 缓存数据后面再上传

                                                })
                                                break;
                                            default:
                                                plus.gallery.pick(function(img) {
                            
                                                    plus.io.resolveLocalFileSystemURL(img, function(entry) {
                                                        pickimg(params, entry)
                                                            // uploadHead(s); /*上传图片*/
                                                    }, function(e) {
                                                        params.onFail && params.onFail.call(_self, e);
                                                    });
                            
                                                }, function(a) {}, {
                                                    filter: "image"
                                                })
                                                break;
                                        }
                            
                                        return this;
                                    }
                            
                                    // 选中图片并进行解析
                                    function pickimg(params, entry) {
                                        var _self = this;
                                        // 如果有传宽度高度,则对图片进行压缩处理
                                        if (params.needCompress) {
                                            params.name = entry.name;
                                            // 压缩需要绝对路径
                                            params.url = entry.toLocalURL();
                                            var onSuccess = params.onSuccess;
                                            params.onSuccess = function(newentry) {
                                                // 分析处数据
                                                var imgdata = parsing(newentry)
                                                onSuccess && onSuccess.call(_self, imgdata, fileData);
                                            }
                                            compressImage(params);
                                            return;
                                        }
                            
                                        // 分析处数据
                                        var imgdata = parsing(entry)
                                        params.onSuccess && params.onSuccess.call(_self, imgdata, fileData);
                                    }
                                    // 提取信息
                                    function parsing(entry) {
                                        var imageData = null,
                                            name = "";
                                        // 资源的地址
                                        if (typeof entry === "string") {
                                            // _doc/xxx.jpg
                                            imageData = uploadData = entry;
                                            var index = imageData.lastIndexOf("/");
                                            name = imageData.substr(index + 1);
                                        } else {
                                            imageData = uploadData = entry.toLocalURL();
                                            name = entry.name;
                                        }
                            
                                        var lastSymbol = imageData.lastIndexOf('?'),
                                            lastIndex = lastSymbol < 0 ? undefined : lastSymbol,
                                            typeName = imageData.slice(imageData.lastIndexOf('.') + 1, lastIndex),
                                            type = "image/" + typeName;
                            
                                        //如果名称不相同,则缓存数据
                                        if (!ui.array.compare(fileData, name, "name")) {
                                            var item = {
                                                name: name,
                                                data: entry,
                                                type: type
                                            };
                            
                                            fileData.push(item);
                                        }
                            
                                        return imageData;
                                    }

                                    // 图片压缩保持文件文件流的形式
                                    function compressImage(params) {
                                        var name = defaultFolder + params.name;
                                        var _self = this;
                                        try {
                                            plus.zip.compressImage({
                                                    src: params.url, //src: (String 类型 )压缩转换原始图片的路径  
                                                    dst: name, //压缩转换目标图片的路径  
                                                    width: params.width, // '500'
                                                    height: params.height, // 'auto'
                                                    quality: params.quality, //quality: (Number 类型 )压缩图片的质量.取值范围为1-100  
                                                    overwrite: true //overwrite: (Boolean 类型 )覆盖生成新文件  
                                                },
                                                function(zip) {
                                                    //页面显示图片 zip.target 
                                                    params.onSuccess && params.onSuccess.call(_self, name, fileData)
                                                },
                                                function(error) {
                                                    params.onFail && params.onFail.call(_self, name, fileData);
                                                });
                                        } catch (e) {
                                            bui.showLog("compressImage 出现了问题.")
                                        }
                            
                                    }
                            
                            
                                    /*
                                     * 把选择的图片文件转换成base64地址,可以直接展示在页面上
                                     *  @method toBase64
                                     *  @param {object} [option] []
                                     *  @param {string} [option.data] [ 通过文件选择  ]
                                     *  @param {function} [option.onSuccess] [成功的回调]
                                     *  @param {function} [option.onFail] [失败的回调]
                                     *  @chainable
                                     *  @example
                                     *
                                     *
                                     *
                                        // 展示本地图片
                                        uiFileSelect.toBase64({
                                            onSuccess: function (imgurl) {
                                                $("#id").append('<img src="'+imgurl+'" />')
                                            }
                                        });
                            
                            
                            
                                     */
                                    function toBase64(param) {
                            
                                        var param = param || {},
                                            data = param.data || uploadData,
                                            onSuccess = param.onSuccess,
                                            onFail = param.onFail;
                            
                                        // 简单判断是否是绝对路径
                                        if (data.indexOf("/storage") > -1 || data.indexOf("/var") > -1) {
                                            data = data;
                                        } else {
                                            data = plus.io.convertLocalFileSystemURL(data);
                                        }
                                        var image = new Image();
                                        image.src = data;
                                        if (param.width) {
                                            image.width = param.width;
                                        }
                                        if (param.height) {
                                            image.height = param.height;
                                        }
                            
                                        image.onload = function() {
                                            var baseurl = transformBase64(image);
                            
                                            param.onSuccess && param.onSuccess(baseurl);
                                        }
                                    }
                            
                                    function transformBase64(img) {
                                        var canvas = document.createElement("canvas");
                                        var width = img.width;
                                        var height = img.height;
                                        // calculate the width and height, constraining the proportions
                                        if (width > height) {
                                            if (width > 100) {
                                                height = Math.round(height *= 100 / width);
                                                width = 100;
                                            }
                            
                                        } else {
                                            if (height > 100) {
                                                width = Math.round(width *= 100 / height);
                                                height = 100;
                                            }
                                        }
                                        canvas.width = width; /*设置新的图片的宽度*/
                                        canvas.height = height; /*设置新的图片的长度*/
                                        var ctx = canvas.getContext("2d");
                                        ctx.drawImage(img, 0, 0, width, height); /*绘图*/
                                        var dataURL = canvas.toDataURL("image/png", 0.8);
                                        return dataURL //.replace("data:image/png;base64,", "");
                            
                                    }
                            
                            
                            
                                    /*
                                     * 删除选择的文件
                                     *  @method remove
                                     *  @param {string} name [ 要删除的图片名字带后缀名 ]
                                     *  @param {string} [key] [ 图片名字在哪个字段, 默认是 "name"]
                                     *  @chainable
                                     *  @example
                                     *
                                        uiFileSelect.remove("resize.jpg");
                            
                                     */
                                    function remove(name, key) {
                                        var key = key || "name";
                                        if (typeof name == "string") {
                                            fileData = ui.array.remove(fileData, name, key);
                                            // 删除某个选择文件后需要修改上传的文件值为新的最后一条数据;
                                            var lastData = fileData.length ? fileData[(fileData.length - 1)]["data"] : null;
                                            uploadData = lastData;
                                            return fileData;
                                        }
                            
                                        return this;
                                    }
                            
                                    /*
                                     * 清空所有选择的文件
                                     *  @method clear
                                     *  @example
                                     *
                                        uiFileSelect.clear();
                            
                                     */
                                    function clear() {
                                        fileData = [];
                                        uploadData = null;
                                        return fileData;
                                    }
                            
                                    /*
                                     * 获取所有选择的文件, 原生默认只返回最后一个选择的文件
                                     *  @method data
                                     *  @example
                                     *
                                        var data = uiFileSelect.data();
                                        // 返回 [{ name: "base64.jpg", data: null, type: "jpg" }]
                                        // 其中 data为要上传的文件值
                            
                                     */
                                    function data() {
                            
                                        return fileData;
                                    }
                            
                                    /*
                                     * 获取最后选择的文件
                                     *  @method value
                                     *  @example
                                     *
                                        var value = uiFileSelect.value();
                                        // 返回值 value为要上传的文件值, web上传跟原生上传得到的值不一样
                            
                                     */
                                    function value() {
                                        return uploadData;
                                    }
                            
                            
                                    return module;
                                };
                            
                                return ui;
