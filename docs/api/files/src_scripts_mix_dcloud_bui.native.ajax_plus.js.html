                            /**
                             * <h3>原生方法库</h3> 
                             * @module Native
                             */
                            ;
                            (function(ui, $) {
                            
                            
                                //数据请求
                                ui.native.ajax = function(option) {
                            
                                    //延时处理
                                    var def = $.Deferred();
                                    var ajaxHandle = null;
                                    //用于option方法的设置参数
                                    var params = $.extend(true, {}, option);
                            
                                    try {
                                        // 本地文件, 安卓dcloud的原生请求不支持请求本地文件,只能改为 bui.isWebapp true 才不会闪退
                                        if (params.url.indexOf("://") < 0 && (params.url.indexOf(".html") > -1 || params.url.indexOf(".json") > -1 || params.url.indexOf(".js") > -1)) {
                                            var url = "_www/" + params.url.indexOf("/") === 0 ? params.url.replace("/", "") : params.url;
                                            params.url = plus.io.convertLocalFileSystemURL(url);
                                        }
                            
                                        // 原生请求暂无 abort 阻止请求方法,所以会导致list 多次重复请求问题. 
                                        var successCallback = function(res, status) {
                            
                                            var obj;
                                            if (typeof res == 'string' && params.dataType == "json") {
                                                try {
                                                    obj = JSON.parse(res);
                                                } catch (e) {
                                                    obj = res;
                                                }
                                            } else {
                                                obj = res || {};
                                            }
                                            // 在请求后, 可以对数据进行转换
                                            obj = params.transformResponse ? (params.transformResponse.call(def, obj) || obj) : obj;
                                            params.success && params.success(obj, status, ajaxHandle);
                                            def.resolve(obj, status);
                                        };
                                        var failCallback = function(res, status) {
                                            var obj;
                                            if (typeof res == 'string' && params.dataType == "json") {
                                                try {
                                                    obj = JSON.parse(res);
                                                } catch (e) {
                                                    obj = res;
                                                }
                                            } else {
                                                obj = res || {};
                                            }
                                            params.fail && params.fail(obj, status, ajaxHandle);
                                            def.reject(obj, status);
                                        };
                            
                                        // 如果是html, 这里外部应该传null
                                        if (params.mimeType == "html" || params.dataType == "html") {
                                            params.data = null;
                                        }
                                        $.ajax({
                                            xhr: function() {
                                                return new plus.net.XMLHttpRequest();
                                            },
                                            success: successCallback,
                                            error: failCallback,
                                            url: params.url, // dcloud 本地文件的请求安卓会导致崩溃  ios 13.4 无法请求
                                            contentType: params.contentType,
                                            mimeType: params.mimeType,
                                            dataType: params.dataType,
                                            processData: params.processData,
                                            type: params.method,
                                            method: params.method,
                                            timeout: params.timeout,
                                            headers: params.headers,
                                            cache: params.cache,
                                            async: params.async,
                                            data: params.data
                                        })
                                    } catch (e) {
                                        ui.showLog("原生请求需要在 bui.isWebapp = false 状态下才有效.")
                                    }
                            
                                    return def;
                            
                                };
                            
                            
                                return ui;
                            
                            })(bui || {}, libs);
