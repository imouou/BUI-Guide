                            /**
                             * <h3>原生方法库</h3> 
                             * @module Native
                             */
                            ;
                            (function (ui, $) {
                            
                              /* link 如果要使用web请求,可以使用这样配置,可以不需要原生请求
                              // 配置全局token,使用web请求
                                app.link.getToken(function (res) {
                                  bui.config.ajax.headers = {
                                    "Authorization": "Bearer " + res.accessToken
                                  }
                                })
                               */
                            
                              //数据请求
                              ui.native.ajax = function (option) {
                            
                                //延时处理
                                var def = $.Deferred();
                                //用于option方法的设置参数
                                var params = $.extend(true, {}, option);
                            
                                // 原生请求的data必须是对象
                                params.data = typeof (params.data) == "string" ? JSON.parse(params.data) : params.data;
                            
                                params.method = params.type || params.method // 原生这里不一定是大写,跟web不太一样
                            
                                var url = params.url;
                                var ajaxOption = {
                                  method: params.method || "get",
                                  params: params.params || {},
                                  data: params.data || {},
                                  timeout: params.timeout || 10,
                                  responseType: params.responseType || "text",
                                  headers: params.headers || {}
                                };
                            
                                // 原生请求暂无 abort 阻止请求方法,所以会导致list 多次重复请求问题. 
                                var successCallback = function (res) {
                                  var result;
                                  var obj = {};
                                  if (typeof res == 'string' && params.dataType == "json") {
                                    try {
                                      result = JSON.parse(res);
                                    } catch (e) {
                                      result = res;
                                    }
                                  } else {
                                    result = res || {};
                                  }
                            
                                  //新 bt 数据层在 data
                                  if (typeof result["data"] == 'string' && params.dataType == "json") {
                                    try {
                                      obj = JSON.parse(result["data"]);
                                    } catch (e) {
                                      obj = result["data"];
                                    }
                                  } else {
                                    obj = result["data"] || {};
                                  }
                                  // 在请求后, 可以对数据进行转换
                                  obj = params.transformResponse ? (params.transformResponse.call(def, obj) || obj) : obj;
                                  obj.code = obj.code || res.status || 200 // 兼容新版ajax做法
                            
                                  params.success && params.success(obj, res.status, {});
                                  def.resolve(obj, res.status, {});
                                };
                                var failCallback = function (res) {
                            
                                  var result;
                                  var obj = {};
                                  if (typeof res == 'string') {
                                    try {
                                      result = JSON.parse(res);
                                    } catch (e) {
                                      result = res;
                                    }
                                  } else {
                                    result = res || {};
                                  }
                            
                                  //bt 包多一层returnValue
                                  if (typeof result["returnValue"] == 'string') {
                                    try {
                                      obj = JSON.parse(result["returnValue"]);
                                    } catch (e) {
                                      obj = result["returnValue"];
                                    }
                                  } else {
                                    obj = result["returnValue"] || {};
                                  }
                                  obj.code = obj.code || res.status || 200 // 兼容新版ajax做法
                                  params.fail && params.fail(obj, res.status, {});
                            
                                  def.reject(obj, res.status, {});
                                }
                            
                                // 默认忽略SSL和信任host
                                params.trustAllServer = params.trustAllServer || true;
                                if (params.trustAllServer) {
                                  var execCallback = function () {
                                    cordova.plugin.http.sendRequest(url, ajaxOption, successCallback, failCallback);
                                  };
                                  cordova.plugin.http.setServerTrustMode("nocheck", execCallback, failCallback);
                                } else {
                                  cordova.plugin.http.sendRequest(url, ajaxOption, successCallback, failCallback);
                                }
                            
                                return def;
                            
                              };
                            
                            
                              return ui;
                            
                            })(window.bui || {}, window.libs);
