                            /**
                             * UI控件库
                             * @module UI
                             */

                            (function (ui, $) {
                                "use strict";
                                /**
                                 * <div class="oui-fluid">
                                 *   <div class="span8">
                                 *     <h2>绑定按钮跳转</h2>
                                 *     <p>一个页面只需要静态绑定一次,在最外层的ID,找到页面所有要跳转的按钮,获取页面参数请查看 {{#crossLink "bui.getPageParams"}}{{/crossLink}}</p>
                                 *     <h3>预览地址: <a href="http://www.easybui.com/demo/index.html#pages/ui_controls/bui.btn.html" target="_blank">demo</a></h3>
                                 *     <h3>方法说明:</h3>
                                 * {{#crossLink "bui.btn/load"}}{{/crossLink}}: 页面跳转,支持单个页面跟容器内的多个按钮 <br>
                                 * {{#crossLink "bui.btn/submit"}}{{/crossLink}}: 提交数据<br>
                                 *   </div>
                                 *   <div class="span4"><a href="http://www.easybui.com/demo/index.html#pages/ui_controls/bui.btn.html" target="_blank"><img src="http://www.easybui.com/static/images/controls/bui-btn_low.gif" alt="控件预览"/></a></div>
                                 * </div>
                                 *  @namespace bui
                                 *  @class btn
                                 *  @constructor
                                 *  @param {object} option
                                 *  @param {string} option.id 页面id
                                 *  @param {string} [option.handle] 监听的按钮
                                 *  @param {string} [option.beforeCallback]  1.5.4新增, 点击前执行,如果返回fasle, 则不执行跳转或者提交事件 
                                 *  @example
                                 *
                                 *
                                 *  html:
                                 *
                                   <div id="page">
                                       <div class="bui-btn" href="btn.html">首页</div>
                                       <div class="bui-btn" href="btn.html?id=222">新闻</div>
                                   </div>
                                *  js:
                                
                                    var uiBtn = bui.btn({
                                        id:"#page",
                                        handle: ".bui-btn"  // 绑定多个样式名 ".bui-btn,.submit"
                                    })
                            
                                    // 所以按钮有href时点击会跳转
                                    uiBtn.load();
                                 *
                                 */
                                ui.btn = function (option) {
                            
                                    var $id,
                                        opt = {},
                                        progress,
                                        replace,
                                        $children;
                            
                                    var module = {
                                        load: load,
                                        submit: submit,
                                        click: click,
                                        off: off
                                    };
                            
                            
                                    //初始化
                                    init(option);
                            
                                    function init(option) {
                            
                                        //id 传的是否是json ,这里统一了控件的初始化方式,快速初始化则不用
                                        if (option && ui.typeof(option) === "object") {
                            
                                            opt.id = option["id"] || "";
                                            opt.handle = option["handle"] || "";
                                            opt.progress = "progress" in option ? option["progress"] : false;
                                            opt.replace = "replace" in option ? option["replace"] : false;
                                            opt.timeout = option["timeout"] || 3000;
                                            opt.beforeCallback = option["beforeCallback"] || null;
                            
                                            //初始化选择器, ui.obj 选择器在 路由的 firstAnimate: true 的情况不能找到
                                            $id = $(opt["id"]);
                                        } else if (typeof (option) === "string") {
                                            // 如果传的是字符串,则可能会有2个参数
                                            opt.id = option || "";
                                            opt.handle = "";
                                            opt.progress = false;
                                            opt.replace = false;
                                            opt.timeout = 3000;
                                            opt.beforeCallback = null;
                            
                                            //初始化选择器, ui.obj 选择器在 路由的 firstAnimate: true 的情况不能找到
                                            $id = ui.obj(opt["id"]);
                            
                                        }
                            
                                        $children = opt["handle"];
                                        progress = opt.progress;
                                        replace = opt.replace;
                            
                                        return this;
                                    }
                            
                                    /**
                                     * 页面跳转,跟 {{#crossLink "bui.load"}}{{/crossLink}} 的区别在于,通过按钮绑定跳转会增加一层防止快速点击加载2次的问题
                                     * 注意: 这个默认跳转会阻止默认事件,a, input的checkbox label的for 等等,包在btn元素里面都会失效
                                     * 页面如何接收参数请查看 {{#crossLink "bui.getPageParams"}}{{/crossLink}}
                                     *  @method load
                                     *  @example
                                     *
                                     *  示例: 绑定页面所有有href的按钮的跳转(多页一个页面只需要初始化一次,单页只需要在首页初始化一次)
                                     *
                                     *  html:
                                     *
                                       <div id="page">
                                           <div class="bui-btn" href="btn.html">首页</div>
                                           <div class="bui-btn" href="btn.html?id=123">新闻</div>
                                       </div>
                                    *  js:
                                    *
                                        bui.btn({
                                            id:"#page",
                                            handle: ".bui-btn"  // 绑定多个 ".bui-btn,.submit,a"
                                        }).load();
                            
                                     */
                                    function load(url) {
                            
                                        //进度条参数是否提前
                                        var url = url;
                            
                                        //重新赋值
                                        var timeout,
                                            loadingTimeout;
                            
                                        var timeEnd = 0;
                                        $id.off("click.bui", $children).on("click.bui", $children, function (e) {
                                            var $this = $(this);
                                            var beforeLoad = opt.beforeCallback || opt.beforeLoad;
                                            var canClick = beforeLoad && beforeLoad.call(module, e);
                                            if (canClick === false) {
                                                // 阻止默认a标签绑定,微信对a标签有特殊处理
                                                e.preventDefault();
                                                e.stopPropagation();
                                                return false;
                                            }
                            
                                            //兼容以前的写法
                                            var href = url || $this.attr("href"),
                                                target = $this.attr("target"),
                                                iframe = $this.attr("iframe") == "true" ? true : false,
                                                firstAnimate = $this.attr("firstAnimate") == "true" ? true : false,
                                                attrDisabled = $this.attr("disabled"),
                                                disabled = $this.hasClass("disabled") || attrDisabled == "" || attrDisabled == "true" || attrDisabled == "disabled";
                                            var param3 = {};
                            
                                            if (!href || disabled) {
                                                return;
                                            }
                                            if (href && href.indexOf("javascript:") > -1) {
                                                return;
                                            }
                            
                            
                                            //传过去的参数 废弃
                                            var paramStr = $this.attr("param") || "";
                                            var param2 = (paramStr && paramStr.indexOf("{") > -1 && paramStr.indexOf("}") > -1) ? JSON.parse($this.attr("param")) : {};
                            
                                            //合并参数
                                            var paramObj = param2;
                                            var loading;
                            
                                            var progress2 = $this.attr("progress") ? ($this.attr("progress") == "false" ? false : true) : progress;
                            
                                            //如果需要跳转加上进度条
                                            if (progress2) {
                                                loading = ui.loading({
                                                    autoTrigger: true,
                                                    display: "block",
                                                    opacity: 0,
                                                    timeout: opt.timeout
                                                });
                                            }
                            
                                            //解决BT的快速点击导致页面加载两次, 如果在滑动的时候,继续点击,还是会触发两次,这个需要在原生层面解决.
                                            var timeNow = +new Date();
                                            if (timeNow - timeEnd < 350) {
                                                return false;
                                            }
                                            timeEnd = timeNow;
                            
                                            //传参数
                                            var params = {
                                                url: href,
                                                param: paramObj,
                                                replace: replace,
                                                firstAnimate: firstAnimate,
                                                iframe: iframe
                                            };
                                            //网址是http://开头
                                            if (target == "_blank") {
                                                ui.run({
                                                    id: href,
                                                    data: paramObj
                                                })
                                            } else {
                                                //页面跳转
                                                ui.load(params);
                                            }
                                            // 阻止默认a标签绑定,微信对a标签有特殊处理
                                            e.preventDefault();
                                            e.stopPropagation();
                                            //阻止原有的事件跳转
                                            return false;
                                        })
                            
                                    };
                            
                            
                                    /**
                                     * 绑定按钮提交数据,提交数据成功以后需要在回调里面关闭进度条
                                     *  @method submit
                                     *  @param {function} callback 提交的回调
                                     *  @param {object} param  loading 的参数修改
                                     *  @example
                            
                                        bui.btn("#btn").submit(function(loading){
                                             //关闭进度条
                                             loading.stop();
                                         });
                            
                                     */
                                    function submit(callback, param) {
                            
                                        var param = param || {};
                                        //创建loading
                                        var loading = null;
                                        if (opt["id"] && opt["handle"]) {
                                            $id.off("click.bui", $children).on("click.bui", $children, bindSubmit);
                                        } else {
                                            $id.click(bindSubmit);
                                        }
                            
                                        function bindSubmit(e) {
                            
                                            var canClick = opt.beforeCallback && opt.beforeCallback.call(this, e);
                                            if (canClick === false) {
                                                // 阻止默认a标签绑定,微信对a标签有特殊处理
                                                e.preventDefault();
                                                e.stopPropagation();
                                                return false;
                                            }
                                            var _self = this;
                                            var myBackground = $(_self).css("background-color");
                            
                                            if ($(this).hasClass("disabled") || $(this).attr("disabled") == "true") {
                                                return;
                                            }
                            
                                            //覆盖原来的颜色
                                            var background = myBackground == "none" ? "#fff" : myBackground;
                            
                                            param.appendTo = _self;
                                            param.background = param.background || background;
                                            param.display = param.display || "inline";
                                            param.width = param.width || 15;
                                            param.height = param.height || 15;
                                            param.text = param.text || "加载中";
                                            param.autoClose = param.autoClose == true ? true : false;
                                            param.autoTrigger = param.autoTrigger == false ? false : true;
                            
                                            //创建loading
                                            if (loading) {
                                                if (loading.isLoading()) {
                                                    return;
                                                } else {
                                                    loading.show();
                                                }
                                            } else {
                                                loading = ui.loading(param);
                                            }
                            
                                            // 在回调中手动关闭进度 loading.stop();
                                            return callback && callback.call(_self, loading, e);
                            
                                        }
                                    };
                            
                            
                                    /**
                                     * 绑定按钮提交数据,提交数据成功以后需要在回调里面关闭进度条
                                     *  @method click
                                     *  @since 1.5.4
                                     *  @param {function} callback 提交的回调
                                     *  @example
                            
                                        bui.btn("#btn").click(function(e){
                                             //关闭进度条
                                             loading.stop();
                                         });
                            
                                     */
                                    function click(callback) {
                                        var timeEnd = 0;
                                        if (opt["id"] && opt["handle"]) {
                                            $id.off("click.bui", $children).on("click.bui", $children, quickclick);
                                        } else {
                                            $id.click(quickclick);
                                        }
                            
                                        function quickclick(e) {
                            
                                            var canClick = opt.beforeCallback && opt.beforeCallback.call(this, e);
                                            if (canClick === false) {
                                                // 阻止默认a标签绑定,微信对a标签有特殊处理
                                                e.preventDefault();
                                                e.stopPropagation();
                                                return false;
                                            }
                                            var _self = this;
                            
                                            if ($(this).hasClass("disabled") || $(this).attr("disabled") == "true") {
                                                return;
                                            }
                            
                                            //解决BT的快速点击导致页面加载两次, 如果在滑动的时候,继续点击,还是会触发两次,这个需要在原生层面解决.
                                            var timeNow = +new Date();
                                            if (timeNow - timeEnd < 350) {
                                                return false;
                                            }
                                            timeEnd = timeNow;
                            
                                            return callback && callback.call(_self, e);
                            
                                        }
                                    };
                            
                                    /**
                                     * 取消事件
                                     *  @method off
                                     *  @example
                            
                                        uiBtn.off();
                            
                                     */
                                    function off() {
                            
                                        $id.off("click.bui", $children);
                            
                                        return this;
                                    }
                            
                                    return module
                                };
                            
                                return ui;
                            
                            })(window.bui || {}, window.libs);
