                                        // 初始化, 如果有多个页面有这个控件, id选择器可以通过当前页查找: id:".page-cart .bui-number"

                                ui.number = function (option) {
                            
                                    //默认配置
                                    var config = {
                                        parentId: ".bui-page",
                                        id: ".bui-number",
                                        uid: "",
                                        type: "text",
                                        min: 0,
                                        max: 100,
                                        step: 1,
                                        value: 1,
                                        disabled: false,
                                        render: true,
                                        tips: false,
                                        autocheck: true,
                                        decimal: 1,
                                        name: "",
                                        prev: ".bui-number-prev",
                                        input: "input",
                                        next: ".bui-number-next",
                                        onInput: null,
                                        autoinit: true,
                                        inited: null,
                                        onBeforeInit: null,
                                        onInited: null,
                                        onChange: null,
                                        callback: null
                                    }
                            
                                    //方法
                                    var module = {
                                        name: "number",
                                        handle: {},
                                        on: on,
                                        off: off,
                                        $el: selector,
                                        disabled: disabled,
                                        enable: enable,
                                        disablePrev: disablePrev,
                                        disableNext: disableNext,
                                        enablePrev: enablePrev,
                                        enableNext: enableNext,
                                        reset: reset,
                                        value: setValue,
                                        values: setValues,
                                        prev: prev,
                                        next: next,
                                        destroy: destroy,
                                        widget: widget,
                                        option: options,
                                        config: param,
                                        init: init
                                    };
                                    //用于option方法的设置参数
                                    var param = module.config = $.extend(true, {}, config, ui.config.number, option);
                            
                                    var id,
                                        max,
                                        min,
                                        step,
                                        //控件自定义模板可以自由绑定想增加或者减少的按钮名称
                                        $prev,
                                        $next,
                                        $input,
                                        hasEventInit = false,
                                        hasId = false,
                                        isDestroy = false,
                                        $pid = null,
                                        $id;
                            
                                    // 初始化
                                    param.autoinit && init(param);
                            
                            
                                    /**
                                     * 初始化方法,用于重新初始化结构,事件只初始化一次
                                     *  @method init
                                     *  @param {object} [option] 参数控件本身
                                     *  @chainable
                                     */
                                    function init(opt) {
                                        var option = $.extend(true, param, opt);
                            
                                        option.onBeforeInit && option.onBeforeInit.call(module, option);
                            
                                        isDestroy = false;
                            
                                        max = option.max;
                                        min = option.min;
                                        step = option.step;
                            
                                        // 如果parentId ==历史记录的id，ui.obj 会找不到,需要使用 $ 选择器
                                        $pid = opt.parentId && ui.obj(option.parentId).length ? ui.obj(option.parentId) : $(option.parentId);
                            
                                        if (typeof option.id === "object") {
                                            $id = ui.obj(option.id);
                                            hasId = false;
                                        } else if (typeof option.id === "string" && option.id.indexOf("#") === 0 || (opt.id && !opt.parentId)) {
                                            $id = ui.obj(option.id);
                                            hasId = true;
                                        } else {
                                            $id = $pid.find(option.id);
                                            hasId = false;
                                        }
                            
                                        //option获取新参数使用
                                        param = module.config = option;
                            
                                        //render = true,就会动态创建模板
                                        if (option.render) {
                                            render(option);
                                        }
                            
                                        $prev = $id.children(option.prev);
                                        $next = $id.children(option.next);
                                        $input = $id.children(option.input);
                            
                                        if (!hasEventInit) {
                                            //绑定按钮事件
                                            bind(option);
                                        }
                            
                                        let vals = [];
                                        if ($id.length) {
                                            $id.each((i, item) => {
                                                let val = item.getAttribute("value") || option.value;
                                                if (!item.id) {
                                                    item.id = ui.guid();
                                                }
                            
                                                vals.push({
                                                    id: item.id,
                                                    value: val
                                                })
                                            })
                                        }
                            
                            
                                        //禁止修改
                                        if (option.disabled) {
                                            $input.attr('disabled', 'disabled');
                                        }
                            
                            
                                        if (typeof option.value == "object") {
                                            setValues(option.value)
                                        } else if (vals.length) {
                                            setValues(vals)
                                        } else {
                                            //设置初始值
                                            set(option.value);
                                        }
                            
                                        option.target = $input;
                                        option.value = option.value;
                            
                            
                                        // 存储实例
                                        option.uid && ui.history.setUI && ui.history.setUI({
                                            uid: option.uid,
                                            ui: module
                                        })
                            
                                        option.onInited && option.onInited.call(module, option);
                                        // 废弃,
                                        option.inited && option.inited.call(module, option);
                            
                                        return this;
                                    }
                            
                                    // 检测是否在区间
                                    function checkBetween(val, min, max) {
                                        min = min || param.min;
                                        max = max || param.max;
                                        if (val < min) {
                                            val = min;
                                        } else if (val > max) {
                                            val = max
                                        }
                            
                                        return val;
                                    }
                            
                                    //模板
                                    function template(option) {
                            
                                        var html = '';
                                        html += '    <div class="bui-number-prev"><i class="' + ui.config.icon.minus + '"></i></div>';
                                        html += '    <input class="bui-number-input" type="' + option.type + '" name="' + (option.name || ui.guid()) + '" value="' + option.value + '" min="' + option.min + '" max="' + option.max + '" step="' + option.step + '"/>';
                                        html += '    <div class="bui-number-next"><i class="' + ui.config.icon.plus + '"></i></div>';
                            
                                        return html;
                                    };
                            
                                    //渲染
                                    function render(option) {
                            
                                        var tpl = template(option);
                            
                                        $id.html(tpl);
                            
                                        return this;
                                    };
                            
                                    // 返回控件自身的选择器
                                    function selector(id) {
                                        return ui.selector.call($id, id);
                                    }
                            
                            
                            
                                    //绑定事件
                                    function bind(option) {
                                        var target = ".bui-number" + " " + option.input;
                                        var targetPrev = ".bui-number" + " " + option.prev;
                                        var targetNext = ".bui-number" + " " + option.next;
                            
                                        var onInput = function (e) {
                                            var val = $(this).val();
                            
                                            // if( /^[0-9]*$/i.test(val) ){
                                            //     set.call(this, val );
                                            // }
                                            e.value = checkBetween(val);
                                            option.onInput && option.onInput.call(module, e);
                                            e.stopPropagation();
                            
                                        };
                            
                                        var onChange = function (e) {
                            
                                            var val = option.autocheck && isNaN(parseFloat($(this).val(), 10)) ? 0 : parseFloat($(this).val(), 10);
                            
                                            if (/^[-\\+]?([1-9]\d*\.?\d*)|(0\.\d*[1-9])$/i.test(val)) {
                                                set.call(this, checkBetween(val));
                                            }
                                            e.value = checkBetween(val);
                            
                                            // option.onChange && option.onChange.call(module, e);
                                            e.stopPropagation();
                            
                                        }
                            
                                        var onPrev = function (e) {
                                            var input = $(this).next(param.input);
                            
                                            e.value = input.val();
                                            var canPrev = option.callback && option.callback.call(module, e);
                            
                                            if (canPrev === false || $(this).hasClass("disabled")) {
                                                return;
                                            }
                                            prev.call(input, e);
                            
                                            e.preventDefault();
                                            e.stopPropagation();
                                        }
                            
                                        var onNext = function (e) {
                                            var input = $(this).prev(param.input);
                            
                                            e.value = input.val();
                            
                                            var canNext = option.callback && option.callback.call(module, e);
                            
                                            if (canNext === false || $(this).hasClass("disabled")) {
                                                return;
                                            }
                                            next.call(input, e);
                            
                                            e.preventDefault();
                                            e.stopPropagation();
                                        }
                            
                                        // 如果通过id初始化, 事件绑定在id上
                                        if (hasId) {
                                            //检测输入的值是否有效
                                            option.onInput && $id.off('input', option.input).on('input', option.input, ui.unit.debounce(onInput, 400))
                            
                                            //检测输入的值是否有效
                                            $id.off('change', option.input).on('change', option.input, onChange);
                            
                                            //点击减少
                                            $id.off('click.bui', option.prev).on('click.bui', option.prev, onPrev);
                            
                                            //点击增加
                                            $id.off('click.bui', option.next).on('click.bui', option.next, onNext);
                            
                                        } else {
                                            //检测输入的值是否有效
                                            option.onInput && $pid.off('input', target).on('input', target, ui.unit.debounce(onInput, 400))
                            
                                            //检测输入的值是否有效
                                            $pid.off('change', target).on('change', target, onChange)
                            
                                            //点击减少
                                            $pid.off('click.bui', targetPrev).on('click.bui', targetPrev, onPrev)
                            
                                            //点击增加
                                            $pid.off('click.bui', targetNext).on('click.bui', targetNext, onNext)
                            
                                        }
                            
                                        hasEventInit = true;
                            
                                        return this;
                                    }
                            
                                    //获取值
                                    function get() {
                            
                                        getSelectors();
                            
                                        var $inputs = this && ui.typeof(this) === "htmldivelement" ? $(this) : $input;
                            
                                        var value = parseFloat($inputs.val(), 10);
                            
                                        return value;
                                    };
                            
                                    //设置值
                                    function set(num) {
                            
                                        getSelectors();
                            
                                        var $inputs = (this && ui.typeof(this) === "object") ? $input : $(this);
                            
                                        Array.prototype.slice.call($inputs).forEach(function (item, i) {
                            
                                            var max = item.getAttribute("max") || item.parentElement.getAttribute("data-max") || param.max;
                                            var min = item.getAttribute("min") || item.parentElement.getAttribute("data-min") || param.min;
                            
                                            var value = param.autocheck ? checkValue(parseFloat(num), parseFloat(min), parseFloat(max)) : parseFloat(num);
                            
                                            item.value = value;
                            
                                            var target = this && ui.typeof(this) === "htmldivelement" ? $input : item;
                            
                                            // 触发b-model 的绑定
                                            $inputs.trigger("input");
                                            // 输入的时候才触发
                                            trigger.call(module, "change", {
                                                target: item,
                                                value: num
                                            });
                                            param.onChange && param.onChange.call(module, {
                                                target: item,
                                                value: num
                                            });
                            
                                        })
                            
                                        return this;
                                    };
                            
                                    function checkValue(num, mins, maxs) {
                                        var value = num || 0;
                            
                                        if (value > maxs) {
                                            value = maxs;
                                        }
                            
                                        if (value < mins) {
                                            value = mins;
                                        }
                            
                                        if (value && value >= mins && value <= maxs) {
                                            value = value;
                                        }
                            
                                        return value;
                                    }
                            
                                    /**
                                     * 设置或者获取值
                                     *  @method value
                                     *  @chainable
                                     *  @param {string} [num] 设置值
                                     *  @example
                            
                                        //设置值
                                        uiNumber.value("110");
                            
                                        //获取值
                                        uiNumber.value();
                            
                                     */
                                    function setValue(num) {
                                        var val = 0;
                            
                                        if (typeof num !== "undefined") {
                                            set.call(this, num);
                            
                                            val = num;
                            
                                        } else {
                                            val = get.call(this);
                            
                                        }
                                        return val;
                                    }
                            
                                    /**
                                     * 重置数据
                                     *  @method reset
                                     *  @since 1.6.2
                                     *  @chainable
                                     *  @example
                            
                                        //重置
                                        uiNumber.reset();
                                     */
                                    function reset() {
                            
                                        var num = param.value || param.min;
                                        set.call(this, num);
                            
                            
                                        return num;
                                    }
                            
                                    /**
                                     * 获取设置多个值
                                     *  @method values
                                     *  @chainable
                                     *  @param {string} [num] 设置值 [{id:"",value:""}]
                                     *  @return {array} [ 返回 [{id:"",value:""}] ]
                                     *  @example
                            
                                        uiNumber.values();
                            
                                     */
                                    function setValues(val) {
                                        if ($id.length < 1) {
                                            $id = $pid.find(param.id);
                                        }
                            
                                        if (val && ui.typeof(val) === "array") {
                                            Array.prototype.slice.call($id).forEach(function (item, i) {
                                                var name = item.id || item.getAttribute("name") || "";
                                                var $numInput = item.querySelector("input");
                            
                                                if (name == val[i]["id"]) {
                            
                                                    // 用于渲染的时候赋值
                                                    $numInput.value = val[i]["value"];
                                                    $numInput.min = val[i]["min"] || param.min;
                                                    $numInput.max = val[i]["max"] || param.max;
                                                    $numInput.step = val[i]["step"] || param.step;
                            
                                                }
                            
                                            })
                                            return this;
                                        } else {
                                            var vals = [];
                                            Array.prototype.slice.call($id).forEach(function (item, i) {
                                                var name = item.getAttribute("name") || item.id || "";
                                                var $numInput = item.querySelector("input");
                                                var val = $numInput.value;
                                                var min = $numInput.getAttribute("min");
                                                var max = $numInput.getAttribute("max");
                            
                                                vals.push({
                                                    id: name,
                                                    value: val,
                                                    name: name,
                                                    min: min,
                                                    max: max
                                                })
                                            })
                                        }
                            
                                        return vals;
                                    }
                            
                                    /**
                                     * 禁止输入
                                     *  @method disabled
                                     *  @chainable
                                     *  @param {boolean} [默认： true禁止｜false]
                                     *  @example
                            
                                        //获取值
                                        uiNumber.disabled();
                            
                                     */
                                    function disabled(bool) {
                            
                                        getSelectors();
                            
                                        var $inputs = this && ui.typeof(this) === "htmldivelement" ? $(this) : $input;
                            
                                        var bool = bool == false ? false : true;
                            
                                        if (bool) {
                            
                                            $inputs.attr('disabled', 'disabled');
                                        } else {
                                            $inputs.removeAttr('disabled');
                                        }
                                        return this;
                                    }
                            
                                    /**
                                     * 1.7.0 允许输入
                                     *  @method enable
                                     *  @chainable
                                     *  @example
                            
                                        uiNumber.enable();
                            
                                     */
                                    function enable(bool) {
                            
                                        getSelectors();
                            
                                        var $inputs = this && ui.typeof(this) === "htmldivelement" ? $(this) : $input;
                            
                                        $inputs.removeAttr('disabled');
                            
                                        return this;
                                    }
                            
                                    /**
                                     * 1.7.0 禁止递减
                                     *  @method disablePrev
                                     *  @chainable
                                     *  @example
                            
                                        uiNumber.disablePrev();
                            
                                     */
                                    function disablePrev() {
                                        $prev.addClass("disabled");
                                        return this;
                                    }
                            
                                    /**
                                     * 1.7.0 禁止递增
                                     *  @method disableNext
                                     *  @chainable
                                     *  @example
                            
                                        uiNumber.disableNext();
                            
                                     */
                                    function disableNext() {
                                        $next.addClass("disabled");
                                        return this;
                                    }
                                    /**
                                     * 1.7.0 允许点击递减
                                     *  @method enablePrev
                                     *  @chainable
                                     *  @example
                            
                                        uiNumber.enablePrev();
                            
                                     */
                                    function enablePrev() {
                                        $prev.removeClass("disabled");
                                        return this;
                                    }
                                    /**
                                     * 1.7.0 允许点击递增
                                     *  @method enableNext
                                     *  @chainable
                                     *  @example
                            
                                        uiNumber.enableNext();
                            
                                     */
                                    function enableNext() {
                                        $next.removeClass("disabled");
                                        return this;
                                    }
                            
                                    function getSelectors() {
                                        // if ($input.length < 1) {
                            
                                        if (typeof param.id === "object") {
                                            $id = ui.obj(param.id);
                                            hasId = false;
                                        } else if (typeof param.id === "string" && param.id.indexOf("#") === 0) {
                                            $id = ui.obj(param.id);
                                            hasId = true;
                                        } else {
                                            $id = $pid.find(param.id);
                                            hasId = false;
                                        }
                            
                                        $prev = $id.children(param.prev);
                                        $next = $id.children(param.next);
                                        $input = $id.children(param.input);
                                        // }
                                    }
                            
                                    /**
                                     * 减少值
                                     *  @method prev
                                     *  @chainable
                                     *  @example
                            
                                        //减少
                                        uiNumber.prev();
                            
                                     */
                                    function prev() {
                                        getSelectors();
                            
                                        var $inputs = this && ui.typeof(this) === "htmldivelement" ? $(this) : (event ? $(event.target).parent().next() : $input);
                            
                                        var data = $inputs.val();
                                        var num = +data;
                                        num = num - step;
                            
                                        var total = /^-?\d*\.\d+$/.test(step) ? parseFloat(num).toFixed(param.decimal) : num;
                                        set.call($inputs, total);
                            
                                        trigger.call(module, "prev", total);
                            
                                        return this;
                            
                                    }
                                    /**
                                     * 增加值
                                     *  @method next
                                     *  @chainable
                                     *  @example
                            
                                        //增加
                                        uiNumber.next();
                            
                                     */
                                    function next() {
                                        getSelectors();
                            
                                        var $inputs = this && ui.typeof(this) === "htmldivelement" ? $(this) : (event ? $(event.target).parent().prev() : $input);
                            
                                        var data = $inputs.val();
                            
                                        var num = +data + (+step);
                            
                                        var total = /^-?\d*\.\d+$/.test(step) ? parseFloat(num).toFixed(param.decimal) : parseFloat(num);
                            
                                        set.call($inputs[0], total);
                                        trigger.call(module, "next", total);
                            
                                        return this;
                                    }
                            
                                    /**
                                     * [销毁控件]
                                     *  @method destroy
                                     *  @since 1.4.2
                                     *  @param {boolean} [bool]  默认: false 销毁部分 | true 销毁全部   
                                     *  @example
                            
                                        //销毁
                                        uiNumber.destroy();
                            
                                     */
                                    function destroy(bool) {
                                        var bool = bool == true ? true : false;
                                        // param.beforeDestroy && param.beforeDestroy.call(module);
                                        $id = $pid.find(param.id);
                                        if ($id) {
                                            $id.off("click.bui");
                                            $id.off("input");
                                            bool && $id.remove();
                                        }
                            
                                        off("prev");
                                        off("next");
                                        off("change");
                            
                                        isDestroy = true;
                            
                                        // param.destroyed && param.destroyed.call(module);
                                    }
                                    /**
                                     * 获取依赖的控件
                                     *  @method widget
                                     *  @param {string} [name]  依赖控件名

                                     */
                                    function widget(name) {

                                    /**

                                     *  @param { string | object } [key]  不传则获取所有参数, 类型为string,没有第2个参数则获取某个参数 
                                     *  @param { string | number | boolean | function } [value]  设置参数的时候要传,设置多个参数不用传,获取参数的时候也不用传 
                                     *  @chainable
                                     *  @example
                            
                            
                                        //获取所有参数
                                         //获取所有参数
                                        var option = uiNumber.option();
                            
                                        //获取某个参数
                                        var id = uiNumber.option( "id" );
                            
                                        //修改一个参数
                                        uiNumber.option( "min",10 );
                            
                                        //修改多个参数

                                    function options(key, value) {

                                    }
                                    /**
                                     * 为控件绑定事件
                                     *  @event on
                                     *  @since 1.3.0

                                        uiNumber.on("change",function () {
                                            // 点击的菜单
                                            console.log(this);
                                        });
                            
                            
                                     */
                                    function on(type, callback) {
                                        ui.on.apply(module, arguments);
                                        return this;
                                    }
                            
                                    /**
                                     * 为控件取消绑定事件
                                     *  @event off
                                     *  @since 1.3.0
                                     *  @param {string} [type]  事件类型: "prev" | "next" | "change" 符合条件时触发改变   
                                     *  @param {function} [callback]  绑定的事件, this 为当前点击的菜单 
                                     *  @example
                            
                                        uiNumber.off("change");
                            
                            
                                     */
                                    function off(type, callback) {
                                        ui.off.apply(module, arguments);
                                        return this;
                            
                                    }

                                     * 触发自定义事件
                                     */
                                    function trigger(type) {
                                        //点击事件本身,或者为空,避免循环引用
                                        module.self = (this == window || this && this.hasOwnProperty("widget")) ? null : this;
                            
                                        ui.trigger.apply(module, arguments);
                                    }
                            
                                    return module;
                                };
