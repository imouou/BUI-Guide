                                    function getFolder(param) {
                            
                                        var config = {
                                            folderName: defaultFolder,
                                            root: false,
                                            create: false,
                                            msg: "",
                                            param: {
                                                create: false,
                                                exclusive: false
                                            },

                                            // 动作

                                        if (params.create) {
                                            action = "创建 ";
                                            params.param.create = true;
                                        } else {
                                            action = "获取 ";
                                            params.param.create = false;
                                        }
                            
                                        // 转换标准文件名才能创建成功
                                        var name = checkPath(params.folderName);
                                        if (params.root) {
                                            name = name;
                                        } else {
                                            name = defaultFolder + "/" + name;
                                        }
                            
                                        //文件系统准备完毕
                                        _fileSystem.then(function (fileSystem) {
                                            //创建文件夹
                                            fileSystem.root.getDirectory(name, params.param, function (dirEntry) {
                                                params.onSuccess && params.onSuccess.call(module, dirEntry, fileSystem);
                                            }, function (e) {
                                                var data = {
                                                    msg: action + name + " 文件夹失败"
                                                }
                                                params.onFail && params.onFail.call(module, data);
                                            });
                                        },function (e) {
                                            var data = {
                                                msg: "文件系统还没准备好."
                                            }
                                            params.onFail && params.onFail.call(module, data);
                                        });
                            
                                        return this;
                                    }
                            
                                    //返回标准路径
                                    function checkPath(names) {
                                        var names = names || [];
                                        var name;
                                        // 如果传的参数带有 './' or '/' 则忽略
                                        if (names[0] == '.' || names[0] == '/' || names[0] == ' ') {
                                            name = names.slice(1);
                                            return checkPath(name);
                                        }
                                        // 返回正确的名称
                                        if (names[0] != '.' && names[0] != '/' && names[0] != ' ') {
                                            return names;
                                        }
                                    }

                                     * 获取文件名
                                     *  @method getFileName
                                     *  @param {string} name [ url地址, 例如: file:开头,http开头  ]

                                     *
                                        var name = uiFile.getFileName("file://bui.jpg?id=abcd");

                                    function getFileName(names) {
                                        var name;
                                        // 获取文件名
                                        if (names && names.indexOf("/") > -1) {
                                            name = names.slice(names.lastIndexOf("/") + 1, (names.indexOf("?") > -1 ? names.indexOf("?") : undefined));
                                            return name;
                                        } else {
                                            return names;
                                        }
                                    }

                                    function getFile(param) {
                                        var config = {
                                            fileName: "",
                                            folderName: "",
                                            root: false,
                                            create: false,
                                            param: {
                                                create: false,
                                                exclusive: false
                                            },
                                            onSuccess: null,
                                            onFail: null
                                        }
                            
                                        var params = $.extend({}, config, param),
                                            // 动作
                                            action = "";
                            
                                        if (params.create) {
                                            action = "创建 ";
                                            params.param.create = true;
                                        } else {
                                            action = "获取 ";
                                            params.param.create = false;
                                        }
                            
                                        // 如果没有传名称,自动跳出
                                        if (!params.fileName || params.fileName.indexOf(".") < 0) {
                                            params.onFail && params.onFail();
                                            return;
                                        }
                            
                                        // 转换标准文件名才能创建成功
                                        var name = getFileName(params.fileName);
                            
                                        //文件系统准备完毕
                                        getFolder({
                                            root: params.root,
                                            folderName: params.folderName,
                                            create: true,
                                            onSuccess: function (folder, fileSystem) {
                            
                                                // 文件默认都创建在默认文件夹下,防止文件混乱
                                                // var fileName = folder?.name ? (defaultFolder + "/" + folder.name + "/" + name) : (defaultFolder + "/" + name);
                                                var fileName = folder?.fullPath+ name;
                            
                                                // 创建文件或者获取文件
                                                fileSystem.root.getFile(fileName, params.param, function (fileEntry) {
                                                    // fileEntry.isFile === true
                                                    // fileEntry.name == 'log.txt'
                                                    // fileEntry.fullPath == '/log.txt'
                                                    // fileEntry.nativeURL == 'file:///xxx/log.txt'
                            
                                                    params.onSuccess && params.onSuccess.call(module, fileEntry, fileSystem);
                            
                                                }, function (e) {
                            
                                                    var data = {
                                                        msg: action + name + " 文件失败"
                                                    }
                                                    params.onFail && params.onFail.call(module, data);
                                                });
                                            },
                                            onFail: function (err) {
                                                params.onFail && params.onFail(err);
                                            }
                                        })
                            
                                        return this;
                                    }

                                                create: false

                                        }
                                        var params = $.extend(true, {}, param);

                                            root: params.root,
                                            folderName: params.folderName,
                                            create: params.create,
                                            onSuccess: function (folder, fileSystem) {
                            
                                                // 里面有文件时，不能删除
                                                // folder.remove(function(){
                                                //     params.onSuccess && params.onSuccess.call(module, folder, fileSystem);

                                                //     params.onFail && params.onFail.call(module, data, folder);
                                                // })
                                                // 先删除文件夹里的文件，再进行删除文件夹的操作
                                                folder.removeRecursively(function () {
                                                    folder.remove(function(){
                                                        params.onSuccess && params.onSuccess.call(module, folder, fileSystem);
                                                    })
                                                }, function (err) {
                                                    var data = {

                                                    msg: "删除 " + params.folderName + " 文件失败"
                                                }
                                                params.onFail && params.onFail.call(module, data);

                                        })
                            
                                        return this;
                                    }

                                    function writeFile(opt) {
                                        
                                        let param = $.extend(false,{},{
                                            file:null,
                                            content:"",
                                            type:"text/plain",
                                            onSuccess:null,
                                            onFail:null,
                                        },opt);
                            
                                        let fileEntry = param.file || {};
                                        // Create a FileWriter object for our FileEntry (log.txt).
                                        fileEntry.createWriter(function (fileWriter) {
                                            
                                            fileWriter.onwriteend = function() {
                            
                                                param.onSuccess && param.onSuccess.call(module, fileWriter);
                                            };
                                    
                                            fileWriter.onerror = function (e) {
                            
                                                param.onFail && param.onFail.call(module, fileWriter);
                                            };
                                            
                                            var dataObj = new Blob([param.content], { type: param.type });
                                    
                                            fileWriter.write(dataObj);
                                        });
                                    }

                                    }
                            
                                    /*
                                     * 把选择的图片文件转换成base64地址,可以直接展示在页面上
                                     *  @method toBase64
                                     *  @param {object} [option] []
                                     *  @param {string} [option.data] [ 通过文件选择  ]
                                     *  @param {function} [option.onSuccess] [成功的回调]
                                     *  @param {function} [option.onFail] [失败的回调]
                                     *  @chainable
                                     *  @example
                                     *
                                     *
                                     *
                                        // 展示本地图片
                                        uiDownload.toBase64({
                                            onSuccess: function (imgurl) {
                                                $("#id").append('<img src="'+imgurl+'" />')
                                            }
                                        });
                            
                            
                            
                                     */
                                    function toBase64(param) {
                            
                                        var param = param || {},
                                            data = param.data || uploadData,
                                            onSuccess = param.onSuccess,
                                            onFail = param.onFail;
                            
                                        // 原生的图片转换
                                        var imageData = data.fullPath || data;
                            
                            
                                        _fileSystem.then(gotFS,fail)
                            
                                        function gotFS(fileSystem) {
                                            let index = imageData.indexOf(defaultFolder);
                            
                                            // 这里要转为相对路径，buiapp开头
                                            let fileurl = imageData.substr(index);
                            
                                            fileSystem.root.getFile(fileurl, {create:false}, gotFileEntry, fail);
                                        }
                            
                                        function gotFileEntry(fileEntry) {
                            
                                            fileEntry.file(gotFile, fail);
                                        }
                            
                                        // 把文件路径转换成base64
                                        function gotFile(file) {
                                            readDataUrl(file);
                                        }
                            
                                        // 转换成base64
                                        function readDataUrl(file) {
                                            var reader = new FileReader();
                                            reader.onloadend = function(evt) {
                                                // evt.target.result 对象只能通过回调传出
                                                onSuccess && onSuccess.call(module, evt.target.result, file)
                                            };
                                            reader.readAsDataURL(file);
                                        }
                            
                                        // 转换成二进制
                                        function readAsText(file) {
                                            var reader = new FileReader();
                                            reader.onloadend = function(evt) {
                                                onSuccess && onSuccess.call(module, evt.target.result, file)
                                            };
                                            reader.readAsText(file);
                                        }
                            
                                        // 转换失败
                                        function fail(evt) {
                                            onFail && onFail.call(module, evt);
                                        }
                            
                                        return this;
                                    }
                            
                                    /*
                                     * 获取依赖的控件
                                     *  @method widget
                                     *  @param {string} [name] [ 依赖控件名 fileselect ]
                                     *  @example
                            
                                        //获取依赖控件
                                        var uiFileSelect = uiFile.widget("fileselect");
                            
                                        //使用uifileselect的方法
                                        uiFileSelect.add({
                                            onSuccess: function(data){
                                                console.log(data);
                                            }
                                        });
                            
                                     */
                                    function widget(name) {
                                        var control = {
                                        };
                                        return ui.widget.call(control, name);
                                    };
                            
                                    return module;
                                };
                            
                                return ui;
                            
                            })(window.bui || {}, window.libs);
