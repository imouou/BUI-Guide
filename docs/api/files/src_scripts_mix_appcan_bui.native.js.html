                             * <h5>数据交互</h5>
                             * {{#crossLink "bui.ajax"}}{{/crossLink}}: 数据请求 <br>
                             * {{#crossLink "bui.upload"}}{{/crossLink}}: 上传文件 <br>

                             * {{#crossLink "bui.file"}}{{/crossLink}}: 文件管理 <br>
                             * {{#crossLink "bui.fileselect"}}{{/crossLink}}: 文件选择 <br>
                             * {{#crossLink "bui.checkVersion"}}{{/crossLink}}: 获取最新版本 <br>
                             * <h5>页面交互</h5>
                             * {{#crossLink "bui.load"}}{{/crossLink}}: 页面跳转及传参 <br>
                             * {{#crossLink "bui.getPageParams"}}{{/crossLink}}: 页面获取参数 <br>

                                /*
                                 * 数据请求
                                 * @namespace  bui
                                 * @class  ajax
                                 * @constructor 
                                 * @param {object} option 
                                 *   @param {string} option.url [请求的地址]
                                 *   @param {object} option.data [传输的数据对象]
                                 *   @param {string} [option.method] [传输的方法 GET | POST]
                                 *   @param {string} [option.dataType] [默认 json | jsonp | script | xml | html | text]
                                 *   @param {string} [option.contentType] [默认 'text/html;charset=UTF-8' | 'application/x-www-form-urlencoded']'
                                 *   @param {number} [option.timeout] [超出这个时间则认为请求失败]
                                 * @example
                                 *     bui.ajax({
                                 *         url: "",
                                 *         data: {}
                                 *     }).then(function(res){
                                 *         console.log(res)
                                 *     },function(res,status){
                                 *         console.log(status);
                                           // status = "timeout" || "error" || "abort", "parsererror"
                                 *     })
                                 * 
                                 */
                                ui.ajax = function (params) {
                            
                                    //延时处理
                                    var def = $.Deferred();
                            
                                    var ajaxHandle,
                                        tag = ui.guid();
                                    params = params || {};
                                    params.data = params.data || {};
                                    params.method = params.method || "GET";
                                    params.dataType = params.dataType || "json";
                                    params.timeout = params.timeout || 60000;
                                    params.headers = params.headers || {};
                                    params.processData = params.processData == false ? false : true;
                                    params.mimeType = params.mimeType || "none";
                                    params.cache = params.cache == false ? false : true;
                            
                            
                                    if (typeof (params.async) == "undefined") {
                                        params.async = true;
                                    }
                                    // 如果没传contentType 就根据不同的请求给予默认的参数值
                                    if (params.method == "GET" && (params.contentType === undefined || params.contentType === "")) {
                                        params.contentType = "text/html;charset=UTF-8";
                                    } else if (params.method == "POST" && (params.contentType === undefined || params.contentType === "")) {
                            
                                        params.contentType = "application/x-www-form-urlencoded";
                                    } else {
                                        params.contentType = params.contentType;
                                    }
                            
                                    // 采用web方式请求
                                    if (this.isWebapp || params.native == false) {
                                        ajaxHandle = $.ajax({
                                            async: params.async,
                                            url: params.url,
                                            data: params.data,
                                            type: params.method,
                                            dataType: params.dataType,
                                            headers: params.headers,
                                            contentType: params.contentType,
                                            timeout: params.timeout,
                                            processData: params.processData,
                                            mimeType: params.mimeType,
                                            cache: params.cache,
                                            success: function (res, status, xhr) {
                                                var obj;
                                                if (typeof res == 'string') {
                                                    try {
                                                        obj = JSON.parse(res);
                                                    } catch (e) {
                                                        obj = res;
                                                    }
                                                } else {
                                                    obj = res || {};
                                                }
                                                def.resolve(obj, status, xhr);
                                            },
                                            error: function (res, status, xhr) {
                                                // status: "timeout", "error", "abort", "parsererror"
                                                var obj;
                                                if (typeof res == 'string') {
                                                    try {
                                                        obj = JSON.parse(res);
                                                    } catch (e) {
                                                        obj = res;
                                                    }
                                                } else {
                                                    obj = res || {};
                                                }
                                                def.reject(obj, status, xhr);
                                            }
                                        })
                            
                                    } else {
                            
                                        // 原生请求暂无 abort 阻止请求方法,所以会导致list 多次重复请求问题. 
                                        var successCallback = function (res, status) {
                            
                                            var obj;
                                            if (typeof res == 'string') {
                                                try {
                                                    obj = JSON.parse(res);
                                                } catch (e) {
                                                    obj = res;
                                                }
                                            } else {
                                                obj = res || {};
                                            }
                                            def.resolve(obj, status);
                                        };
                                        var failCallback = function (res, status) {
                                            var obj;
                                            if (typeof res == 'string') {
                                                try {
                                                    obj = JSON.parse(res);
                                                } catch (e) {
                                                    obj = res;
                                                }
                                            } else {
                                                obj = res || {};
                                            }
                                            def.reject(obj, status);
                                        };
                            
                                        ajaxHandle = new plus.net.XMLHttpRequest();
                                        ajaxHandle.onreadystatechange = function () {
                                            switch (ajaxHandle.readyState) {
                                                case 0:
                                                    // outLine( "xhr请求已初始化" );
                                                    break;
                                                case 1:
                                                    // outLine( "xhr请求已打开" );
                                                    break;
                                                case 2:
                                                    // outLine( "xhr请求已发送" );
                                                    break;
                                                case 3:
                                                    // outLine( "xhr请求已响应");
                                                    break;
                                                case 4:
                                                    // outLine( "xhr请求已完成");
                                                    if (ajaxHandle.status == 200) {
                                                        successCallback(ajaxHandle.responseText, ajaxHandle.status)
                                                    } else {
                                                        failCallback({}, ajaxHandle.status)
                                                    }
                                                    break;
                                                default:
                                                    break;
                                            }
                                        }
                            
                                        // 超时处理
                                        ajaxHandle.responseType = params.dataType;
                                        // 超时处理
                                        ajaxHandle.timeout = params.timeout;
                                        ajaxHandle.ontimeout = function (e) {
                                            failCallback({}, "timeout");
                                        }
                                        ajaxHandle.overrideMimeType(params.contentType);
                                        var async = "async" in params ? params.async : true;
                                        ajaxHandle.open(params.method, params.url, async);
                            
                                        // 请求头处理
                                        for (name in params.headers) {
                                            ajaxHandle.setRequestHeader(name, params.headers[name]);
                                        }
                            
                                        ajaxHandle.send(params.data ? params.data : null);
                            
                                    };
                            
                                    //中断请求
                                    def.abort = function () {
                            
                                        ajaxHandle && ajaxHandle.abort();
                                        ajaxHandle = null;
                            
                                    };
                            
                                    return def;
                            
                                };
                            
                                /**
                                 * 页面跳转可以传参 获取参数请看 {{#crossLink "bui.getPageParams"}}{{/crossLink}} 方法
                                 * @namespace  bui
                                 * @class  load
                                 * @constructor 
                                 * @param {object} option 
                                 *   @param {string} option.url 请求的地址
                                 *   @param {object} option.param 传过去另外页面的参数
                                 * @example
                                 *     bui.load({ 
                                 *         url: "",
                                 *         param:{}
                                 *     })
                                 * 
                                 */
                                ui.load = function (params) {
                                    var newurl, param,
                                        url = params.url,
                                        localUrl = window.location.href;
                            
                                    if (!params) {
                                        ui.hint("should be object like {url:'http://domain',params:{....}}");
                                        return;
                                    }
                                    if (!url) {
                                        ui.hint("url is necessary!");
                                        return;
                                    }
                            
                            
                                    //传过去目标页面的值
                                    param = params.param;
                                    //如果开启调试,使用url的方式传值
                                    if (this.isWebapp || params.native == false) {
                            
                                        newurl = ui.setUrlParams(params.url, param);
                            
                                        window.location.href = newurl;
                            
                                    } else {
                                        if (url.indexOf("http://") < 0 || url.indexOf("https://") < 0 || url.indexOf("file://") < 0 || url.indexOf("/") < 0) {
                                            var lastIndex1 = localUrl.lastIndexOf("/"),
                                                lastIndex2 = localUrl.lastIndexOf(".");
                                            url = localUrl.substring(0, lastIndex1) + "/" + url;
                                            params.url = url;
                            
                                            var urlname = url.slice(url.lastIndexOf("/") + 1, (url.lastIndexOf(".") > -1 ? url.lastIndexOf(".") : undefined));
                            
                                            var webviewContent = plus.webview.create(params.url, urlname, {
                                                scrollIndicator: 'none',
                                                scalable: false,
                                                popGesture: 'none'
                                            }, param);
                            
                                            // ws.show("slide-in-right");
                                            webviewContent.addEventListener("loaded", function () { //注册新webview的载入完成事件
                            
                                                webviewContent.show("slide-in-right", 200); //把新webview窗体显示出来，显示动画效果为速度200毫秒的右侧移入动画
                                            }, false);
                                        }
                                    }
                                };
                            
                                /**
                                 * 获取页面参数,配合 {{#crossLink "bui.load"}}{{/crossLink}} 使用
                                 * @namespace  bui
                                 * @class  getPageParams
                                 * @constructor 
                                 * @param {function} callback 数据都在回调里面
                                 * @example
                                 *     bui.getPageParams(function(res){
                                 *         var urlparam = res;
                                 *
                                 *          console.log(urlparam) //{}
                                 *     });
                                 * 
                                 */
                                ui.getPageParams = function (option) {
                            
                                    var callback,
                                        native;
                                    if (typeof option === "function") {
                                        callback = option || function () { };
                                        native = true;
                                    } else if (option && option.callback) {
                                        callback = option.callback || function () { };
                                        native = option.native;
                                    }
                            
                                    //如果开启调试,使用url的方式取值
                                    if (this.isWebapp || native == false) {
                            
                                        var params = this.getUrlParams();
                            
                                        callback && callback(params);
                            
                                    } else {
                            
                                        var params = plus.webview.currentWebview();
                            
                                        callback && callback(params);
                                    }
                                };
                                /**
                                 * 后退, 原生可以使用回调,pc调试只能在当前页面触发回调
                                 * @namespace  bui
                                 * @class  back
                                 * @constructor 
                                 * @param {function} callback 在后退的页面触发
                                 * @example
                                 *     bui.back();
                                 * 
                                 */
                                ui.back = function (option) {
                            
                                    var callback,
                                        native;
                                    if (typeof option === "function") {
                                        callback = option || function () { };
                                        native = true;
                                    } else if (option && option.callback) {
                                        callback = option.callback || function () { };
                                        native = option.native;
                                    }
                                    if (this.isWebapp || native == false) {
                            
                                        //pc不能在后退以后执行操作
                                        window.history.back();
                            
                                    } else {
                                        // 关闭webview而不是隐藏,释放内存
                                        var ws = plus.webview.currentWebview();
                            
                                        plus.webview.close(ws);
                            
                                        callback && callback(ws);
                                    }
                            
                                };
                                /**
                                 * 刷新页面
                                 * @namespace  bui
                                 * @class  refresh
                                 * @constructor 
                                 * @example
                                 *     bui.refresh();
                                 * 
                                 */
                                ui.refresh = function (option) {
                            
                                    var native = option && option.native;
                                    if (this.isWebapp || native == false) {
                                        //PC调试无法回调
                                        window.location.reload();
                                    } else {
                                        var wobj = plus.webview.currentWebview();
                                        wobj.reload(true);
                                    }
                                }

                                 * @namespace  bui
