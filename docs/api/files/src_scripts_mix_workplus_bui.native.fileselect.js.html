                            /**
                             * @module Native
                             */
                            ;
                            (function(ui, $) {
                            
                                /*
                                 * <h3>选择文件, 支持web跟安卓,ios</h3>
                                 */
                                ui.native.fileselect = function(option, depend) {
                            
                                    var options = option,
                                        depend = depend || {};
                            
                                    var module = {
                                        add: add,
                                        remove: remove,
                                        clear: clear,
                                        value: value,
                                        data: data,
                                        toBase64: toBase64,
                                        init: init
                                    };
                            
                                    var uploadData = null,
                                        // 每次初始化清空缓存数据
                                        fileData = [];
                            
                                    // 初始化
                                    init(options);
                            
                                    //选择器初始化
                                    function init() {
                            
                                        fileData = [];
                                    }
                            
                                    /*
                                     * 添加文件, 参数的配置全部针对原生选择
                                     *  @method add
                                     *  @param {object} [option] [对象]
                                     *  @param {number} [option.quality] [图片质量,默认0.8]
                                     *  @param {number} [option.width] [图片宽度,默认800, targetWidth和targetHeight设置为undefined 则添加原图]
                                     *  @param {number} [option.height] [图片高度,默认800, targetWidth和targetHeight设置为undefined 则添加原图]
                                     *  @param {string} [option.from] [ 图片来源, 默认来自 photo(相册) | camera(相机) | savephoto(来自保存的相册) ]
                                     *  @param {string} [option.destinationType] [ file (返回路径值) | data (返回base64位值) ]
                                     *  @param {function} [option.onSuccess] [ 成功的回调 ]
                                     *  @param {function} [option.onFail] [ 失败的回调 ]
                                     *  @chainable
                                     *  @example 
                                     *
                                        uiFileSelect.add({
                                            onSuccess: function(file){
                                                // this 指向uiFileSelect
                                                // console.log(this)
                                            }
                                        })
                            
                                     */
                                    function add(param) {
                            
                                        uploadData = null;
                                        var _self = this;
                            
                                        var params = $.extend(true, { width: 800, height: 800 }, options, param);
                            
                                        addPicture(params);
                                        // 添加图片类型 添加的图片会自动修改为质量80,分辨率 800*800 的jpg文件
                                        // 要显示本地图片,需要使用destinationType 设置为 data; 返回 data:image/jpeg;base64,
                                        function addPicture(params) {
                            
                                            switch (params.from) {
                                                case "picture":
                                                case "photo":
                                                case "savephoto":
                                                    // config.sourceType = Camera.PictureSourceType.PHOTOLIBRARY;
                                                    selectImage({
                                                        success: function(fileEntry) {
                                                            pictureProcessing(fileEntry)
                                                        },
                                                        fail: function(error) {
                                                            params.fail && params.fail(error);
                                                        }
                                                    });
                                                    break;
                                                case "camera":
                                                    // config.sourceType = Camera.PictureSourceType.CAMERA;
                                                    takePhoto({
                                                        success: function(fileEntry) {
                                                            pictureProcessing(fileEntry)
                                                        },
                                                        fail: function(error) {
                                                            params.fail && params.fail(error);
                                                        }
                                                    });
                                                    break;
                                                default:
                                                    selectImage({
                                                        success: function(fileEntry) {
                                                            pictureProcessing(fileEntry)
                                                        },
                                                        fail: function(error) {
                                                            params.fail && params.fail(error);
                                                        }
                                                    });
                                                    break;
                            
                                            }
                            
                            
                                        }
                            
                                        // 图片处理,返回一些基本信息
                                        function pictureProcessing(fileEntry) {
                                            var imageData = fileEntry.imageURL || fileEntry.key; // imageURL 压缩的图片 // key 原图
                            
                                            // 缓存数据后面再上传
                                            uploadData = imageData;
                            
                                            var lastSymbol = imageData.lastIndexOf('?'),
                                                lastIndex = lastSymbol < 0 ? undefined : lastSymbol,
                                                typeName = imageData.slice(imageData.lastIndexOf('.') + 1, lastIndex),
                                                // 经过修改的图片的名称都会是相同的,
                                                name = imageData.slice(imageData.lastIndexOf('/') + 1, lastIndex),
                                                type = "image/" + typeName;
                            
                                            //如果名称不相同,则缓存数据
                                            if (!ui.array.compare(name, fileData, "name")) {
                                                var item = {
                                                    name: name,
                                                    data: imageData,
                                                    type: type
                                                };
                            
                                                fileData.push(item);
                                            }
                            
                                            params.onSuccess && params.onSuccess.call(_self, imageData, fileData, fileEntry);
                            
                                        }
                                    }
                            
                            
                                    // 拍照
                                    /*
                                    请求返回数据：
                                        {
                                            "imageURL":"压缩后图像存在本地的地址",
                                            "key":"原图像存在本地的地址",
                                            "imageInfo": //相对于图像的信息
                                            {
                                                "height":"图像高",
                                                "width":"图像宽",
                                                "size":"图像大小"
                                            }
                                        }
                                    */
                                    function takePhoto(option) {
                                        option = option || {};
                                        cordova.exec(function(result) {
                                                // alert(JSON.stringify(result, null, 4));
                                                option.success && option.success(result);
                                            },
                                            function(error) {
                                                // alert("调用失败");
                                                option.fail && option.fail(error);
                            
                                            },
                                            "WorkPlus_Image",
                                            "takePhoto", []
                                        );
                            
                                    }
                            
                                    // 选择图片
                                    /*
                                    请求返回数据：
                                        {
                                            "imageURL":"压缩后图像存在本地的地址",
                                            "key":"原图像存在本地的地址",
                                            "imageInfo": //相对于图像的信息
                                            {
                                                "height":"图像高",
                                                "width":"图像宽",
                                                "size":"图像大小"
                                            }
                                        }
                                    */
                                    function selectImage(option) {
                                        option = option || {};
                                        cordova.exec(function(result) {
                                                // alert(JSON.stringify(result, null, 4));
                                                option.success && option.success(result);
                                            },
                                            function(error) {
                                                // alert("调用失败");
                                                option.fail && option.fail(error);
                            
                                            },
                                            "WorkPlus_Image",
                                            "selectImage", []
                                        );
                                    }
                            
                            
                                    /*
                                     * 把选择的图片文件转换成base64地址,可以直接展示在页面上
                                     *  @method toBase64
                                     *  @param {object} [option] []
                                     *  @param {string} [option.data] [ 通过文件选择  ]
                                     *  @param {function} [option.onSuccess] [成功的回调]
                                     *  @param {function} [option.onFail] [失败的回调]
                                     *  @chainable
                                     *  @example 
                                     *
                                     *  
                                     *  
                                        // 展示本地图片
                                        uiFileSelect.toBase64({
                                            onSuccess: function (imgurl) {
                                                $("#id").append('<img src="'+imgurl+'" />')
                                            }
                                        });
                            
                            
                                        
                                     */
                                    function toBase64(param) {
                            
                                        var param = param || {},
                                            data = param.data || uploadData,
                                            onSuccess = param.onSuccess,
                                            onFail = param.onFail;
                            
                                        // 
                                        try {
                                            base64(data, onSuccess, onFail);
                            
                                        } catch (e) {
                                            console.log(e);
                                        }
                            
                                        function base64(path, success, error) {
                                            function resolveImgURI(fileUrl, callback) {
                                                if (cordova.platformId.toLowerCase() === 'android') {
                                                    fileUrl = 'file://' + fileUrl;
                                                }
                                                // 无法走通在workplus
                                                window.resolveLocalFileSystemURL(fileUrl, function(fileEntry) {
                                                    fileEntry.file(function(file) {
                                                        readDataUrl(file, callback);
                                                    }, fail);
                                                }, fail);
                                            }
                            
                                            function readDataUrl(file, callback) {
                                                var reader = new FileReader();
                                                reader.onloadend = function(evt) {
                                                    if (evt.target._result) {
                                                        evt.target._result = evt.target._result.split(',')[1];
                                                    }
                                                    callback(evt.target._result, evt);
                                                };
                                                reader.readAsDataURL(file);
                                            }
                            
                                            function fail(evt) {
                                                error(evt);
                                            }
                                            resolveImgURI(path, success);
                                        }
                            
                                        return this;
                                    }
                            
                                    /*
                                     * 删除选择的文件
                                     *  @method remove
                                     *  @param {string} name [ 要删除的图片名字带后缀名 ]
                                     *  @param {string} [key] [ 图片名字在哪个字段, 默认是 "name"]
                                     *  @chainable
                                     *  @example 
                                     *
                                        uiFileSelect.remove("resize.jpg");
                            
                                     */
                                    function remove(name, key) {
                                        var key = key || "name";
                                        if (typeof name == "string") {
                                            fileData = ui.array.remove(name, fileData, key);
                                            // 删除某个选择文件后需要修改上传的文件值为新的最后一条数据;
                                            var lastData = fileData.length ? fileData[(fileData.length - 1)]["data"] : null;
                                            uploadData = lastData;
                                            return fileData;
                                        }
                            
                                        return this;
                                    }
                            
                                    /*
                                     * 清空所有选择的文件
                                     *  @method clear
                                     *  @example 
                                     *
                                        uiFileSelect.clear();
                            
                                     */
                                    function clear() {
                                        fileData = [];
                                        uploadData = null;
                                        return fileData;
                                    }
                            
                                    /*
                                     * 获取所有选择的文件, 原生默认只返回最后一个选择的文件
                                     *  @method data
                                     *  @example 
                                     *
                                        var data = uiFileSelect.data();
                                        // 返回 [{ name: "base64.jpg", data: null, type: "jpg" }]
                                        // 其中 data为要上传的文件值
                            
                                     */
                                    function data() {
                            
                                        return fileData;
                                    }
                            
                                    /*
                                     * 获取最后选择的文件
                                     *  @method value
                                     *  @example 
                                     *
                                        var value = uiFileSelect.value();
                                        // 返回值 value为要上传的文件值, web上传跟原生上传得到的值不一样
                            
                                     */
                                    function value() {
                                        return uploadData;
                                    }
                            
                            
                                    return module;
                                };
                            
                                return ui;
                            
                            })(bui || {}, libs);
