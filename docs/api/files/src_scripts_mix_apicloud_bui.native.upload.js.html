                            /**
                             * @module Native
                             */
                            ;(function (ui,$) {
                                
                                // 上传控件
                                ui.native.upload = function (option,depend) {
                                    
                                    var options = option,
                                        depend = depend || {};
                            
                                    var module = {
                                        init : init,
                                        start: start,
                                        stop: stop
                                    };
                            
                            
                                    var uiLoading = depend.loading || function(){},
                                        uiSelect = depend.fileselect || function(){},
                                        uiAjax,
                                        //fileData 是一个对象 document.getElementById('fileToUpload').files
                                        fileData,
                                        uploadUrl,
                                        // 是否显示进度条
                                        showProgress = options.showProgress,
                                        //对进度条单独处理
                                        onProgress = options.onProgress,
                                        progressPercent;
                            
                                    // 初始化
                                    init(options);
                            
                                    function init(param) {
                                        
                                        if( param.data ){
                                           //上传
                                            start(param); 
                                        }
                                        
                                    }
                            
                                    /*
                                     * 开始上传
                                     *  @method start
                                     *  @param {object} [option] [参数同初始化一样,在初始化时配置,则是公共的, data不传则获取最后一次选择]
                                     *  @chainable
                                     *  @example 
                                     *
                                        uiUpload.start({
                                            data: null,
                                            onSuccess: function (res) {
                                                // 成功
                                            },
                                            onFail: function (res,status) {
                                                // 失败 status = "timeout" || "error" || "abort", "parsererror"
                                            }
                                        })
                            
                                     */
                                    function start(param) {
                            
                                        var params = $.extend(true,{},options,param);
                            
                                        uploadUrl = params.url;
                            
                                        fileData = uiSelect.value();
                            
                                        showProgress = params.showProgress;
                            
                                        if( !fileData ){
                                            return;
                                        }
                                        
                                        showProgress && uiLoading.show();
                            
                                        // 原生的上传
                                        nativeAjax(params);
                            
                                        return this;
                                    }
                            
                                    /*
                                     * 停止上传
                                     *  @method stop
                                     *  @chainable
                                     *  @example 
                                     *
                                        uiUpload.stop()
                            
                                     */
                                    function stop(callback) {
                                        uiLoading && uiLoading.stop();
                                        uiAjax && uiAjax.abort();
                            
                                        callback && callback.call(module,uiLoading,uiAjax);
                            
                                        return this;
                            
                                    }
                            
                                    // 原生上传
                                    function nativeAjax(param) {
                            
                                        var onSuccess = param.onSuccess;
                                        var onFail = param.onFail;
                            
                                        uploadUrl = encodeURI(param.url);
                            
                                        var mediaType;
                                        // 切换类型
                                        switch( param.mediaType ){
                                            case "picture":
                                            mediaType = navigator.camera.MediaType.PICTURE;
                                            break;
                                            case "video":
                                            mediaType = navigator.camera.MediaType.VIDEO;
                                            break;
                                            default :
                                            mediaType = navigator.camera.MediaType.ALLMEDIA;
                                            break;
                                        }
                            
                                        var fileOptions = new FileUploadOptions();
                                            fileOptions.fileKey = param.fileKey;// || "file";
                                            fileOptions.mimeType = param.mimeType;// || "image/jpeg";
                                            fileOptions.fileName = fileData.substr(fileData.lastIndexOf('/') + 1) ||  "image.jpg";
                                            fileOptions.params = param.data;
                                            fileOptions.mediaType = mediaType;
                            
                                        uiAjax = new FileTransfer();
                                        // 上传进度
                            
                                        uiAjax.onprogress = uploadProgress;
                            
                                        uiAjax.upload(fileData, uploadUrl, function (res) {
                            
                                            // 关闭进度条
                                            stopLoading();
                            
                                            var response = JSON.parse(res.response);
                            
                                                onSuccess&&onSuccess.call(module,response);
                            
                                        }, function (res) {
                                            stopLoading();
                            
                                            onFail&&onFail.call(module,res);
                            
                                        }, options);
                            
                                        return this;
                            
                                    }
                            
                                    // web 上传进度
                                    function uploadProgress(evt) {
                                        //安卓4.3以下不支持
                                        window.requestAnimationFrame = window.requestAnimationFrame ||window.webkitRequestAnimationFrame ;
                            
                                        // 防止掉帧
                                        window.requestAnimationFrame(function() {
                                            if (evt.lengthComputable) {
                            
                                                var percentComplete = Math.round(evt.loaded * 100 / evt.total);
                                                progressPercent = percentComplete.toString() + '%';
                            
                                                if( percentComplete < 100 ){
                                                    startLoading(progressPercent)
                                                }else{
                                                    stopLoading();
                                                }
                                                // 原生进度
                                                // navigator.notification.progressValue(Math.round(( progressEvt.loaded / progressEvt.total ) * 100));
                            
                                                onProgress && onProgress.call(module,progressPercent);
                                            }
                                        })
                                        
                                    }
                            
                                    // 开启进度
                                    function startLoading(percent) {
                                        uiLoading &&  uiLoading.show({
                                            text : percent
                                        });
                            
                                        return this;
                                    }
                            
                                    // 关闭进度
                                    function stopLoading() {
                                        uiLoading &&  uiLoading.stop();
                            
                                        return this;
                                    }
                            
                                    return module;
                                };
                            
                                return ui;
                            
                            })( bui || {}, libs );
