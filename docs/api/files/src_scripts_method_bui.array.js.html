                                 * 合并2个数组, A数组+B数组+C数组 并会触发 A数组的更新
                                 * @namespace  bui
                                 * @method  merge
                                 * @since  1.5.1
                                 * @constructor
                                 * @param {array} array [数组]
                                 * @param {array} arrayA [支持多个数组]
                                 * @return {array} [空数组]

                                ui.array.merge = function (arr) {
                            
                                    if (arr && ui.typeof(arr) !== "array") {
                                        return;
                                    }
                            
                                    var argumentArr = arguments;
                                    var lastLength = argumentArr.length - 1
                                    var lastArg = argumentArr[lastLength];
                                    var lastType = typeof lastArg == "string";
                            
                                    // 最后一个是字符串,说明这个要合并的是个对象,而且是合并字段; 至少3个参数
                                    if (lastType && argumentArr.length > 1) {
                                        let arglenth = lastLength; // 最后一个是字段,不是数组
                                        let newArr = [];
                                        // 遍历多少个参数
                                        for (let i = 0; i < arglenth; i++) {
                                            // 复制新的数组操作
                                            let target = Array.prototype.slice.call(argumentArr[i], 0);
                                            arr.forEach(function (item, index) {
                                                // 跟this的每个对象值,lastArg 为比对的字段名
                                                let itemObj = item[lastArg] || item;
                                                ui.typeof(target) === "array" && target.forEach(function (aitem, aindex) {
                                                    let aitemObj = aitem[lastArg] || aitem;
                                                    // 如果之间的唯一值比对是一致的,则进行合并
                                                    if (itemObj === aitemObj) {
                                                        // 新增每个字段
                                                        for (let key in aitem) {
                                                            item[key] = aitem[key];
                                                        }
                                                        // 存在则删掉当前数组的数据
                                                        target.splice(aindex, 1);
                                                    }
                                                })
                                            })
                                            newArr.push(target);
                                        }
                            
                                        // 再次调用自身数组合并,不改变之前顺序
                                        return ui.array.merge(arr, ...newArr);
                                    }
                                    // 最后一条插入数据, 扁平化数组
                                    var updates = [arr.length, 0].concat(...Array.prototype.slice.call(arguments, 1));
                                    arr.splice.apply(arr, updates);
                            
                                    return arr;
                                }

                                 * @param {array}  array [数组]
                                 * @param {string} value [要查找的值]
                                 * @param {string} [key] 如果是object,需要传该值在哪个键名里
                                 * @return {array} [返回数组]
                                 * @example
                                 *
                                     //普通数组:
                                     var arr = ["hello","bui","hi","easybui"];
                                     var item = bui.array.getAll( arr,"bui" );
                            
                                     // console.log(item) //结果: ["bui"]
                            
                                     //JSON数组:
                                     var arr = [{ "id":1,value:"hello"},{ "id":2,value:"bui"},{ "id":3,value:"bui"}];
                                     var item = bui.array.getAll( arr, "bui", "value" );
                                     // console.log(newArr) //结果: [{ "id":2,value:"bui"},{ "id":3,value:"bui"}]
                                 *
                                 */
                            
                                ui.array.getAll = function (arr, name, key) {
                                    var arr = arr || [];
                                    var data = [],
                                        item = null;
                                    for (var i in arr) {
                            
                                        try {
                                            item = arr[i];
                                            var arrs = item && item[key] || item;
                            
                                            if (name && ui.typeof(name) === "array") {
                                                name.forEach(function (itemname, i) {
                                                    //检测的数组是对象
                                                    if (arrs === itemname) {
                                                        data.push(item);
                                                    }
                                                })
                                            } else {
                                                //检测的数组是对象
                                                if (arrs === name) {
                                                    data.push(item);
                                                }
                                            }
                            
                                        } catch (e) { }
                                    }
                            
                                    // 如果没有匹配,返回undefined
                                    return data;
                                };
                                /**
                                 * 修改数组或数组对象里面的某个值,结合store触发数据变更, 有3个参数时,第2个是索引值或者数组某个值, 如果是对象,需要第4个参数
                                 * @namespace  bui
                                 * @method  set
                                 * @since  1.5.2
                                 * @constructor
                                 * @param {array} array [数组]
                                 * @param {number|string|object|array} key|value [ 数字索引值,或者是值, 如果是数字,设置索引, 如果是其它值则比对值 ]
                                 * @param {number|string|object} [value]  修改的值 
                                 * @param {string} [keyname] 如果比对的是对象,则需要这个键值
                                 * @return {array} [空数组]
                                 * @example
                                 *
                            
                                    例子1: 修改第几个
                                    var arr = ["hello","hi","easybui"];
                                    bui.array.set( arr, 1, "new hi");
                                    // ["hello","new hi","easybui"]
                                    
                                     1.5.3 版本以后可以这样
                                    arr.$set(1, "new hi");

                                    var arr = ["hello","hi","easybui"];

                                     1.5.3 版本以后可以这样
                                    arr.$set("hi", "new hi");

                                    例子3: 修改对象值
                                    var arr = [{name:"hello"},{name:"hi"},{name:"easybui"}];

                                    // [{name:"hello"},{name:"new hi"},{name:"easybui"}]
                            
                                     1.5.3 版本以后可以这样
                                    arr.$set(1, {name:"new hi"});
                                    // arr 结果: [{name:"hello"},{name:"new hi"},{name:"easybui"}]
                            
                                    例子4: 修改对象某个字段值, 需要传多一个唯一值的字段名
                            
                                    var arr = [{name:"hello"},{name:"hi"},{name:"easybui"}];
                                    bui.array.set( arr, "hi", {name:"new hi"}, "name" );
                                    // [{name:"hello"},{name:"new hi"},{name:"easybui"}]
                                    
                                     1.5.3 版本以后可以这样
                                     // 单独修改某个值
                                    arr.$set("hello", "hi bui", "name");
                                    // 修改整个对象,不同的key值则会一起合并过去
                                    arr.$set("hi", {name:"new hi"}, "name");
                                    // arr 结果: [{name:"hello"},{name:"new hi"},{name:"easybui"}]
                            
                                 *
                                 */
                            
                                ui.array.set = function (arr, index, value, keyname) {
                                    keyname = keyname || true;
                                    if (arr && ui.typeof(arr) !== "array") {
                                        return;
                                    }
                                    var isNumber = /\d/.test(index) && typeof index === "number";
                            
                                    // 如果是数字,直接替换
                                    if (isNumber && typeof value !== "undefined") {
                                        let item = arr[index] || {};
                            
                                        // 替换
                                        if (keyname === true) {
                                            arr.splice(index, 1, value);
                                            return arr;
                                        }
                                        if (ui.typeof(value) === "object") {
                                            // 如果是对象,会覆盖相同值,并把不相同的值合并过来
                                            for (let key in value) {
                                                item[key] = value[key];
                                            }
                                        } else if (ui.typeof(value) !== "object" && typeof keyname === "string") {
                                            item[keyname] = value;
                                        }
                                        arr.splice(index, 1, item);
                                        return arr;
                                    }
                                    for (let i = 0; i < arr.length; i++) {
                                        let item = arr[i];
                                        let val = typeof keyname === "undefined" ? item : item[keyname] || item;
                                        let rlue = isNumber ? i === index : val === index;
                                        if (rlue) {
                                            if (typeof keyname === "string") {
                                                // 如果是对象,会覆盖相同值,并把不相同的值合并过来
                                                if (value && ui.typeof(value) === "object") {
                                                    for (let key in value) {
                                                        item[key] = value[key];
                                                    }
                                                } else {
                                                    item[keyname] = value;
                                                }
                                                arr.splice(i, 1, item);
                                            } else {
                                                // 非对象直接替换
                                                arr.splice(i, 1, value);
                                            }
                                            return arr;
                                        }
                                    }
                            
                                    return arr;
                                }

                                 * 修改所有相同的值
                                 * @namespace  bui
                                 * @method  setAll
                                 * @since  1.5.2
                                 * @constructor
                                 * @param {array} array [数组]
                                 * @param {number|string|object|array} key|value [ 数字索引值,或者是值, 如果是数字,设置索引, 如果是其它值则比对值 ]
                                 * @param {number|string|object} [value]  修改的值 
                                 * @param {string} [keyname] 如果比对的是对象,则需要这个键值
                                 * @return {array} [空数组]
                                 * @example
                                 *
                            
                                    例子1: 修改值等于 hi 为新值 new hi
                                    var arr = ["hello","hi","easybui","hi","hi"];
                                    bui.array.setAll( arr, "hi", "new hi");
                                    // ["hello","new hi","easybui","new hi","new hi"]
                            
                            
                                    例子2: 修改对象某个字段值, 需要传多一个唯一值的字段名
                                    var arr = [{name:"hello"},{name:"hi"},{name:"hi"},{name:"easybui"}];
                                    bui.array.setAll( arr, "hi", "new hi", "name" );
                                    // [{name:"hello"},{name:"new hi"},{name:"new hi"},{name:"easybui"}]
                                    
                            
                                    例子3: 替换对象
                                    var arr = [{name:"hello"},{name:"hi"},{name:"hi"},{name:"easybui"}];
                                    bui.array.setAll( arr, "hi", {name:"new hi","id":123}, "name" );
                                    // [{name:"hello"},{name: "new hi", id: 123},{name: "new hi", id: 123},{name:"easybui"}]
                                    
                                 *
                                 */
                                ui.array.setAll = function (arr, index, value, keyname) {
                            
                                    if (arr && ui.typeof(arr) !== "array") {
                                        return;
                                    }
                            
                                    arr.filter(function (item, i) {
                                        let val = typeof keyname === "undefined" ? item : item[keyname] || item;
                                        if (i == index || val === index) {
                                            if (keyname) {
                                                // 如果是对象,会覆盖相同值,并把不相同的值合并过来
                                                if (value && ui.typeof(value) === "object") {
                                                    for (let key in value) {
                                                        item[key] = value[key];
                                                    }
                                                } else {
                                                    item[keyname] = value;
                                                }
                                                arr.splice(i, 1, item);
                                            } else {
                                                // 非对象直接替换
                                                arr.splice(i, 1, value);
                                            }
                                        }
                                    })
                                    return arr;
                                }
                            
                                /**
                                 * 删除后返回索引, 会改变原有数组
                                 * @namespace  bui
                                 * @method  delete
                                 * @since  1.5.2
                                 * @constructor
                                 * @param {array}  array [数组]
                                 * @param {string|array} value [要删除的数据,值]
                                 * @param {string|number} [key] object的键名,非object可以不用传
                                 * @return {array} [返回删除前的所在位置]
                                 * @example
                                 *
                                    //例子1: 删除值:
                                    var arr = ["hello","bui","hi"];
                                    bui.array.delete(arr , "bui" );
                            
                            
                                    // 例子2: 删除值在哪个字段:
                                    var arr = [{ "id":1,value:"hello"},{ "id":2,value:"bui"}];
                                    bui.array.delete( arr, "bui", "value" );
                            
                                    // 例子3: 1.7.0 新增 
                                    var arr = [{ "id":1,value:"hello"},{ "id":2,value:"bui"}];
                                    bui.array.delete( arr, { "id":2,value:"bui"} );
                                    
                                 *
                                 */
                            
                                ui.array.delete = function (arr, value, key) {
                            
                                    if (arr && ui.typeof(arr) !== "array") {
                                        return;
                                    }
                            
                                    var indexs = [];
                                    var newarr = ui.array.copy(arr);
                                    if (value && ui.typeof(value) === "array") {
                                        value.forEach(function (item, i) {
                                            var removeVal = typeof key !== "undefined" ? item[key] || item : item;
                                            removeSingle(removeVal)
                                        })
                                    } else {
                                        removeSingle(value)
                                    }
                            
                                    function removeSingle(str) {
                                        var index = 0;
                                        for (let i = 0; i < newarr.length; i++) {
                                            let newindex = i - index;
                                            let hasChange = arr.length !== newarr.length;
                                            let item = hasChange ? (arr[newindex] || {}) : arr[i] || {};
                                            let val = typeof key !== "undefined" ? item[key] : item;
                                            let valname = (typeof val === "function") ? val.name.replace("bound ", "") : val;
                                            let strname = (typeof str === "function") ? str.name : str;
                            
                                            if (typeof (val) === "object" && typeof (value) === "object") {
                                                try {
                                                    let str1 = JSON.stringify(val);
                                                    let str2 = JSON.stringify(value);
                                                    if (str1 === str2) {
                                                        // 长度被修改
                                                        if (hasChange) {
                                                            arr.splice(newindex, 1);
                                                        } else {
                                                            arr.splice(i, 1);
                                                        }
                                                        indexs.push(i);
                                                        ++index;
                                                        // return;
                                                    }
                                                } catch (e) {
                            
                                                }
                                            } else if (val === str || valname === strname) {
                            
                                                // 长度被修改
                                                if (hasChange) {
                                                    arr.splice(newindex, 1);
                                                } else {
                                                    arr.splice(i, 1);
                                                }
                                                indexs.push(i);
                                                ++index;
                                                // return;
                                            }
                            
                                        }
                                    }
                                    return indexs;
                                };

                                ui.array.deleteIndex = function (arr, value) {
                            
                                    if (arr && ui.typeof(arr) !== "array") {
                                        return;
                                    }
                            
                                    var deleteArr = [];
                                    if (value && ui.typeof(value) === "array") {
                                        value.forEach(function (item, i) {
                                            deleteArr.push(arr[item]);
                                            ui.array.remove(arr, arr[item]);
                                        })
                                    } else {
                                        deleteArr.push(arr[value]);
                                        ui.array.remove(arr, arr[value]);
                                    }
                            
                                    return deleteArr;
                                };
