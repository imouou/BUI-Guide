                            /**
                             * @module Native
                             */
                            ;
                            (function (ui, $) {
                            
                                /**
                                 * ## 后退方法
                                 * 后退方法的回调,只在bui.isWebapp=false才会触发,会根据平台使用不同的后退方法
                                 * @namespace  bui
                                 * @class  back
                                 * @constructor
                                 * @param {object} [option]
                                 * @param {number} [option.index] 默认:-1 后退的层次, 原生不支持后退多层
                                 * @param {function} [option.callback] 后退以后触发
                                 * @param {string} [option.name] 1.4.1新增,后退到指定模块,单页开发才有效
                                 * @param {string} [option.param] 1.6.4新增, 单页路由且,后退传参, 在生命周期的 show回调里面可以获取到url, 可以处理多个不同后退传参,被动式接收.
                                 * @example
                            
                                 * 例子1:
                                 *    bui.back();
                                 *
                                 * 例子2:
                                 *    // 后退两层
                                 *    bui.back({ index:-2 })
                                 *
                                 */
                                ui.back = function (opt) {
                            
                                    var config = {
                                        index: -1,
                                        name: "",
                                        effect: "",
                                        delay: true,
                                        param: null,
                                        native: true,
                                        beforeBack: true,
                                        callback: null
                                    };
                            
                                    var params;
                                    if (typeof opt === "function") {
                                        config.callback = opt;
                                        params = $.extend(true, {}, config, ui.config.back);
                                    } else if (opt && ui.typeof(opt) === "object") {
                                        params = $.extend(true, {}, config, ui.config.back, opt);
                                    } else {
                                        params = $.extend(true, {}, config, ui.config.back);
                                    }
                                    // 如果没有传 needNative 以 isWebapp 的状态为主
                                    var webapp = typeof opt === "object" && typeof opt.needNative !== "undefined" ? !opt.needNative : ui.isWebapp;
                            
                                    var lastPageHistory = ui.hasRouter ? router.history && router.history.getLast() || null : null;
                                    // 能不能后退
                                    var canback = typeof params.beforeBack === "function" ? params.beforeBack.call(this, {
                                        prevTarget: null,
                                        target: lastPageHistory
                                    }) : params.beforeBack;
                            
                                    // 不允许后退
                                    if (canback === false) {
                                        return;
                                    }
                            
                                    if (opt && opt.needNative || ui.nativeRouter) {
                                        // 采用原生后退
                                        ui.native.back && ui.native.back(params);
                                        return;
                                    }
                                    // 优先使用单页,然后才是根据平台切换
                                    if (ui.hasRouter) {
                            
                                        // 如果配置是同步历史记录,由历史记录来操作
                                        if (window.router.config.syncHistory) {
                            
                                            // 后退到最后一页
                                            var routerHistory = ui.history.get(),
                                                lastLen = routerHistory.length - 1;
                            
                            
                                            // 支持指定后退到哪个模块,如果有多个模块相同,则取就近
                                            if (params.name) {
                                                // var nameIndex = ui.array.index(routerHistory,params.name,"pid");
                                                // params.index = nameIndex > 0 ? -(routerHistory.length-nameIndex-1) : -1;
                                                var nameIndex = ui.array.indexs(routerHistory, params.name, "name");
                            
                                                var nameLen = nameIndex.length;
                            
                                                // 有多个同模块则采用就近模块
                                                if (nameLen) {
                                                    params.index = -(routerHistory.length - nameIndex[nameLen - 1] - 1);
                                                } else {
                                                    params.index = -1;
                                                }
                                            }
                                            // hash后退不判断最大值,因为hash拿不到之前的历史记录,长度为1
                                            if (Math.abs(params.index) >= routerHistory.length) {
                                                params.index = -lastLen;
                                            }
                                            // hash 后退的最大值要跟 window.history.length 取最大值, 不然不进行后退
                                            if (Math.abs(params.index) >= window.history.length && window.router.config.hash) {
                                                params.index = window.router.config.reloadCache ? -lastLen : -(window.history.length - 1);
                                            }
                                            // 至少后退一层
                                            if (params.index == 0) {
                                                params.index = -1;
                                            }
                            
                                            // 如果有特效，通过sessionStorage 传给pushState
                                            if (params.effect) {
                                                sessionStorage.setItem("backEffect", params.effect);
                                            }
                                            // hash 跳转的时候,使用window.location.go() 不判断历史记录长度, window.history.length 只会累加, reloadCache 代表会有历史缓存
                                            if (window.router.config.hash && !window.router.config.reloadCache) {
                                                cacheBack();
                                                return;
                                            }
                            
                                            // 不是hash跳转的时候,要判断历史记录的长度
                                            if (routerHistory.length > 1 || window.history.length > 1) {
                                                cacheBack();
                                            }
                                        } else {
                                            // 单页后退
                                            window.router.back && window.router.back(params);
                                        }
                            
                                    } else {
                                        // 后退不刷新
                                        cacheBack();
                                    }
                            
                            
                                    var timeout = null;
                            
                                    function cacheBack() {
                            
                                        // ios 12 ios 13 对定时器的处理不同, 模块获取应该在外面才能确保正确
                                        // 单页要获取到上一个模块, 多页获取最后一个模块
                                        var lastPage = ui.hasRouter && ui.history.getPrev() || {};
                                        var lastPageInstance = lastPage["exports"] || {};
                            
                                        // 后退
                                        window.history.go(params.index);
                            
                                        // 后退传参
                                        if (ui.hasRouter && params.param) {
                                            lastPage.param = params.param;
                                            var url = window.location.origin + window.location.pathname + "#" + lastPage.pid;
                                            var newurl = ui.setUrlParams(url, lastPage.param);
                                            lastPage.component = {};
                                            lastPage.page = {};
                                            window.history.hasOwnProperty("replaceState") && window.history.replaceState(lastPage, null, newurl);
                                        }
                                        // 延时执行,因为页面切换有动画200毫秒的延迟,正常不使用bui.refresh则不需要延迟.
                                        if (timeout) {
                                            clearTimeout(timeout);
                                        }
                                        // 单页后退刷新的时候,使用历史记录,需要延迟才能保持跟一致
                                        if (params.delay && params.callback) {
                                            timeout = setTimeout(function () {
                            
                                                // 统一单页多页的后退的回调
                                                if (ui.hasRouter) {
                                                    params.callback.call(router, lastPageInstance);
                                                } else {
                                                    // 多页直接执行
                                                    params.callback.call(ui, {});
                                                }
                            
                                            }, 500)
                                        } else {
                                            params.callback && params.callback();
                                        }
                            
                                    }
                            
                                };
                            
                            
                                return ui;
                            
                            })(window.bui || {}, window.libs);
