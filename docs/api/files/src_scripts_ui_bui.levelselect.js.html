                                    function getArrayLayer(arr, field, index = 1) {
                                        let newIndex = index;
                            
                                        arr.forEach(function (item, i) {
                                            let tempIndex = index
                            
                                            if (item[field]) {
                                                tempIndex = getArrayLayer(item[field], field, index + 1)
                                                if (tempIndex > newIndex) {
                                                    newIndex = tempIndex
                                                }
                                            }
                                        })
                                        return newIndex
                                    }
                                    // 获取层级
                                    function getDataLayer(data, field) {
                                        let levels = 0;
                            
                                        if (typeof (field) === "string") {
                                            levels = getArrayLayer(data, field);
                            
                                        } else if (ui.typeof(field) === "array") {
                            
                                            field.forEach(function (it, n) {
                                                levels += getArrayLayer(data, field[n]);
                                            })
                                        }
                            
                                        return levels;
                                    }

                                    function show(callback) {

                                        if (typeof callback === "function") {
                                            _opt.callback = function () {
                                                callback && callback.call(module);
                                                trigger.call(module, "show");
                                            };
                                        } else if (typeof callback === "string") {
                                            _opt.effect = callback;
                                        } else if (typeof callback === "object") {
                                            _opt = callback;
                                        }
                            
                                        uiDialogCity && uiDialogCity.open(_opt);
                                    }

                                        if (typeof callback === "function") {

                                        citySelect.value([{value:"广东"},{value:"广州市"},{value:"天河区"}]);
                            
                                     */
                                    function value(vals) {
                                        if (typeof vals === "undefined") {
                                            return selectValue;
                                        } else {
                                            // 在初始化数据以后赋值
                                            if (vals && ui.typeof(vals) === "array") {
                                                vals.forEach(function (item, i) {
                                                    if (typeof item === "string") {
                                                        selectInstance[i].value(item)
                                                    } else if (typeof item === "object") {
                                                        selectInstance[i].text(item.name || "");
                                                        selectInstance[i].value(item.value || "");
                                                    }
                                                })
                                            } else {
                                                ui.showLog("levelselect.value() 参数是一个数组类型");
                                            }
                                        }
                                    }

                                    function reset() {

                                        })

                                    }

                                            navstyle = `style="display:block;white-space:nowrap;text-align:left;"`

                                        var html = '',
                                            i = 0;
                            
                                        let hasNav = opt.showValue ? templateNav(opt) : "";
                                        // 没有弹窗
                                        if (!opt.popup) {
                                            html += '<div id="' + gid + '" class="bui-levelselect">';
                                        }
                                        // 展示才对
                                        html += hasNav;
                            
                                        html += '<div id="' + gidSlide + '" class="bui-tab bui-levelselect-tab">';
                                        html += '    <div class="bui-tab-main">';
                                        html += '        <ul>';
                            
                                        // 遍历层级

                                            html += '            <li>';
                                            html += '                <div class="select-level-' + i + '"></div>';
                                            html += '            </li>';
                                        }
                                        html += '        </ul>';
                                        html += '    </div>';
                                        html += '</div>';
                                        // 没有弹窗
                                        if (!opt.popup) {
                            
                                            html += '</div>';
                                        }
                                        return html;
