                            /**
                             * @module Native
                             */
                            ;
                            (function (ui, $) {
                            
                                /**
                                 * 打开第三方应用, 参数needNative 为true 才能调用原生打开, false打开网址
                                 * @namespace  bui
                                 * @class  run
                                 * @param {object} option
                                 * @param {string} option.id 应用的id,如果是"http://"则打开浏览器
                                 * @param {object} option.data 启动参数(bingotouch)
                                 * @param {string} option.name 应用名称(dcloud)
                                 * @param {boolean} [option.needNative]  1.5.1新增, 是否使用原生上传, 不受bui.isWebapp状态决定 默认 true|false 通过这个切换对应的效果,可以通过全局配置 bui.config.run = {needNative:true} 
                                 * @param {function} option.onFail 失败回调(dcloud)
                                 * @constructor
                                 * @example
                                 *     bui.run({id: "http://www.easybui.com/demo/api/"});
                                 *
                                 */
                                ui.run = function (option) {
                                    var params = {};
                                    var config = {
                                        id: "",
                                        name: "",
                                        data: null,
                                        onFail: null,
                                        native: true,
                                        needNative: true
                                    }
                                    if (typeof option === "string") {
                                        params.id = option;
                                    } else if (option && ui.typeof(option) === "object") {
                                        params = $.extend(true, {}, config, ui.config.run, option);
                                    }
                                    var sid = String(params.id);
                                    // 如果没有传 needNative 以 isWebapp 的状态为主
                                    var isWebapp = typeof option === "object" && typeof option.needNative !== "undefined" ? !option.needNative : ui.isWebapp; // (params.native && ui.isWebapp) || (!ui.isWebapp && !params.native)
                            
                                    if (!option.needNative) {
                                        if (sid.indexOf("http://") > -1 || sid.indexOf("https://") > -1) {
                                            var newurl = ui.setUrlParams(params.id, params.data);
                                            if (ui.platform.isIos()) {
                                                window.location.href = newurl;
                                                return;
                                            }
                                            // 安卓或chrome
                                            var open_link = window.open('', '_blank') || window.open('', '_newtab');
                                            open_link.location.href = newurl;
                                        }
                                    } else {
                                        ui.native.run && ui.native.run(params);
                                    }
                                }
                            
                            
                                return ui;
                            
                            })(window.bui || {}, window.libs);
