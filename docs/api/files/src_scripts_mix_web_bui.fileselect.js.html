                                    function resize(opt) {
                                        var params = $.extend(true, {
                                            destinationType: "file", // 转换后的目标类型
                                            data: null,
                                            mimeType: "",
                                            quality: 0.8,
                                            width: 800,
                                            height: 800,
                                            needCompress: true, // 区分是否是选择以后自动压缩, add 默认只是添加图片,不压缩
                                            needChangeData: false, // 是否要修改源文件数据,上传的时候才是修改后的数据, add方法添加的是true
                                            onSuccess: null,
                                            onFail: null
                                        }, options, opt);
                            
                                        params.data = params.data && ui.typeof(params.data) === 'array' ? params.data.pop() : params.data;
                            
                                        var newimg = new Image;
                                        var reader = new FileReader();
                            
                                        // 图片压缩以后处理
                                        newimg.onload = function () {
                                            var file = this;
                                            // 图片原始尺寸
                                            var originWidth = this.width;
                                            var originHeight = this.height;
                                            var files = null;
                            
                                            // 最大尺寸限制
                                            var maxWidth = params.width,
                                                maxHeight = params.height;
                            
                                            // 目标尺寸
                                            var targetWidth = originWidth,
                                                targetHeight = originHeight;
                                            // 图片尺寸超过设置的尺寸的限制
                                            if (params.needCompress && (originWidth > maxWidth || originHeight > maxHeight)) {
                                                if (originWidth / originHeight > maxWidth / maxHeight) {
                                                    // 更宽，按照宽度限定尺寸
                                                    targetWidth = maxWidth;
                                                    targetHeight = Math.round(maxWidth * (originHeight / originWidth));
                                                } else {
                                                    targetHeight = maxHeight;
                                                    targetWidth = Math.round(maxHeight * (originWidth / originHeight));
                                                }
                                            }
                            
                                            // canvas对图片进行缩放
                                            canvas.width = targetWidth;
                                            canvas.height = targetHeight;
                                            // 清除画布
                                            context.clearRect(0, 0, targetWidth, targetHeight);
                                            // 图片压缩
                                            context.drawImage(newimg, 0, 0, targetWidth, targetHeight);
                            
                                            // 返回什么数据
                                            var base64 = canvas.toDataURL(params.mimeType || params.data && params.data.type || "image/jpeg", params.quality);
                                            var name = getName(params.data.name) || "";
                                            switch (params.destinationType) {
                                                case "file":
                                                    // dcloud 转换会报错, needCompress 需要为false,或者使用base64直接上传
                                                    files = toFile(base64, name);
                                                    break;
                                                case "blob":
                                                    files = toBlob(base64);
                                                    break;
                                                case "data":
                                                    files = base64;
                                                    break;
                                            }
                            
                                            // 从add过来才有这个参数
                                            if (params.needChangeData) {
                                                // 默认展示不影响最终上传是以文件的形式
                                                uploadData = files;
                                                var index = ui.array.index(fileData, params.data.name, "name");
                            
                                                if (index > -1) {
                                                    fileData[index]["data"] = files;
                                                    files.size && (fileData[index]["size"] = files.size);
                                                    files.type && (fileData[index]["type"] = files.type);
                                                }
                                            }
                            
                                            params.onSuccess && params.onSuccess.call(module, files, fileData)
                            
                                            return files;
                                        }
                            
                                        newimg.onerror = function () {
                                            // 错误
                                            params.onFail && params.onFail.call(module, newimg.src)
                                        }
                            
                                        // 设置图片的地址,触发newimg.onload
                                        reader.onload = function (evt) {
                                            newimg.src = evt.target.result;
                                        }
                            
                                        try {
                                            var type = params.data && params.data.type || "";
                                            if (typeof type === "object") {
                                                type = type.type || "";
                                            } else {
                                                type = type || "";
                                            }
                            
                                            // 选择的文件是图片
                                            if (params.data && type.indexOf("image") == 0) {
                                                // 读取file,触发img的压缩
                                                reader.readAsDataURL(opt.data);
                                            }
                                        } catch (e) {
                                            ui.showLog(e)
                                        }
                            
                            
                                    }

                                    /*
                                     * 将base64转换成file对象, 这些转换不会影响到控件的 value 得到的值.
                                     *  @method toFile
                                     *  @param {object} data [ 图片base64位的值 ]
                                     *  @param {string} [filename]  图片转成文件的文件名,默认 file 
                                     *  @chainable
                                     *  @example
                                     *
                                        uiFileSelect.toBase64({
                                            onSuccess: function(data){
                                                var file = uiFileSelect.toFile(data,"resize")
                            
                                            }
                                        })

                                    function toFile(dataurl, filename) {
                                        try {
                            
                                            filename = filename || 'file';
                                            var arr = dataurl.split(',');
                                            var arrmatch = arr[0].match(/:(.*?);/) || [];
                                            var mime = arrmatch[1] || "image/jpeg";
                                            var suffix = mime.split('/')[1];
                                            var bstr = atob(arr[1]);
                                            var n = bstr.length;
                                            var u8arr = new Uint8Array(n);
                                            while (n--) {
                                                u8arr[n] = bstr.charCodeAt(n);
                                            }
                                            return new File([u8arr], filename + "." + suffix, {
                                                type: mime
                                            });
                                        } catch (e) {
                                            ui.showLog(e);
                                        }
                                    }

                                    /*
                                     * 将base64转换成文件流,

                                    function toBlob(dataurl) {
                                        var arr = dataurl.split(','),
                                            mime = arr[0].match(/:(.*?);/)[1],
                                            bstr = atob(arr[1]),
                                            n = bstr.length,
                                            u8arr = new Uint8Array(n);
                                        while (n--) {
                                            u8arr[n] = bstr.charCodeAt(n);
                                        }
                                        return new Blob([u8arr], {
                                            type: mime
                                        });
                                    }
                            
                                    //渲染web的添加input 隐藏
                                    function renderInputFile() {
                                        if ($btnFile == undefined) {
                            
                                            guid = bui.guid();
                                            var from = options.from ? 'capture="' + options.from + '"' : '';
                                            // display:none 有可能会导致部分手机无法触发选择图片功能
                                            var needMultiple = options.multiple ? "multiple" : "";
                                            var html = '<input type="file" accept="' + mediaType + '" name="' + options.name + '" id="' + guid + '" ' + from + ' ' + needMultiple + ' style="display:none" />';
                                            $(options.appendTo).append(html);
                            
                                            $btnFile = $("#" + guid) || $('input[name="' + options.name + '"]');
                                        }
                                    }

                                    function changeMultiple(){
                                        $btnFile.attr("multiple","true");
                                    }
                            
                                    // 改为单选
                                    function changeSingle(){
                                        $btnFile.removeAttr("multiple");
                                    }
                            
                                    function addMore(opt){
                                        
                                        opt.multiple = true;
                            
                                        add(opt);
                            
                                        return this;
                                    }
                            
                                    /**
                                     * 删除选择的文件, 安卓选择后的文件名自动更改为 resize.jpg
                                     *  @method remove
                                     *  @param {string} name [ 要删除的图片名字带后缀名 ]
                                     *  @param {string} [key]  图片名字在哪个字段, 默认是 "name"
                                     *  @chainable
                                     *  @example
                                     *
                                        uiFileSelect.remove("resize.jpg");
                            
                                     */
                                    function remove(name, key) {
                                        var key = key || "name";
                            
                                        if (webapp) {
                                            if (typeof name == "string") {
                                                fileData.length && ui.array.remove(fileData, name, key);
                            
                                                // console.log(fileData, name)
                                                // 删除某个选择文件后需要修改上传的文件值为新的最后一条数据;
                                                var lastData = fileData.length ? fileData[(fileData.length - 1)]["data"] : null;
                                                uploadData = lastData;
                            
                                                // 重置input,不然相同文件不能选择
                                                $btnFile[0].value = "";
                                                return fileData;
                                            }
                                        } else {
                            
                                            fileData = uiFileselect.remove(name, key);
                            
                                            return fileData;
                                        }
                            
                                        return this;
                                    }
                            
                                    /**
                                     * 清空所有选择的文件
                                     *  @method clear
                                     *  @example
                                     *
                                        uiFileSelect.clear();
                            
                                     */
                                    function clear() {
                            
                                        fileData = [];
                                        uploadData = null;
                            
                                        if (webapp) {
                                            // 重置input,不然相同文件不能选择
                                            $btnFile[0].value = "";
                                        } else {
                                            uiFileselect.clear();
                                        }
                            
                                        return fileData;
                                    }
                            
                                    /**
                                     * 获取所有选择的文件, 原生默认只返回最后一个选择的文件
                                     *  @method data
                                     *  @example
                                     *
                                        var data = uiFileSelect.data();
                                        // 返回 [{ name: "base64.jpg", data: null, type: "jpg" }]
                                        // 其中 data为要上传的文件值
                            
                                     */
                                    function data() {
                            
                                        if (webapp) {
                                            fileData = fileData;
                                        } else {
                                            fileData = uiFileselect.data();
                                        }
                                        return fileData;
                                    }
                            
                                    /**
                                     * 获取最后选择的文件
                                     *  @method value
                                     *  @example
                                     *
                                        var value = uiFileSelect.value();
                                        // 返回值 value为要上传的文件值, web上传跟原生上传得到的值不一样
                            
                                     */
                                    function value() {
                                        if (webapp) {
                                            uploadData = uploadData;
                                        } else {
                                            uploadData = uiFileselect.value();
                                        }
                                        return uploadData;
                                    }
                            
                                    /**
                                     * 获取最后选择的文件名,删除的时候
                                     *  @method currentName
                                     *  @since 1.5.2
                                     *  @example
                                     *
                                        var currentName = uiFileSelect.currentName();
                            
                                     */
                                    function currentName() {
                                        var n = "";
                                        if (webapp) {
                                            n = fileData[fileData.length - 1]["name"];
                                        } else {
                                            n = uiFileselect.value()["name"];
                                        }
                                        return n;
                                    }
                                    /**
                                     * 获取最后选择的文件名,删除的时候
                                     *  @method currentIndex
                                     *  @since 1.5.2
                                     *  @example
                                     *
                                        var currentIndex = uiFileSelect.currentIndex();
                            
                                     */
                                    function currentIndex() {
                                        var n = "";
                                        if (webapp) {
                                            n = fileData.length - 1;
                                        } else {
                                            n = uiFileselect.data().length - 1;
                                        }
                                        return n;
                                    }
                            
                            
                                    return module;
                                };
