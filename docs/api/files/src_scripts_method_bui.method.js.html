                            /**
                             * <h3>常用方法库</h3> 
                             * <h5>数组比对</h5>
                             * {{#crossLink "bui.array"}}{{/crossLink}}: 常用数组比对 <br>
                             * <h5>对象存储</h5>
                             * {{#crossLink "bui.storage"}}{{/crossLink}}: 常用于搜索历史记录, 可以存储普通数据也可以存储对象数据, 返回值必定是数组
                             * <h5>平台检测</h5>
                             * {{#crossLink "bui.platform"}}{{/crossLink}}: 属于什么系统跟浏览器的检测 <br>
                             * <h5>其它常用</h5>
                             * {{#crossLink "bui.guid"}}{{/crossLink}}: 生成唯一id <br>
                             * {{#crossLink "bui.typeof"}}{{/crossLink}}: 检测对象类型 <br>
                             * {{#crossLink "bui.checkVersion"}}{{/crossLink}}: 获取最新版本 <br>
                             * 
                             * @module Method
                             */
                            ;
                            (function (ui, $) {
                            
                            
                            
                                /*
                                 * 常用页面交互操作,传值,取值
                                 * -------------------------------------------------------
                                 */
                                /*
                                 * 设置传过去的地址转码,默认中文转码
                                 * @namespace  bui
                                 * @class  setUrlParams
                                 * @constructor 
                                 * @param {string} url [url地址]
                                 * @param {object} param [object参数]
                                 * @param {boolean} bool [ true | false 是否转码,默认转码 true]
                                 * @return {string} [返回字符串]
                                 * @example
                                 *       var url = bui.setUrlParams("index.html",{"id":"111"})
                                 *       console.log(url) //index.html?id=111
                                 * 
                                 */
                                ui.setUrlParams = function (url, param, bool) {
                            
                                    var bool = bool == false ? false : true,
                                        param = typeof (param) == 'object' ? param : {},
                                        newurl;
                            
                                    var params = ui.unit.objectToKeyString(param, bool);
                            
                                    //有对象字面量才修改url地址
                                    if (params == "") {
                                        newurl = url;
                                    } else {
                                        newurl = url.indexOf('?')>-1 ? url+"&"+params : url + '?' + params; //取第1位到结束的字符
                                    }
                            
                                    return newurl;
                                };
                                /*
                                 * 获取通过网址传递的参数
                                 * @namespace  bui
                                 * @class  getUrlParams
                                 * @constructor 
                                 * @param {boolean} bool [ true | false 是否需要中文转码,默认转码 true ]
                                 * @return {object} [返回网址的参数]
                                 * @example
                                 *       var url = bui.getUrlParams()
                                 *        console.log(url) //{id:111}
                                 * 
                                 */
                                ui.getUrlParams = function (bool) {
                            
                                    var bool = bool == false ? false : true;
                                    var url = window.location.href; //获取url中"?"符后的字串
                                    var hasParam = url.indexOf("?");
                                    var obj = {};
                                    if (hasParam > -1) {
                                        var str = url.substr(hasParam + 1);
                                        var strIndex = str.indexOf("?");
                                        // 微信返回的url会有2个问号的情况
                                        if( strIndex > -1 ){
                                            var strs = str.split("?");
                                            var strBefore = strs[0];
                                            var strAfter = strs[1];
                            
                                            var obj1 = ui.unit.keyStringToObject(strBefore, bool);
                                            var obj2 = ui.unit.keyStringToObject(strAfter, bool);
                            
                                            obj = $.extend(true,obj1,obj2);
                            
                                        }else{
                                            obj = ui.unit.keyStringToObject(str, bool);
                                        }
                            
                                        // 把模块也解析
                                        // var hash = url.split("#")[1] || "";
                                        // obj.module = hash.split("?")[0] || "main";
                                    }
                                    return obj;
                                };
                            
                                /*
                                 * 获取url的指定字段
                                 * @namespace  bui
                                 * @class  getUrlParam
                                 * @constructor 
                                 * @param {string} name [指定某个参数名]
                                 * @return {string} [返回参数值]
                                 * @example
                                 *       var val = bui.getUrlParam("id");
                                 *        console.log(val) //111
                                 * 
                                 */
                                ui.getUrlParam = function (name) {
                            
                                    var url = window.location.href;
                                    //获取url中"?"符后的键值
                                    var reg = new RegExp("(^|&|.)" + name + "=([^&]*)(&|$)", "i");
                                    var r = url.substr(1).match(reg);
                                    if (r != null) return decodeURIComponent(r[2]);
                                    return null;
                            
                                };
                            
                            
                                /**
                                 * 判断类型, 不能用来判断jq对象,会返回一个数组
                                 * @namespace  bui
                                 * @class  typeof
                                 * @constructor 
                                 * @param {string} value [要判断的对象]
                                 * @return {string} [返回 json | regexp | array | object | function | string | number | boolean | null | undefined ]
                                 * @example
                                 *       var type = bui.typeof({"id":"123"});
                                 *        console.log(type) // "object"
                                 * 
                                 */
                                ui.typeof = function (value) {
                            
                                    var valueStr = Object.prototype.toString.call(value).slice(8, -1);
                            
                                    valueStr = valueStr.toLowerCase();
                            
                                    return valueStr;
                                };
                            
                                /**
                                 * 获取及设置控件默认参数
                                 * @namespace  bui
                                 * @class  setting
                                 * @since  1.5.5
                                 * @constructor 
                                 * @param {string} [key] 控件名
                                 * @param {string} [value] 修改的全局参数
                                 * @return {string} [返回 json ]
                                 * @example
                                   
                                 *       // 例子1: 获取所有默认配置
                                 *       var setting = bui.setting();
                                 * 
                                 *       // 例子2: 获取某个配置
                                 *       var ajaxSetting = bui.setting("ajax");
                                 * 
                                 *       // 例子3: 设置更改所有ajax 加上headers的默认配置
                                 *       var ajaxSetting = bui.setting("ajax",{headers:{token:"bui"}});
                                 * 
                                 *       // 例子4: 修改所有alert的宽高
                                 *       var alertSetting = bui.setting("alert",{"width":600,height:500});
                                 *       
                                 *       bui.alert("bui will be better!")
                                 * 
                                 */
                                ui.setting = function (key, value) {
                            
                                    if (typeof key === "string" && typeof value === "object") {
                                        return ui.config[key] = value;
                                    } else if (typeof key === "string" && typeof value === "undefined") {
                                        return ui.config[key];
                                    } else {
                                        return ui.config;
                                    }
                                };
                            
                            
                                /**
                                 * 执行ui的某个指令
                                 * @namespace  bui
                                 * @class  cmd
                                 * @since  1.6.6
                                 * @constructor 
                                 * @param {string} [uid] 初始化控件的时候，可以执行某个命令
                                 * @return {object}} [返回实例 ]
                                 * @example
                                   
                                 *       // 例子: 
                                        // 实例初始化
                                        var uiAccordion = bui.accordion({uid:"uiAccordion"});
                                         
                                         // 1.获取到实例
                                 *       var accordion = bui.cmd("uiAccordion");
                                         
                                         // 2.执行实例某个方法
                                 *       bui.cmd("uiAccordion:show");
                            
                                         // 3.执行实例某个方法并传参数
                                 *       bui.cmd("uiAccordion:show",{});
                                 * 
                                 */
                                 ui.cmd = function (uid) {
                            
                                    var uids = uid.split(":") || [];
                                    var uiInstance = ui.history.getUI(uids[0]);
                                    
                                    if( uids.length > 1){
                                        var args = [].slice.apply(arguments);
                                            args.shift();
                                            var uiMethod = uiInstance[uids[1]];
                            
                                            if( !uiMethod ){
                                                ui.showLog(`找不到该实例的${uids[1]}方法`);
                                                return;
                                            }
                            
                                            return uiMethod.apply(uiInstance,args);
                                    }else{
                                        return uiInstance;
                                    }
                                };
                            
                            
                            
                                return ui;
                            
                            })(window.bui || {}, window.libs);
