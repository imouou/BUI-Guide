                            /**
                             * 核心
                             * @module Core
                             */
                            
                            ;
                            (function (ui, $) {
                            
                                /**
                                 * [单页跟多页的公共历史记录,反映模块的加载关系, router.history 将废弃使用 ]
                                 * @namespace bui
                                 * @class history
                                 * @since 1.5.6
                                 * @static
                                 * @example
                                 *
                                    // 获取历史记录
                                    var history = bui.history.get()
                                 *
                                 */
                            
                                ui.history = (function () {
                            
                                    // 把单页多页的路由历史抽出来
                                    var historys = [];
                            
                                    var module = {};
                            
                                    /**
                                     * 获取所有历史记录
                                     *  @method get
                                     *  @param {string} [value] 通过模块名获取某个历史记录,或者历史记录
                                     *  @param {string} [key] 默认:"name", 也可以传页面id
                                     *  @return {array} option [返回数组]
                                     *  @example
                             
                                    *  方法:
                                    *
                                        var allHistory = bui.history.get();
                                        
                                       方法2:
                                        // 通过模块名获取某个历史记录
                                        bui.history.get("pages/login/login")
                             
                                       方法3:
                                        // 通过页面id获取某个历史记录
                                        bui.history.get("buife172-aafb-0295-180b-6bc17fbd1cfb","id")
                                    *
                                    *
                                    */
                                    module.get = function (val, key) {
                                        if (typeof val === "undefined") {
                                            return historys;
                                        } else {
                                            let keyname = key || "name" || "pid";
                                            return ui.array.get(historys, val, keyname);
                                        }
                                    }
                            
                                    /**
                                     * 获取最后的历史记录
                                     *  @method getLast
                                     *  @param {string} [field] 返回最后一条记录的某个key值
                                     *  @return {object} option 返回对象
                                     *  @example
                             
                                    *  方法:
                                    *
                                        var lastHistory = bui.history.getLast();
                             
                                        方法2: 1.6.x
                                        var currentHistory = bui.history.getLast("exports");
                                        
                                    *
                                    *
                                    */
                                    module.getLast = function (field) {
                                        var last = historys.length - 1;
                                        var lastHistory = historys[last] || {};
                            
                                        if (field) {
                                            return lastHistory[field];
                                        } else {
                                            return lastHistory;
                                        }
                                    }
                            
                                    /**
                                     * 获取上一页历史记录
                                     *  @method getPrev
                                     *  @since 1.6.0
                                     *  @param {string} [field] 返回最后一条记录的某个key值
                                     *  @return {object} option 返回对象
                                     *  @example
                             
                                    *  方法:
                                    *
                                        var lastHistory = bui.history.getPrev();
                             
                                        方法2: 1.6.x
                                        var currentHistory = bui.history.getPrev("exports");
                                        
                                    *
                                    *
                                    */
                                    module.getPrev = function (field) {
                            
                                        var last = historys.length - 2;
                                        var lastHistory = historys[last] || {};
                            
                                        if (field) {
                                            return lastHistory[field];
                                        } else {
                                            return lastHistory;
                                        }
                                    }
                            
                            
                            
                                    /**
                                    *  综合的获取参数方法, 如果模块会被多种方式加载, 则应该这个方法 bui.history.getParams
                                    *  @method getParams
                                    *  @param {string} [type]  参数的类型, 如果不传则获取多种方式,依次获取, component, page, url, 也可以指定某种类型 page(弹窗加载) | component(局部加载的参数) | url(路由跳转的参数) 
                                    *  @example
                             
                                    * 方法1: 如果不确定该模块会被哪种方式使用, 则可以使用这种方式获取. module.id 在loader.define(function(req,ext,module){ }) 的第三个参数
                                    
                                    var params = bui.history.getParams(module.id);
                                    
                                    *  方法2: 获取单页路由或者多页路由的跳转传参 bui.load({url:"",param:{}})
                                    
                                    var params = bui.history.getParams("url");
                             
                                    *  方法3: 获取以bui.page({url:"",param:{}}) 加载的的参数, 通过以下方式获取
                                    
                                    var pageParams = bui.history.getParams("page");
                             
                                    *  方法4: 获取以router.loadPart, loader.load , component标签 加载的的参数, 通过以下方式获取
                             
                                    var partParams = bui.history.getParams("component");
                             
                                    *
                                    *
                                    */
                                    module.getParams = function (type) {
                                        var lastHistory = this.getLast();
                                        var param = {};
                            
                                        switch (type) {
                                            case "page":
                                                param = this.getPageParam() || param;
                                                break;
                                            case "part":
                                            case "component":
                                                param = this.getComponentParam() || param;
                                                break;
                                            case "url":
                                                param = lastHistory.param || bui.getUrlParams();
                                                break;
                                            default:
                                                param = this.getComponentParam(type) || this.getPageParam(type) || lastHistory.param || bui.getUrlParams();
                                                break;
                                        }
                                        return param;
                                    }
                            
                                    /*
                                    * 设置组件的参数,挂载到历史记录的最后一页 components
                                    *  @method setComponent
                                    *  @param {object} option 参数是一个对象
                                    *  @param {string} id  id不含#
                                    *  @param {string} name 模块名 
                                    *  @param {string} url 模块指向的模板地址 
                                    *  @param {object} param 上一个模块传的参数 
                                    *  @example
                             
                                    *  方法1: 获取单页路由或者多页路由的跳转传参
                                    
                                        bui.history.setComponent({
                                            id: "id",
                                            name: "模块名",
                                            url: "模板的地址",
                                            param: {}
                                        });
                             
                                    
                                    *
                                    *
                                    */
                            
                                    module.setComponent = function (value) {
                                        let lastHistory = this.getLast();
                                        let keyname = value.param ? Object.keys(value.param) : [];
                                        // 转布尔值
                                        if (keyname.length) {
                                            value.param = ui.unit.changeParamBoolean(value.param);
                                        }
                                        lastHistory["component"][value.id] = value;
                                        return value;
                                    }
                            
                                    /*
                                    * 无需自行设定, 这是每个页面的实例，扁平化的，根据id查找
                                    *  @method setUI
                                    *  @param {object} option 参数是一个对象
                                    *  @param {string} uid 要调用的唯一名称 
                                    *  @param {string} ui 实例
                                    *  @example
                             
                                    *  示例：
                                    
                                        // 第1种，
                                        var uiNumber = bui.number({id:"#id",uid:"num"});
                             
                                        // 第2种，内部已经初始化一次
                                        bui.history.setUI({
                                            uid: "num",
                                            ui: uiNumber,
                                        });
                             
                                    
                                    *
                                    *
                                    */
                            
                                    module.setUI = function (opt) {
                                        // 控件存在相互嵌套的情况，会造成id指向问题，这里采用控件名分类，再从id查找
                                        var lastHistory = this.getLast();
                                        var uid = opt.uid;
                            
                                        var uiTarget = lastHistory["ui"];
                            
                                        uiTarget[uid] = opt.ui;
                            
                                        return module;
                                    }
                            
                                    /**
                                    * 需要在控件初始化的时候，传uid参数，才能获取得到。
                                    *  @method getUI
                                    *  @since 1.6.6
                                    *  @param {string} uid uid,组件传了uid，才可以通过这个方法获取 
                                    *  @example
                             
                                    *  示例：
                            
                                        var uiNumber = bui.history.getUI("num");
                             
                                    *
                                    *
                                    */
                            
                                    module.getUI = function (uid) {
                                        if (!uid) {
                                            return ui.showLog("uid不能为空，指ui自定义唯一的名字");
                                        }
                                        var lastHistory = this.getLast();
                                        var uiTarget = lastHistory["ui"];
                                        return uiTarget[uid] || null;
                                    }
                            
                                    /*
                                    * 设置组件的参数,挂载到历史记录的最后一页 components
                                    *  @method getComponentParam
                                    *  @param {string} [id] 获取某个id下的参数,一般不需要传
                                    *  @example
                             
                                    *  方法1: 获取单页路由或者多页路由的跳转传参
                                    
                                        var param = bui.history.getComponentParam();
                                    
                                    *
                                    *
                                    */
                                    module.getComponentParam = function (id) {
                                        var lastHistory = this.getLast();
                                        var hasCurrentPage = id || lastHistory["currentComponent"];
                                        var param = hasCurrentPage && lastHistory["component"][hasCurrentPage] && lastHistory["component"][hasCurrentPage]["param"] || null;
                                        return param;
                                    }
                            
                                    /**
                                    * 获取编译后的组件实例
                                    *  @method getComponent
                                    *  @param {string} id 获取编译后的组件实例,如果有return, 就能拿到该对象
                                    *  @param {string} [field] 默认: exports 字段, 可以不传 
                                    *  @return {object}
                                    *  @example
                             
                                    *  方法1: 获取编译后的实例
                                        <component id="list" name="pages/list/index"></component>
                             
                                        var instance = bui.history.getComponent("list");
                                    
                                    *
                                    *
                                    */
                                    module.getComponent = function (id, field) {
                                        if (typeof id === "undefined") {
                                            ui.showLog("必须传component id 才能获取实例");
                                            return;
                                        }
                                        field = field || "exports";
                                        var lastHistory = this.getLast();
                            
                                        var instance = lastHistory["component"][id] && lastHistory["component"][id][field] || null;
                                        return instance;
                                    }
                            
                                    /**
                                    * 获取编译后的多个组件实例
                                    *  @method getComponents
                                    *  @param {string} name 获取编译后的组件实例,如果有return, 就能拿到该对象
                                    *  @param {string} [field] 默认: exports 字段, 可以不传 
                                    *  @return {array}
                                    *  @example
                             
                                    *  方法1: 获取编译后的实例
                                        <component id="list" name="pages/list/index"></component>
                                        <component id="list1" name="pages/list/index"></component>
                            
                                        // 返回数组
                                        var instances = bui.history.getComponent("pages/list/index");
                                    
                                    *
                                    *
                                    */
                                    module.getComponents = function (id, field) {
                                        if (typeof id === "undefined") {
                                            ui.showLog("必须传component name 才能获取实例");
                                            return;
                                        }
                                        field = field || "exports";
                                        var lastHistory = this.getLast();
                                        let comp = lastHistory["component"];
                                        let keys = Object.keys(comp);
                                        let comps = [];
                                        for (let i = 0; i < keys.length; i++) {
                                            let item = comp[keys[i]];
                                            if (item.name === id) {
                                                comps.push(item)
                                            }
                                        }
                            
                                        return comps;
                                    }
                            
                                    /**
                                    * 获取父组件实例, 在第一层component获取到的是页面组件
                                    *  @method getParentComponent
                                    *  @since 1.6.2
                                    *  @example
                             
                                    *  方法1: 
                                        var instance = bui.history.getParentComponent();
                                    
                                    *
                                    *
                                    */
                                    module.getParentComponent = function () {
                            
                                        var comParam = this.getComponentParam();
                            
                                        return comParam && comParam.parentId && this.getComponent(comParam.parentId) || this.getLast("exports");
                                    }
                            
                                    /**
                                    * 获取父组件实例, 在第一层component获取到的是页面组件
                                    *  @method getParentParams
                                    *  @since 1.6.2
                                    *  @example
                             
                                    *  方法1: 
                                        var instance = bui.history.getParentParams();
                                    
                                    *
                                    *
                                    */
                                    module.getParentParams = function () {
                            
                                        var comParam = this.getComponentParam();
                            
                                        return this.getParams(comParam.parentId) || this.getLast("param") || null;
                                    }
                            
                            
                                    /**
                                    * 获取插入页面的实例
                                    *  @method getPage
                                    *  @param {string} id id 或者模块名, 不传则是最后一次打开的实例 
                                    *  @param {string} [field] 默认: exports 字段, 可以不传 
                                    *  @example
                             
                                    *  方法: 
                             
                                        var instance = bui.history.getPage("pages/list");
                                    
                                    *
                                    *
                                    */
                                    module.getPage = function (id, field) {
                                        if (typeof id === "undefined") {
                                            ui.showLog("必须传page id 或者 模块名才能获取实例");
                                            return;
                                        }
                                        field = field || "exports";
                                        var lastHistory = this.getLast();
                                        var instance = null;
                                        var pages = lastHistory["page"];
                                        if (pages.hasOwnProperty(id)) {
                                            instance = pages[id] && pages[id][field] || {};
                                        } else if (id) {
                                            var keynames = Object.keys(pages);
                            
                                            for (let i = 0; i < keynames.length; i++) {
                                                let page = pages[keynames[i]] || {};
                                                if (page.name === id) {
                                                    return page[field];
                                                }
                                            }
                                        } else {
                            
                                            var currentPage = lastHistory.currentPage;
                                            instance = pages[currentPage] && pages[currentPage][field] || {};
                                        }
                                        return instance;
                                    }
                            
                            
                                    /**
                                    * 在模块内部获取对话框的实例
                                    *  @method getPageDialog
                                    *  @param {string} id 获取编译后的组件实例,如果有return, 就能拿到该对象
                                    *  @example
                             
                                    *  方法1: 获取编译后的实例
                             
                                        var instance = bui.history.getPageDialog(module.id);
                                        instance.close();
                                    
                                    *
                                    *
                                    */
                                    module.getPageDialog = function (id) {
                            
                                        if (typeof id === "undefined") {
                                            ui.showLog("必须传page id才能获取对话框信息");
                                            return;
                                        }
                            
                                        var lastHistory = this.getLast();
                                        var instance = null;
                            
                                        // if (id) {
                                        instance = lastHistory["page"][id] && lastHistory["page"][id]["dialog"] || null;
                                        return instance;
                                    }
                            
                                    /*
                                    * 设置bui.page传的一些信息,挂载到历史记录的最后一页 page
                                    *  @method setPage
                                    *  @param {object} option 参数是一个对象
                                    *  @param {string} id id不含#
                                    *  @param {string} name  模块名 
                                    *  @param {string} url 模块指向的模板地址 
                                    *  @param {object} param 上一个模块传的参数 
                                    *  @param {object} dialog dialog弹窗的实例 
                                    *  @example
                             
                                    *  方法1: 获取单页路由或者多页路由的跳转传参
                                    
                                        bui.history.setPage({
                                            id: "id",
                                            name: "模块名",
                                            url: "模板的地址",
                                            dialog: {},
                                            param: {}
                                        });
                             
                                    
                                    *
                                    *
                                    */
                                    module.setPage = function (value) {
                                        var lastHistory = this.getLast();
                                        // 合并参数
                                        if (typeof value !== "object") {
                                            return value;
                                        }
                            
                                        if (!lastHistory["page"]) {
                                            lastHistory["page"] = {}
                                        }
                                        if (value && value.hasOwnProperty("id")) {
                            
                                            var opts = $.extend({}, lastHistory["page"][value.id], value);
                                            var keyname;
                                            // 针对数组被合并问题,这里对数组单独处理
                                            for (keyname in opts.param) {
                                                if (opts.param[keyname] && ui.typeof(opts.param[keyname]) == "array") {
                                                    opts.param[keyname] = value.param[keyname] || []
                                                }
                                            }
                            
                                            opts.param = ui.unit.changeParamBoolean(opts.param);
                            
                                            // 参数如果是数组
                                            // opts.param = value.param || {};
                                            lastHistory["page"][value.id] = opts;
                            
                                            // lastHistory["page"][value.id]["exports"] = value.exports;
                                            return value;
                                        }
                            
                                    }
                            
                                    /*
                                    * 获取 bui.page 加载的参数
                                    *  @method getPageParam
                                    *  @param {string} [id] 获取某个id下的参数,一般不需要传
                                    *  @example
                             
                                    *  方法1: 
                                        bui.page({url:"pages/list.html",param:{id:123}})
                             
                             
                                        // pages/list.js
                                        var param = bui.history.getPageParam();
                                        console.log(param)  // {id:123}
                                    
                                    *
                                    *
                                    */
                                    module.getPageParam = function (id) {
                                        var lastHistory = this.getLast();
                                        var hasCurrentPage = id || lastHistory["currentPage"];
                                        var param = hasCurrentPage && lastHistory["page"][hasCurrentPage] && lastHistory["page"][hasCurrentPage]["param"] || null;
                                        return param;
                                    }
                            
                                    /*
                                    * 增加历史记录, 一般无需使用, 主要用于bui.page控件同步路由以后的处理
                                    *  @method add
                                    *  @param {object} option [对象]
                                    *  @param {string} option.id [页面的唯一id, router.$ 会去取这个id作为父级的选择器 ]
                                    *  @param {string} option.name [模块名, 一般为路径名 ]
                                    *  @param {string} option.url [页面跳转的模板地址 ]
                                    *  @param {boolean} option.replace [是否使用的替换历史记录 ]
                                    *  @param {object} option.param [从上一个页面传过来的参数 ]
                                    *  @param {object} option.part [当前页面里面有哪些局部加载的模块 ]
                                    *  @param {object} option.ui [当前页面里面有哪些加载的控件实例 ]
                                    *  @param {object} option.page [当前页面里面有哪些使用bui.page加载的模块 ]
                                    *  @param {object} option.toggle [当前页面使用的用来切换的对象, 必须有 hide show 方法,比方要在历史记录插入一个弹窗,那后退的时候,如果有这个对象,就会调用该对象的hide方法]
                                    *  @param {string} option.effect [跟路由一致,默认为"push"]
                                    *  @param {string} option.syncHistory [同步浏览器的路由, 默认 true ]
                                    *  @return {object} option [返回对象]
                                    *  @example
                             
                                    *  方法:
                                    *
                                    bui.history.add({
                                            id: "gid",
                                            name: "pageid",
                                            url: "url",
                                            replace: false,
                                            param: {},
                                            part: {},
                                            page: {},
                                            toggle: null,
                                            effect: "push"
                                    });
                                    *
                                    *
                                    */
                                    module.add = function (opts) {
                                        // 合并历史记录的相同对象
                                        var opt = $.extend(true, {
                                            id: "",
                                            name: "",
                                            url: "",
                                            replace: false,
                                            syncHistory: true,
                                            component: {},
                                            param: {},
                                            ui: {},
                                            part: {},
                                            page: {},
                                            toggle: null,
                                            effect: "push"
                                        }, opts);
                            
                                        var _history = historys;
                                        // 外部操控对象
                                        if (opts.toggle) {
                                            var newopt = $.extend(true, {}, opt, opts);
                                            // 浏览器的历史记录参数不能内嵌对象,这个对象是给单页路由操控的, 必须有 hide show 方法
                                            opt.toggle = null;
                                            _history.push(newopt);
                                        } else {
                                            _history.push(opt);
                                        }
                            
                                        var url = window.location.origin + window.location.pathname + "#" + opt.name;
                            
                            
                                        var newurl = ui.setUrlParams(url, opt.param);
                            
                                        opt.syncHistory && "pushState" in window.history && window.history.pushState(opt, null, newurl);
                                        return _history;
                                    }

                                    //     var _history = this.get();
                            
                                    //     var url = window.location.origin + window.location.pathname + "#" + opt.name;
                                    //     var newurl = ui.setUrlParams(url, opt.param);
                            
                                    //     _history.unshift(opt);
                            
                                    //     return _history;
                                    // }
                            
                                    /* 
                                     * 替换历史记录
                                     *  @method replace
                                     *  @param {object} [option]  [替换当前历史记录]
                                    *  @param {string} option.id [页面的唯一id, router.$ 会去取这个id作为父级的选择器 ]
                                    *  @param {string} option.name [模块名, 一般为路径名 ]
                                    *  @param {string} option.url [页面跳转的模板地址 ]
                                    *  @param {boolean} option.replace [是否使用的替换历史记录 ]
                                    *  @param {object} option.param [从上一个页面传过来的参数 ]
                                    *  @param {object} option.part [当前页面里面有哪些局部加载的模块 ]
                                    *  @param {object} option.ui [当前页面里面有哪些控件实例 ]
                                    *  @param {object} option.page [当前页面里面有哪些使用bui.page加载的模块 ]
                                    *  @param {object} option.toggle [当前页面使用的用来切换的对象, 必须有 hide show 方法,比方要在历史记录插入一个弹窗,那后退的时候,如果有这个对象,就会调用该对象的hide方法]
                                    *  @param {string} option.effect [跟路由一致,默认为"push"]
                                    *  @param {string} option.syncHistory [同步浏览器的路由, 默认 true ]
                                    *  @example
                             
                                        bui.history.replace({
                                            id: "gid",
                                            name: "pageid",
                                            url: "url",
                                            replace: false,
                                            param: {},
                                            part: {},
                                            ui: {},
                                            page: {},
                                            toggle: null,
                                            effect: "push"
                                    });
                             
                                    *
                                    */
                                    module.replace = function (opt) {
                                        // 合并历史记录的相同对象
                                        var opt = $.extend(true, {
                                            id: "",
                                            name: "",
                                            url: "",
                                            replace: false,
                                            syncHistory: true,
                                            param: {},
                                            part: {},
                                            ui: {},
                                            page: {},
                                            toggle: null,
                                            effect: "push"
                                        }, opts);
                            
                                        var _history = historys;
                            
                                        // 有替换时,最后一个历史记录的index
                                        var index = _history.length - 1;
                                        var url = window.location.origin + window.location.pathname + "#" + opt.name;
                                        var newurl = ui.setUrlParams(url, opt.param);
                            
                            
                                        // 如果存在则替换索引,否则添加记录
                                        if (index > -1) {
                                            // 外部操控对象
                                            if (opts.toggle) {
                                                var newopt = $.extend(true, {}, opt, opts);
                                                // 浏览器的历史记录参数不能内嵌对象,这个对象是给单页路由操控的, 必须有 hide show 方法
                                                opt.toggle = null;
                                                _history.splice(index, 1, newopt);
                                            } else {
                                                _history.splice(index, 1, opt);
                                            }
                                            opt.syncHistory && "replaceState" in window.history && window.history.replaceState(opt, null, newurl);
                                        }
                            
                                        return _history;
                                    }

                                     * 刷新页面
                                     *  @method refresh
                                     *  @example
                             
                                        bui.history.refresh();

                                    module.refresh = function () {
                            
                                        if (ui.hasRouter) {

                                            loader.import(_history.url, function (res) {
                            
                                                res = ui.unit.handleContent(res);
                                                res = res.replace(/#module\.id/g, "#" + _history.id);
                                                res = res.replace(/#module\.path\//g, _module.path);
                            
                                                $("#" + _history.id).html(res);
                                                //计算main高度
                                                ui.config.init.auto && ui.init();
                                                // 编译view 
                                                router.config.needView && loader.views();
                                                // 重新执行模块方法
                                                _module.loaded();
                                                // 编译component
                                                router.config.needComponent && loader.components();
                                            })
                                        } else {
                                            window.location.reload(true);
                                        }
                            
                                    }

                                    * 删除指定的index包含之后的历史记录
                                    *  @method removeNext
                                    *  @param {object} option 
                                    *  @param {number} option.index [删除第几个索引之后的历史] 
                                    *  @param {boolean} option.syncHistory [是否把浏览器历史记录也删掉] 
                                    *  @example
                             
                                        // 删除第3条
                                        bui.history.removeNext({index:2});
                             
                                    *
                                    */
                                    module.removeNext = function (opt) {
                                        var _history = historys;
                                        var index = opt.index;
                                        if (parseInt(index) < 0) {
                                            ui.showLog("index 必须大于或者等于0");
                                            return _history;
                                        }
                                        var len = _history.length - index;
                                        _history.splice(index, len);
                                        var lastHistory = this.getLast();
                                        var url = window.location.origin + window.location.pathname + "#" + lastHistory.name;
                                        var newurl = ui.setUrlParams(url, lastHistory.param);
                            
                                        opt.syncHistory && "replaceState" in window.history && window.history.replaceState(lastHistory["param"], null, newurl);
                                        return _history;
                                    }

                                    /** 
                                     * 获取当前hash的值
                                     *  @method getHash
                                     *  @since 1.7.4
                                     *  @return {object} {pid:"",url:"",param:{}}
                                     *  @example
                             
                                        //  var hash = bui.history.getHash();
                             
                                    *
                                    */
                                    module.getHash = function (boolDecode) {
                                        var url = window.location.hash || window.location.search,
                                            // 问号在hash 前面, 可以通过search拿
                                            searchs = window.location.search,
                                            searchsIndex = searchs && searchs.indexOf("?"),
                                            bool = boolDecode == false ? false : true,
                                            hashIndex = url && url.indexOf("?"),
                                            pathName = hashIndex > -1 ? url && url.substr(1, (hashIndex - 1)) : window.location.hash.substr(1),
                                            pointIndex = pathName && pathName.indexOf(".html"),
                                            // pid = pointIndex > -1 ? pathName.substr(0, pointIndex) : pathName,
                                            pid = pathName.replace(/\.js|\.html/gi, ''),
                                            // purl = pointIndex > -1 ? pathName : pathName + ".html",
                                            mod = loader.get(pid) || {},
                                            purl = mod.template ? mod.template : pid + '.html',
                                            theRequest = {};
                            
                            
                                        // 如果有传参
                                        if (hashIndex > -1) {
                                            // 获取search后面的参数
                                            getHashParam(searchs, searchsIndex);
                                            // 获取hash后面的参数
                                            getHashParam(url, hashIndex);
                            
                                        }
                            
                                        function getHashParam(urlstr, index) {
                                            var str = urlstr && urlstr.substr((index + 1));
                                            var strIndex = str.indexOf("?");
                                            // 有2个问号的情况
                                            if (strIndex > -1) {
                                                var newstrs = str.split("?");
                                                strSplit(newstrs[0]);
                                                strSplit(newstrs[1]);
                                            } else {
                                                strSplit(str);
                                            }
                            
                                            function strSplit(str) {
                                                var strs = str && str.split("&");
                                                for (let i = 0; i < strs.length; i++) {
                                                    let strItem = strs[i];
                                                    let val = bool ? decodeURIComponent(strItem.split("=")[1]) : strItem.split("=")[1];
                                                    if (val === 'false') {
                                                        val = false;
                                                    } else if (val === 'true') {
                                                        val = true;
                                                    }
                                                    // 把false,true值统一转成布尔值
                                                    theRequest[strItem.split("=")[0]] = val;
                                                }
                                            }
                                        }
                            
                                        return {
                                            pid: pid,
                                            url: purl,
                                            param: theRequest
                                        }
                                    }

                                     * 解析路径

                                     *  @example
                             
                                    *  方法:
                                    *
                                        var allHistory = bui.history.resolveUrl(url);
                                    *
                                    *
                                    */
                                    var resolveUrl = function (path) {
                            
                                        path = path && path.replace(window.location.origin, "") || window.location.pathname;
                                        var hasHtml = path.indexOf(".html");
                                        var hasParam = path.indexOf("?");
                                        var pid = "",
                                            url = "",
                                            src = "",
                                            param = null;
                                        var modules = loader.map()["modules"];
                                        // 解析页面传的参数
                                        if (hasParam > -1) {
                                            var paramstr = path.substr(hasParam + 1);
                                            param = ui.unit.keyStringToObject(paramstr, true);
                                        }
                                        if (hasHtml > -1) {
                                            pid = path.indexOf("/") == 0 ? path.substr(1, (hasHtml - 1)) : path.substr(0, (hasHtml - 1));
                                            url = pid + ".html";
                                            src = pid + ".js";
                                        } else {
                                            pid = path;
                                            url = modules[pid] && modules[pid]["template"] || "";
                                            src = modules[pid] && modules[pid]["script"] || "";
                                        }
                                        return {
                                            pid: pid,
                                            name: pid,
                                            url: url,
                                            src: src,
                                            param: param
                                        }
                                    }

                                        })
                                    }
