                            /**
                             * UI控件库 
                             * @module UI
                             */
                            
                            ;
                            (function (ui, $) {
                            
                                /**
                                 * 计时器,可以递减或者递增, 可以用来做秒杀, 1.5.6新增,支持不同类型设定times
                                 * 预览地址: <a href="http://www.easybui.com/demo/index.html#pages/ui_controls/bui.timer.html" target="_blank">demo</a>
                                 * @namespace  bui
                                 * @class  timer
                                 * @constructor 
                                 * @param  {object} option  
                                 * @param  {string} option.target  时间显示在某个id 上
                                 * @param  {string} [option.uid]  1.6.6新增，自定义控件的唯一id,非必须
                                 * @param  {function} option.onEnd  时间到以后执行回调
                                 * @param  {function} [option.onProcess]  过程处理函数,参数为当前秒数
                                 * @param  {string} [option.type]  1.5.6新增, 定义times的类型, 默认 number(跟second类似,输出是大于60的数字) | second | minute | hour  | date | day 
                                 * @param  {number} [option.time]  1.5.6新增, 默认:10秒 当 type=minute 10等于10分钟 依次类推, 当type="date" 时, time 传某个日期 "2020-3-20 10:00" 
                                 * @param  {number} [option.times]   废弃 
                                 * @param  {number} [option.fixZero]  1.5.6新增, 小余10是否在前面补零, 默认 true | false
                                 * @param  {boolean} [option.reduce]  默认:true 递减 | false 递增
                                 * @example
                                    
                                    // 例子1: 倒计时10秒
                                    html: 
                                    <div id="time"></div>
                                    
                                    js: 
                                    var timer = bui.timer({
                                        target: "#time",
                                        onEnd: function (argument) {
                                            bui.alert("时间到")
                                        },
                                        type: "number",
                                        time: 10
                                    })
                            
                                    timer.start();
                            
                                    // 例子2: 倒计时10分钟
                                    js: 
                                    var timer = bui.timer({
                                        target: "#time",
                                        onEnd: function (argument) {
                                            bui.alert("时间到")
                                        },
                                        type: "minute",
                                        time: 10
                                    })
                            
                                    timer.start();
                            
                                    // 例子3:倒计时1天
                                    js: 
                                    var timer = bui.timer({
                                        target: "#time",
                                        onEnd: function (argument) {
                                            bui.alert("时间到")
                                        },
                                        type: "minute",
                                        time: 10
                                    })
                            
                                    timer.start();
                            
                                    // 例子4: 离今晚12点还有多少时间
                                    var afterDate = bui.date.after(new Date(), 1);
                                    time = bui.timer({
                                        target: "#time",
                                        onEnd: function (argument) {
                                            bui.alert("时间到")
                                        },
                                        type: "date",
                                        time: afterDate
                                    });
                                    time.start();
                                 * 
                                 */
                                ui.timer = function (opt) {
                            
                                    // 增加对跨天的支持,时分秒的格式化
                                    var config = {
                                        uid: "",
                                        interval: 1000,
                                        target: null,
                                        reduce: true,
                                        onEnd: null,
                                        onProcess: null,
                                        fixZero: true,
                                        type: "number", // second minute hour day date 
                                        time: 10,
                                        times: 0 // 废弃
                                    }
                            
                                    var option = $.extend({}, config, opt);
                                    var reduce = option.reduce; // 是否递减
                                    var $this = null;
                            
                                    var leftTime = 0,
                                        isPause = false;
                                    var timesValue = option.time;
                            
                                    // 正在倒计时中
                                    var countdowning = false;
                                    var countnum = null,
                                        timeout = null;
                            
                                    var module = {
                                        name: "timer",
                                        stop: function (callback) {
                                            leftTime = 0;
                                            countnum = countTime(option);
                                            clearTimeout(timeout);
                                            timeout = null;
                            
                                            callback && callback.call(this);
                                        },
                                        start: reduce ? countdown : countup,
                                        restart: restart,
                                        status: function () {
                                            return countdowning;
                                        },
                                        pause: pause
                                    }
                            
                                    // 初始化
                                    init(option);
                                    
                                    function init(opt){
                                    
                                        option = $.extend({}, config, opt);
                                        reduce = option.reduce; // 是否递减
                                        $this = option.target ? ui.obj(option.target) : null;
                            
                                        leftTime = 0;
                                        isPause = false;
                                        timesValue = option.times || option.time;
                            
                                        // 正在倒计时中
                                        countdowning = false;
                                        countnum = countTime(option);
                                        timeout = null;
                            
                                        // 存储实例
                                        option.uid && ui.history.setUI && ui.history.setUI({
                                            uid: option.uid,
                                            ui: module
                                        })
                            
                                        return this;
                                    }
                            
                                    // 获得类型不同得到的times不同
                                    function countTime(opt) {
                                        var time = null;
                                        // 逐减
                                        switch (opt.type) {
                                            case "minute":
                                                time = parseInt(timesValue, 10) * 60;
                                                break;
                                            case "hour":
                                                time = parseInt(timesValue, 10) * 60 * 60;
                                                break;
                                            case "day":
                                                time = parseInt(timesValue, 10) * 60 * 60 * 24;
                                                break;
                                            case "date":
                                                // 距离某个日期还有多少时间
                                                time = (ui.date.convert(timesValue).getTime() - new Date().getTime()) / 1000;
                                                break;
                                            case "second":
                                            default:
                                                time = parseInt(timesValue, 10);
                                                break;
                                        };
                            
                                        // 递增
                                        if (!reduce) {
                                            timesValue = time;
                                            time = 0;
                                        }
                                        return time;
                                    }
                                    /**
                                     * 暂停
                                     * @namespace  bui
                                     * @method  pause
                                     * @constructor 
                                     * @example
                            
                                        timer.pause();
                                     
                                     *      
                                     * 
                                     */
                                    function pause() {
                                        leftTime = countnum;
                                        isPause = true;
                                        clearTimeout(timeout);
                                        return this;
                                    }
                            
                                    /**
                                     * 重新计时
                                     * @namespace  bui
                                     * @method  restart
                                     * @constructor 
                                     * @example
                            
                                        timer.restart();
                                     
                                     *      
                                     * 
                                     */
                                    function restart() {
                                        leftTime = 0;
                                        countnum = countTime(option);
                                        timeout = clearTimeout(timeout);
                                        start();
                            
                                        return this;
                                    }
                            
                                    function getTime(num) {
                                        var dateObj = new Date();
                                        var date = ui.date.format(dateObj, "yyyy-MM-dd");
                                        var hour = 0,
                                            minute = 0,
                                            day = 0,
                                            value = '',
                                            second = parseInt(num % 60, 10);
                            
                                        switch (option.type) {
                                            case "minute":
                                                minute = parseInt(num / 60, 10);
                                                value = `${fixZero(minute)}:${fixZero(second)}`;
                            
                                                break;
                                            case "hour":
                                                hour = parseInt(num / 60 / 60, 10);
                                                minute = parseInt(num / 60 % 60, 10);
                                                value = `${fixZero(hour)}:${fixZero(minute)}:${fixZero(second)}`;
                                                break;
                                            case "date":
                                                // 距离某个日期还有多少时间
                                                date = ui.date.format((option.times || option.time), "yyyy-MM-dd");
                                                day = parseInt(num / 60 / 60 / 24, 10);
                                                hour = parseInt(num / 60 / 60 % 24, 10);
                                                minute = parseInt(num / 60 % 60, 10);
                                                value = `剩余${day}天 ${fixZero(hour)}:${fixZero(minute)}:${fixZero(second)}`;
                                                break;
                                            case "day":
                                                // 距离某个日期还有多少时间
                                                date = ui.date.format(dateObj.getTime() + num * 1000, "yyyy-MM-dd");
                                                day = parseInt(num / 60 / 60 / 24, 10);
                                                hour = parseInt(num / 60 / 60 % 24, 10);
                                                minute = parseInt(num / 60 % 60, 10);
                                                value = `剩余${day}天 ${fixZero(hour)}:${fixZero(minute)}:${fixZero(second)}`;
                                                break;
                                            case "second":
                                                minute = parseInt(num / 60, 10);
                                                value = `${fixZero(minute)}:${fixZero(second)}`;
                                                break;
                                            case "number":
                                                minute = parseInt(num / 60, 10);
                                                value = `${fixZero(parseInt(num, 10))}`;
                                                break;
                                        }
                            
                                        return {
                                            init: init,
                                            date: date,
                                            day: day,
                                            hour: hour,
                                            minute: minute,
                                            second: second,
                                            count: num,
                                            value: value,
                                            target: $this && $this[0]
                                        }
                            
                                    }
                            
                                    function fixZero(num) {
                                        return num < 10 && option.fixZero ? "0" + num : num;
                                    }
                            
                            
                                    /**
                                     * 开始计时,递减或者递增取决于初始化的reduce是否为true
                                     * @namespace  bui
                                     * @method  start
                                     * @constructor 
                                     * @example
                            
                                        timer.start();
                                     
                                     *      
                                     * 
                                     */
                                    // 递减
                                    function countdown() {
                                        if (isPause) {
                                            countnum = leftTime;
                                            isPause = false;
                                        }
                            
                                        var time = getTime(countnum);
                                        if (countnum <= 0) {
                                            countnum = 0;
                                            $this && $this.text(time.value);
                                            option.onEnd && option.onEnd.call(this, time);
                                            clearTimeout(timeout);
                                            countdowning = false;
                            
                                            return;
                                        } else {
                                            $this && $this.text(time.value);
                                            option.onProcess && option.onProcess.call(this, time);
                                            countnum--;
                            
                                            countdowning = true;
                                        }
                                        // 自执行
                                        timeout = setTimeout(function () {
                                            countdown()
                                        }, option.interval)
                            
                                        return this;
                                    }
                                    // 递增
                                    function countup() {
                                        if (isPause) {
                                            countnum = leftTime;
                                            isPause = false;
                                        }
                            
                                        var time = getTime(countnum);
                            
                                        if (countnum >= timesValue) {
                                            $this && $this.text(time.value);
                                            countnum = timesValue;
                                            option.onEnd && option.onEnd.call(this, time);
                                            clearTimeout(timeout);
                                            countdowning = false;
                                            return;
                                        } else {
                                            $this && $this.text(time.value);
                                            option.onProcess && option.onProcess.call(this, time);
                                            countnum++;
                                            countdowning = true;
                                        }
                                        // 自执行
                                        timeout = setTimeout(function () {
                                            countup()
                                        }, option.interval)
                            
                                        return this;
                                    }
                            
                                    var start = reduce ? countdown : countup;
                            
                                    /**
                                     * 停止计时
                                     * @namespace  bui
                                     * @method  stop
                                     * @constructor 
                                     * @example
                            
                                        timer.stop();
                                     
                                     *      
                                     * 
                                     */
                                    /**
                                     * 是否在计时中
                                     * @namespace  bui
                                     * @method  status
                                     * @constructor 
                                     * @example
                                        
                                        // true 正在进行倒计时
                                        var status = timer.status();
                                     
                                     *      
                                     * 
                                     */
                                    return module
                                };
                            
                            
                                return ui;
                            
                            })(window.bui || {}, window.libs);
