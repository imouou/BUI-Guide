                            /**
                             * @module Native
                             */
                            ;
                            (function (ui, $) {
                                /**
                                 * <h3><del>下载控件,</del></h3>
                                 * <h3>预览地址: <a href="http://www.easybui.com/demo/index.html#pages/ui_controls/bui.download.html" target="_blank">demo</a></h3>
                                 * <p>下载文件在本地,web不支持</p>
                                 * {{#crossLink "bui.download/getFile"}}{{/crossLink}}: 获取文件,获取不到则下载文件 <br>
                                 * {{#crossLink "bui.download/start"}}{{/crossLink}}: 下载文件 <br>
                                 * {{#crossLink "bui.download/stop"}}{{/crossLink}}: 停止下载 <br>
                                 * {{#crossLink "bui.download/toBase64"}}{{/crossLink}}: 下载的图片转换成可以插入页面的图片 <br>
                                 * {{#crossLink "bui.download/widget"}}{{/crossLink}}: 获取依赖的控件 <br>
                                 *  @namespace bui
                                 *  @class download
                                 *  @constructor
                                 *  @param {object} option
                                 *  @param {string} [option.url]  下载的地址, 中文地址, needEncode 参数应该设置为true 
                                 *  @param {boolean} [option.needEncode]  1.5.1新增 地址是否需要转码 默认: false | true , 如果是中文地址,需要转码 
                                 *  @param {object} [option.data]  请求下载需要传的参数 
                                 *  @param {number} [option.timeout]  触发timeout的时间默认60000 
                                 *  @param {string} [option.fileName]  文件名称，默认是url的最后地址
                                 *  @param {string} [option.folderName]  文件夹名称,默认是download 
                                 *  @param {boolean} [option.showProgress]  是否显示进度条 默认true  
                                 *  @param {boolean} [option.needNative]  1.5.1新增, 是否使用原生下载, 默认 true|false 通过这个切换对应的效果,可以通过全局配置 bui.config.download = {needNative:true} 
                                 *  @param {function} [option.onProgress]  自定义进度条回调,接收百分比值 
                                 *  @param {function} [option.onSuccess]  成功的回调 
                                 *  @param {function} [option.onFail]  失败的回调 
                                 *  @example
                                 *
                                 *   js:
                                 *
                                        // 1. 初始化 这里如果传url初始化,则马上下载
                                        var uidownload = bui.download();
                            
                                        // 2. 开始下载
                                        uiUpload.start({
                                            data: null
                                        })
                                 *
                                 */
                                ui.download = function (option) {
                                    var config = {
                                        uid: "",
                                        url: "",
                                        data: {},
                                        headers: {},
                                        method: "GET",
                                        showProgress: true,
                                        timeout: 60000,
                                        fileName: "",
                                        folderName: "download",
                                        native: true,
                                        needNative: true,
                                        onProgress: null,
                                        onSuccess: null,
                                        onFail: null
                                    }
                            
                                    //方法
                                    var module = {
                                        open: open,
                                        getFile: getFile,
                                        removeFile: removeFile,
                                        removeFolder: removeFolder,
                                        start: start,
                                        stop: stop,
                                        toBase64: toBase64,
                                        init: init,
                                        widget: widget
                                    };
                            
                                    //用于option方法的设置参数
                                    var options = module.config = $.extend(true, {}, config, ui.config.download, option);
                            
                                    var uiAjax,
                                        uiFile,
                                        uiNativeDownload,
                                        // 如果没有传 needNative 以 isWebapp 的状态为主
                                        webapp = typeof option === "object" && typeof option.needNative !== "undefined" ? !option.needNative : ui.isWebapp, //(options.native && ui.isWebapp) || (!options.native && !ui.isWebapp),
                                        targetUrl,
                                        targetEncodeUrl,
                                        folderPath,
                                        filePath,
                                        showProgress = options.showProgress,
                                        onProgress = options.onProgress,
                                        onFail = options.onFail,
                                        progressPercent;
                            
                                    var uiLoading = null;
                                    var loadingAnimate = null;
                                    // 初始化
                                    init(options);
                            
                                    //初始化
                                    function init(option) {
                            
                                        //引用loading控件
                                        uiLoading = option.showProgress && ui.loading({
                                            // display: "block",
                                            display: "block",
                                            width: 30,
                                            height: 30,
                                            opacity: 0,
                                            callback: function () {
                                                stop();
                                            }
                                        })
                            
                                        //引用file控件
                                        uiFile = ui.file({
                                            native: option.native,
                                            needNative: option.needNative
                                        });
                            
                                        // 为true时,是web平台
                                        if (option.needNative) {
                            
                                            // 原生下载
                                            uiNativeDownload = ui.native.download && ui.native.download(option, {
                                                file: uiFile,
                                                loading: uiLoading,
                                                module: module
                                            }) || {};
                            
                                        } else {
                                            webDownload(option)
                                        }
                            
                                        function webDownload(option) {
                                            // 获取文件夹
                                            uiFile.getFolder({
                                                folderName: option.folderName,
                                                create: false,
                                                onSuccess: function (folder, file) {
                                                    //文件夹路径
                                                    folderPath = folder.nativeURL;
                                                },
                                                onFail: function (err) {
                                                    // 创建文件夹
                                                    uiFile.getFolder({
                                                        folderName: option.folderName,
                                                        create: true,
                                                        onSuccess: function (folder, file) {
                                                            //文件夹路径
                                                            folderPath = folder.nativeURL;
                                                        },
                                                        onFail: function (err) {
                                                            onFail && onFail.call(module, err);
                                                        }
                                                    });
                                                }
                                            });
                            
                                            if (!option.url) {
                                                // onFail && onFail(err);
                                                return;
                                            }
                            
                                            //开始下载
                                            start(option);
                                        }
                            
                                        // 存储实例
                                        option.uid && ui.history.setUI && ui.history.setUI({
                                            uid: option.uid,
                                            ui: module
                                        })
                                        return this;
                                    }
                            
                                    /**
                                     * 开始下载
                                     *  @method start
                                     *  @param {object} [option] 参数同初始化一样,查看最顶部的参数
                                     *  @chainable
                                     *  @example
                                     *
                                        uiDownload.start({
                                            url: "http://"
                                        })
                            
                                     */
                                    function start(param) {
                            
                                        var params = $.extend(true, {}, options, param);
                            
                                        targetUrl = params.needEncode ? encodeURI(params.url) : params.url;
                            
                                        onProgress = params.onProgress;
                                        showProgress = params.showProgress;
                            
                                        showProgress && uiLoading && uiLoading.show({
                                            text: "0%"
                                        });
                            
                                        //文件存储路径
                                        filePath = folderPath + "/" + (param.fileName || uiFile.getFileName(params.url));
                            
                            
                                        if (params.needNative) {
                            
                                            uiNativeDownload.start(params);
                                        } else {
                                            // stopLoading();
                            
                                            // ui.hint("暂不支持web下载")
                                            // onFail && onFail.call(module);
                            
                                            // return;
                                            // web下载
                                            // 重置参数
                                            params.cache = false;
                                            params.contentType = false;
                                            params.processData = false;
                            
                                            uiAjax = $.ajax({
                                                url: targetUrl,
                                                type: params.method,
                                                data: params.data,
                                                cache: params.cache,
                                                headers: params.headers,
                                                contentType: params.contentType, //不可缺
                                                processData: params.processData, //不可缺
                                                timeout: params.timeout,
                                                xhr: function () {
                                                    var xhr = $.ajaxSettings.xhr();
                            
                                                    // 处理下载的进度
                            
                                                    if (onprogress && xhr) {
                                                        xhr.addEventListener("progress", onprogress, false);
                                                        return xhr;
                                                    }
                                                },
                                                success: function (result) {
                            
                                                    var name = param.fileName || uiFile.getFileName(params.url);
                            
                                                    // 获取本地文件,如果找不到则下载
                                                    uiFile.getFile({
                                                        fileName: name,
                                                        folderName: params.folderName,
                                                        create: true,
                                                        onSuccess: function (file, entry) {
                                                            // console.log(file.toURL())
                                                            params.onSuccess && params.onSuccess.call(module, file.nativeURL, entry);
                                                        }
                                                    })
                                                    // onSuccess && onSuccess.call(module,result);
                                                },
                                                fail: function (result) {
                                                    stopLoading();
                            
                                                    params.onFail && params.onFail.call(module, result);
                                                }
                                            });
                            
                                        }
                                    }
                            
                                    /**
                                     * 获取下载的文件,如果没有则重新下载
                                     *  @method getFile
                                     *  @param {object} option [参数同初始化一样]
                                     *  @param {string} option.url [下载地址]
                                     *  @param {string} option.fileName [保存的文件名,含后缀]
                                     *  @param {string} [option.folderName] 保存的文件夹, 默认download文件夹,可以不用传
                                     *  @param {function} [option.onSuccess] 成功的回调
                                     *  @param {function} [option.onFail] 失败的回调
                                     *  @chainable
                                     *  @example
                                     *
                                        uiDownload.getFile({
                                            url: "http://",
                                            fileName:"",
                                            onSuccess: function (url,file) {
                                                console.log(url);
                                            }
                                        })
                            
                                     */
                                    function getFile(param) {
                            
                                        var params = $.extend(true, {
                                            autoDown: true
                                        }, options, param);
                            
                                        var name = param.fileName || uiFile.getFileName(params.url);
                            
                                        // 获取本地文件,如果找不到则下载
                                        uiFile.getFile({
                                            fileName: name,
                                            folderName: params.folderName,
                                            onSuccess: function (file, entry) {
                            
                                                params.onSuccess && params.onSuccess.call(module, file.nativeURL || file, file, entry);
                                            },
                                            onFail: function (err) {
                            
                                                //开始下载
                                                params.autoDown = start(params);
                            
                                            }
                                        })
                                    }
                            
                                    /**
                                     * 删除文件
                                     *  @method removeFile
                                     *  @since 1.5.2
                                     *  @param {object} option [参数同初始化一样]
                                     *  @param {string} [option.fileName] 删除的文件名称
                                     *  @param {string} [option.folderName] 删除的文件名称在哪个文件夹, 默认是: "download", 一般无需更改
                                     *  @param {function} [option.onSuccess] 成功的回调
                                     *  @param {function} [option.onFail] 失败的回调
                                     *  @chainable
                                     *  @example
                                     *
                                        uiDownload.removeFile({
                                            fileName: "customName.jpg",
                                            onSuccess: function (url,file) {
                                                console.log(url);
                                            }
                                        })
                            
                                     */
                                    function removeFile(param) {
                            
                                        var name = param.fileName;
                            
                                        if (!name) {
                                            ui.showLog("请告诉我们你要删除哪个文件, fileName ");
                                            return this;
                                        }
                                        param.folderName = param.folderName || options.folderName;
                                        // 删除文件
                                        uiFile.removeFile(param)
                                        return this;
                                    }
                                    /**
                                     * 删除文件夹,默认是 "download"
                                     *  @method removeFolder
                                     *  @since 1.5.2
                                     *  @param {object} option [参数同初始化一样]
                                     *  @param {string} [option.folderName] 删除的文件名称在哪个文件夹, 默认是: "download"
                                     *  @param {function} [option.onSuccess] 成功的回调
                                     *  @param {function} [option.onFail] 失败的回调
                                     *  @chainable
                                     *  @example
                                     *
                                        uiDownload.removeFolder({
                                            folderName: "customName.jpg",
                                            onSuccess: function (url,file) {
                                                console.log(url);
                                            }
                                        })
                            
                                     */
                                    function removeFolder(param) {
                            
                                        param.folderName = param.folderName || options.folderName;
                                        // 获取本地文件,如果找不到则下载
                                        uiFile.removeFolder(param)
                            
                                        return this;
                                    }
                            
                                    /**
                                     * 本地程序打开文件, web不支持
                                     *  @method open
                                     *  @since 1.5.2
                                     *  @param {object} [option] 
                                     *  @param {string} [option.url]  文件路径,一般是通过getFile得到的路径  
                                     *  @param {function} [option.onSuccess] 成功的回调
                                     *  @param {function} [option.onFail] 失败的回调
                                     *  @chainable
                                     *  @example
                                     *
                                        uiDownload.open({
                                            url: "file://",
                                            onSuccess: function(url){
                                                // 文件的地址
                                                console.log(url)
                                            }
                                        })
                            
                                     */
                                    function open(param) {
                            
                                        // 获取本地文件,如果找不到则下载
                                        uiFile.open(param)
                            
                                        return this;
                                    }
                            
                                    /**
                                     * 停止下载
                                     *  @method stop
                                     *  @chainable
                                     *  @example
                                     *
                                        uiDownload.stop()
                            
                                     */
                                    function stop(callback) {
                            
                                        if (!param.needNative) {
                                            stopLoading();
                                            uiAjax && uiAjax.abort();
                            
                                            // 取消动画加载
                                            loadingAnimate && window.cancelAnimationFrame(loadingAnimate);
                                        } else {
                                            uiNativeDownload.stop();
                                        }
                                        callback && callback.call(module, uiLoading, uiAjax);
                            
                                    }
                            
                                    // web 下载进度
                                    function onprogress(evt) {
                            
                                        //安卓4.3以下不支持
                                        window.requestAnimationFrame = window.requestAnimationFrame || window.webkitRequestAnimationFrame;
                            
                                        // 防止掉帧
                                        loadingAnimate = window.requestAnimationFrame(function () {
                                            if (evt.lengthComputable) {
                                                var percentComplete = Math.round(evt.loaded * 100 / evt.total);
                                                progressPercent = percentComplete.toString() + '%';
                            
                                                if (percentComplete < 100) {
                                                    startLoading(progressPercent)
                                                } else {
                                                    stopLoading();
                                                }
                            
                                                onProgress && onProgress.call(module, progressPercent);
                                            }
                                        })
                                    }
                            
                            
                                    // 开启进度
                                    function startLoading(percent) {
                                        showProgress && uiLoading && uiLoading.show({
                                            text: percent
                                        });
                                    }
                            
                                    // 关闭进度
                                    function stopLoading() {
                                        showProgress && uiLoading && uiLoading.stop();
                            
                                    }
                            
                                    /**
                                     * 把选择的图片文件转换成base64地址,可以直接展示在页面上
                                     *  @method toBase64
                                     *  @param {object} option
                                     *  @param {string} [option.data]  通过getFile得到的文件  
                                     *  @param {function} [option.onSuccess] 成功的回调
                                     *  @param {function} [option.onFail] 失败的回调
                                     *  @chainable
                                     *  @example
                                     *
                                        uiDownload.toBase64({
                                            data: "file:///", //本地图片路径
                                            onSuccess: function (imgurl) {
                                                $("#id").append('<img src="'+imgurl+'" />')
                                            }
                                        });
                            
                                     */
                                    function toBase64(param) {
                            
                                        uiFile.toBase64(param);
                                        return this;
                                    }
                            
                                    /**
                                     * 获取依赖的控件
                                     *  @method widget
                                     *  @param {string} [name]  依赖控件名 loading 
                                     *  @example
                            
                                        //获取依赖控件
                                        var uiLoading = uiFile.widget("loading");
                            
                                        //使用uiLoading的方法
                                        uiLoading.start();
                            
                                     */
                                    function widget(name) {
                                        var control = {
                                            loading: uiLoading,
                                            file: uiFile,
                                            ajax: uiAjax
                                        };
                                        return ui.widget.call(control, name);
                                    };
                            
                                    return module;
                                };
                            
                                return ui;
                            
                            })(window.bui || {}, window.libs);
