                            /**
                             * 方法库
                             * @module Method
                             */
                            
                            ;
                            (function (ui, $) {
                            
                                /**
                                 * 仅用于安卓及ios移动平台的检测
                                 * @namespace  bui
                                 * @class  platform
                                 *
                                 */
                                ui.platform = (function () {
                            
                                    var u = navigator.userAgent;
                            
                                    /**
                                     * 是否是window平台
                                     *  @method isWindow
                                     *  @chainable
                                     *  @return {boolean} [true | false]
                                     *  @example
                            
                                           var bool = bui.platform.isWindow();
                            
                                     */
                                    function isWindow(argument) {
                                        return /Windows NT/i.test(u) || navigator.platform.indexOf("Win32") > -1 || navigator.platform.indexOf("Win64") > -1;
                                    }
                                    /**
                                     * 是否是mac平台
                                     *  @method isMac
                                     *  @chainable
                                     *  @return {boolean} [true | false]
                                     *  @example
                            
                                           var bool = bui.platform.isMac();
                            
                                     */
                                    function isMac() {
                                        // iphone12 会检测到跟mac相同的值, 这里的正则做了修改;
                                        return /macintosh;.+mac os x/i.test(u) || navigator.platform.indexOf("Mac") > -1;
                                    }
                                    /**
                                     * 是否是PC;
                                     *  @method isPC
                                     *  @chainable
                                     *  @return {boolean} [true | false]
                                     *  @example
                            
                                           var bool = bui.platform.isPC();
                            
                                     */
                                    function isPC() {
                                        return !isMobile();
                                    }
                                    /**
                                     * 是否是移动端, 包含, 安卓,iphone, ipad 等都算移动端
                                     *  @method isMobile
                                     *  @since 1.6.4
                                     *  @chainable
                                     *  @return {boolean} [true | false]
                                     *  @example
                            
                                           var bool = bui.platform.isMobile();
                            
                                     */
                                    function isMobile() {
                            
                                        return isAndroid() || isIphone() || isIpad();
                                    }
                                    /**
                                     * 是否是安卓手机
                                     *  @method isAndroid
                                     *  @chainable
                                     *  @return {boolean} [true | false]
                                     *  @example
                            
                                           var bool = bui.platform.isAndroid();
                            
                                     */
                                    function isAndroid(argument) {
                                        return /(Android|Linux)/i.test(u);
                                    }
                                    /**
                                     * 是否是Iphone
                                     *  @method isIphone
                                     *  @chainable
                                     *  @return {boolean} [true | false]
                                     *  @example
                            
                                           var bool = bui.platform.isIphone();
                            
                                     */
                                    function isIphone(argument) {
                                        return /(iPhone)/i.test(u);
                                    }
                                    /**
                                     * 是否是全面屏 Iphonex xs xr xsMax 更新到iphone14promax
                                     *  @method isIphoneX
                                     *  @since 1.6.2
                                     *  @chainable
                                     *  @return {boolean} [true | false]
                                     *  @example
                            
                                           var bool = bui.platform.isIphoneX();
                            
                                     */
                                    function isIphoneX(argument) {
                                        var isX = false;
                                        // x xs 的分辨率
                            
                                        // iPhone 13 mini | iPhone 12 mini | iPhone 11 Pro | iPhone X | iPhone XS	
                                        var xDevice = window.screen.width == 375 && window.screen.height == 812;
                                        // iPhone 11 | iPhone XR | iPhone 11 Pro Max | iPhone XS Max
                                        var xrDevice = window.screen.width == 414 && window.screen.height == 896;
                                        // iPhone 14 iPhone 13 Pro iPhone 13 iPhone 12 Pro iPhone 12
                                        var x12Device = window.screen.width == 390 && window.screen.height == 844;
                                        // iphone 14 pro  实际可能也是 390 844
                                        var x14Device = window.screen.width == 393 && window.screen.height == 852;
                                        //iPhone 14 Plus iPhone 13 Pro Max iPhone 12 Pro Max
                                        var x14pDevice = window.screen.width == 428 && window.screen.height == 926;
                                        // iphone 14 pro max
                                        var x14proDevice = window.screen.width == 430 && window.screen.height == 932;
                            
                                        if (xDevice || xrDevice || x12Device || x14Device || x14pDevice || x14proDevice) {
                                            isX = true;
                                        }
                                        return isX;
                                    }
                                    /**
                                     * 是否是Iphone 全面屏 ，通过条件宽松些，可能会有误判
                                     *  @method isIphoneFull
                                     *  @chainable
                                     *  @return {boolean} [true | false]
                                     *  @example
                            
                                           var bool = bui.platform.isIphoneFull();
                            
                                     */
                                    function isIphoneFull(argument) {
                                        var isX = false;
                                        // iPhone 13 mini | iPhone 12 mini | iPhone 11 Pro | iPhone X | iPhone XS	
                                        var xwidth = window.screen.width >= 375 && window.screen.width <= 430;
                                        // iPhone 11 | iPhone XR | iPhone 11 Pro Max | iPhone XS Max
                                        var xheight = window.screen.height >= 812 && window.screen.height <= 926;
                            
                                        if (xwidth && xheight) {
                                            isX = true;
                                        }
                                        return isX;
                                    }
                                    /**
                                     * 是否是Ipad
                                     *  @method isIpad
                                     *  @chainable
                                     *  @return {boolean} [true | false]
                                     *  @example
                            
                                           var bool = bui.platform.isIpad();
                            
                                     */
                                    function isIpad(argument) {
                                        return /(iPad)/i.test(u);
                                    }
                                    /**
                                     * IOS平台包括Iphone Ipad
                                     *  @method isIos
                                     *  @chainable
                                     *  @return {boolean} [true | false]
                                     *  @example
                            
                                           var bool = bui.platform.isIos();
                            
                                     */
                                    function isIos(argument) {
                                        return /(iPhone|iPad|iOS)/i.test(u);
                                    }
                                    /**
                                     * 微信浏览器
                                     *  @method isWeiXin
                                     *  @chainable
                                     *  @return {boolean} [true | false]
                                     *  @example
                            
                                           var bool = bui.platform.isWeiXin();
                            
                                     */
                                    function isWeiXin(argument) {
                                        return /(micromessenger)/i.test(u);
                                    }
                                    /**
                                     * 微信小程序环境
                                     *  @method isWeiXinMini
                                     *  @chainable
                                     *  @return {boolean} [true | false]
                                     *  @example
                            
                                           var bool = bui.platform.isWeiXinMini();
                            
                                     */
                                    function isWeiXinMini(argument) {
                                        return window.__wxjs_environment === 'miniprogram';
                                    }
                                    /**
                                     * BT应用
                                     *  @method isBingotouch
                                     *  @chainable
                                     *  @return {boolean} [true | false]
                                     *  @example
                            
                                           var bool = bui.platform.isBingotouch();
                            
                                     */
                                    function isBingotouch(argument) {
                                        return /(crosswalk)/i.test(u);
                                    }
                                    /**
                                     * LINK应用
                                     *  @method isLink
                                     *  @chainable
                                     *  @return {boolean} [true | false]
                                     *  @example
                            
                                           var bool = bui.platform.isLink();
                            
                                     */
                                    function isLink(argument) {
                                        return navigator.userAgent.indexOf("linkmessenger") > -1
                                    }
                                    /**
                                     * 是否在ie
                                     *  @method isIE
                                     *  @chainable
                                     *  @return {boolean} [true | false]
                                     *  @example
                            
                                           var bool = bui.platform.isIE();
                            
                                     */
                                    function isIE(argument) {
                                        return /(trident)/i.test(u);
                                    }
                                    return {
                                        isAndroid: isAndroid, //是否是安卓
                                        isIphone: isIphone, //是否是Iphone
                                        isIpad: isIpad, //是否是Ipad
                                        isIos: isIos, //是否是IOS系统,包含Iphone及Ipad
                                        isWeiXin: isWeiXin, //是否是微信浏览器
                                        isPC: isPC, //是否是PC 
                                        isMobile: isMobile, //是否是移动端
                                        isWeiXinMini: isWeiXinMini, //是否是小程序环境
                                        isBingotouch: isBingotouch, //是否是BT应用
                                        isLink: isLink, //是否是Link应用
                                        isMac: isMac, //是否是Mac系统
                                        isIE: isIE, //是否IE浏览器
                                        isIphoneX: isIphoneX, //是否是Mac系统
                                        isIphoneFull: isIphoneFull, //是否是Iphone全面屏
                                        isWindow: isWindow //是否是Window系统
                                    }
                                })();
                            
                                return ui;
                            
                            })(window.bui || {}, window.libs);
