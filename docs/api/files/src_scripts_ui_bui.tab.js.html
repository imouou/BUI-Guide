                                ui.tab = function (option) {
                                    var config = {
                                        id: "",
                                        uid: "",
                                        menu: ".bui-tab-head > ul",
                                        children: ".bui-tab-main > ul",
                                        header: "header",
                                        footer: "footer",
                                        main: ".bui-tab-main",
                                        item: "li",
                                        prev: ".bui-tab-prev",
                                        next: ".bui-tab-next",
                                        data: null,
                                        showNav: true,
                                        navscroll: false,
                                        loadall: false,
                                        position: "top", // top bottom left right
                                        iconBadge: true,
                                        iconPosition: "left",
                                        loaded: null,
                                        compiled: null,
                                        sideWidth: 0,
                                        align: "center",
                                        alignClassName: "",
                                        stopHandle: "",
                                        width: 0,
                                        height: 0,
                                        relative: false,
                                        zoom: false,
                                        swipe: true,
                                        animate: true,
                                        bufferEffect: false,
                                        visibleNum: 1,
                                        scrollNum: 1,
                                        distance: 40,
                                        direction: "x",
                                        autoplay: false,
                                        autoheight: false,
                                        scroll: true,
                                        index: 0,
                                        fullscreen: false,
                                        autopage: false,
                                        autoload: false,
                                        autoinit: true,
                                        template: null,
                                        templateMenuItem: null,
                                        async: true, // 如果改为同步, 会导致"load"事件监听第一次不触发
                                        callback: null,
                                        onRemove: null,
                                        onBeforeInit: null,
                                        onClickNav: null,
                                        onInited: null,
                                        onStart: null,
                                        onMove: null,
                                        onEnd: null
                                    };
                                    //用于option方法的设置参数
                                    var param = module.config = $.extend(true, {}, config, ui.config.tab, option);
                                    var uiTab = null;
                                    var datas = null;
                                    var templates = null;
                                    var menuGuid = ui.guid();
                            
                                    var cacheItem = null;
                            
                                    // 初始化
                                    uiTab = init(param);
                            
                                    function init(opt) {
                            
                                        datas = $.extend(true, [], opt.data);
                            
                                        if (opt.data) {
                            
                                            opt.menu = opt.menu.indexOf("#") > -1 ? opt.menu : `.${menuGuid}-menu`;
                                            opt.children = `.${menuGuid}-main`;
                                            render(datas);
                                            opt.data = null;
                                        }
                            
                                        var $side = ui.$(".bui-tab-sidenav");
                            
                                        // 左边需要单独处理
                                        if (datas && opt.position == "left") {
                            
                            
                                            opt.sideWidth && $side.width(opt.sideWidth);
                            
                                            var sideNavWidth = opt.sideWidth || $side.width();
                                            opt.width = opt.width || $(window).width() - sideNavWidth;
                            
                                            // 增加样式
                                            ui.$(opt.id).addClass("bui-tab-left");
                                            ui.$(opt.id).css({
                                                "margin-left": sideNavWidth
                                            });
                                            $side.css("left", -sideNavWidth);
                            
                                        } else if (datas && opt.position == "right") {
                                            opt.sideWidth && $side.width(opt.sideWidth);
                            
                                            var sideNavWidth = opt.sideWidth || $side.width();
                                            opt.width = opt.width || $(window).width() - sideNavWidth;
                            
                                            // 增加样式
                                            ui.$(opt.id).addClass("bui-tab-right");
                            
                                            $side.css({
                                                "left": "100%"
                                            });
                                        }
                            
                                        // 初始化
                                        var onTo = opt.onTo || null;
                                        opt.onTo = null;
                                        var tab = ui.slide(opt);
                            
                            
                                        // 实例化以后才能修改overflow属性
                                        if (opt.position == "left" || opt.position == "right") {
                            
                                            ui.$(opt.id).css({
                                                "overflow": "visible"
                                            });
                                        }
                            
                                        // 有动态数据才需要编译
                                        if (datas) {
                                            // tab有多种操作跳转,滑动,点击,都会触发to事件, 在这里拿到对应的索引, 执行加载.
                                            tab.on("to", function (i, opt) {
                            
                                                var index = this.index();
                                                var data = datas && datas[index] || {};
                                                var id = param.loadall ? ".delay-item" : `#${data["id"]}`;
                            
                                                // 加载delay属性的组件
                                                data.name && loader.delay({
                                                    id: id,
                                                    param: $.extend({}, data["param"], opt),
                                                    everytime: data.everytime,
                                                    loaded: param.loaded,
                                                    compiled: param.compiled
                                                })
                            
                                                // uiTab.showContent();
                            
                                                onTo && onTo.call(tab, index);
                                            });
                                            let needAnimate = param.animate ? "" : "none";
                                            tab.to(opt.index, needAnimate);
                                        }
                            
                                        // 存储实例
                                        opt.uid && ui.history.setUI && ui.history.setUI({
                                            uid: opt.uid,
                                            ui: tab
                                        })
                            
                                        return tab;
                                    }
                            
                            
                                    uiTab.name = "tab";
                                    uiTab.templateNav = templateNav;
                                    uiTab.renderNav = renderNav;
                                    uiTab.addBadge = addBadge;
                                    uiTab.removeBadge = removeBadge;
                            
                                    var oldAdd = uiTab.add;
                                    var oldRemove = uiTab.remove;
                                    var hasRemoveEvent = false;
                            
                                    uiTab.add = function (opt) {
                            
                                        var _opt = $.extend(true, {
                                            title: "",
                                            needRemove: false,
                                            templateMenuItem: null
                                        }, opt);
                                        // 不为空都要做默认的菜单渲染
                            
                                        _opt.menuTemplate = opt.templateMenuItem || param.templateMenuItem && param.templateMenuItem(_opt, datas.length) || _opt.title && templateMenuItem(_opt, datas.length) || "";
                            
                                        let onBeforeRemove = _opt.onBeforeRemove || param.onBeforeRemove || null;
                                        let onRemove = _opt.onRemove || param.onRemove || null;
                            
                                        datas.push(_opt);
                            
                                        // 后续再看是否需要把新增数据跟datas对应起来。
                                        if (_opt.needRemove && !hasRemoveEvent) {
                            
                                            var $id = ui.obj(this.config.id);
                                            var _slideption = this.config;
                                            var $menu = _slideption.menu.indexOf("#") > -1 ? ui.obj(_slideption.menu) : $id.find(_slideption.menu).eq(0);
                                            $menu.off("click.bui", ".icon-close").on("click.bui", ".icon-close", function (e) {
                                                var index = $(this).parents(".bui-btn").index();
                                                // 1.7.6
                                                e.index = index;
                                                let canRemove = onBeforeRemove && onBeforeRemove.call(uiTab, { index: index }, e)
                                                if (canRemove === false) {
                                                    return;
                                                }
                            
                                                uiTab.remove({ index: index }, e);
                            
                                                onRemove && onRemove.call(uiTab, { index: index }, e);
                                            })
                            
                                            hasRemoveEvent = true;
                                        }
                            
                                        return oldAdd(_opt);
                                    }
                            
                                    // 移除，支持模块名，重写remove
                                    uiTab.remove = function (opt) {
                            
                                        var _opt = $.extend(true, {
                                            index: -1,
                                            keyname: "id",
                                            target: "",
                                            callback: null
                                        }, opt);
                            
                                        let index = _opt.index;
                                        let keyname = _opt.keyname;
                                        let target = _opt.target.indexOf("#") > -1 ? _opt.target.substr(1) : _opt.target || "";
                            
                                        // 静态
                                        if (ui.typeof(datas) === "array" && target) {
                                            // 检测模块名的索引
                                            index = ui.array.index(datas, target, keyname);
                            
                                            index > -1 && ui.array.deleteIndex(datas, index);
                            
                                        } else if (ui.typeof(datas) === "array" && _opt.index > -1) {
                                            ui.array.deleteIndex(datas, index);
                                        }
                            
                                        _opt.index = index;
                            
                                        return oldRemove(_opt);
                                    }
                            
                                    // 重写获取数据
                                    uiTab.getData = function (opt) {
                            
                                        return datas;
                                    }
                            
                                    // 重写获取数据
                                    /**
                                     * 获取data整个item数据， data 参数的时候才会有效
                                     *  @method getItem
                                     *  @since 1.6.7
                                     *  @param {number|string} [index] [ 获取第几个的数据，默认为当前tab的索引,字符串则获取id比对的数据 ]
                                     *  @return {object} [返回数组]
                                     *  @chainable
                                     * 
                                     * // 默认是当前索引
                                     * let data = uiTab.getItem();
                                     * // 通过索引
                                     * let data = uiTab.getItem(1);
                                     * // 通过id获取
                                     * let data = uiTab.getItem("id");
                                     */
                                    uiTab.getItem = function (currentIndex) {
                                        if (typeof currentIndex === "string") {
                                            return ui.array.get(datas, currentIndex, "id");
                                        }
                                        let cindex = typeof currentIndex === 'number' ? currentIndex : uiTab.index();
                                        let item = datas[cindex];
                                        return item;
                                    }
                            
                            
                                    // 设置当前页的参数
                                    /**
                                     * 设置修改data每个item的数据，有传 data 参数的时候才会有效
                                     *  @method setItem
                                     *  @since 1.6.7
                                     *  @param {number|string} [index] [ 获取第几个的数据，默认为当前tab的索引,传字符串会比对id ]
                                     *  @return {object} [返回数组]
                                     *  @chainable
                                     * 
                                     * uiTab.setItem(1,{param:{}});
                                     */
                                    uiTab.setItem = function (index, opt) {
                                        if (typeof index === "string" && datas) {
                                            let item = ui.array.get(datas, index, "id");
                                            item = $.extend(item, opt);
                                            return item;
                                        }
                                        var index = typeof index === "number" ? index : nowIndex;
                            
                                        datas[index] && (datas[index] = $.extend(datas[index], opt));
                                        return datas[index];
                                    }
                            
                            
                                    /**
                                     * 获取data的每个item的param数据， data 参数的时候才会有效
                                     *  @method getItemParam
                                     *  @since 1.6.7
                                     *  @param {number|string} [index] [ 获取第几个的数据，默认为当前tab的索引,字符串则获取id比对的数据 ]
                                     *  @return {object} [返回数组]
                                     *  @chainable
                                     * 
                                     * // 默认是当前索引
                                     * let data = uiTab.getItemParam();
                                     * // 通过索引
                                     * let data = uiTab.getItemParam(1);
                                     * // 通过id获取
                                     * let data = uiTab.getItemParam("id");
                                     */
                                    uiTab.getItemParam = function (index) {
                            
                                        if (typeof index === "string") {
                                            return ui.array.get(datas, index, "id")["param"];
                                        }
                                        var index = typeof index === "number" ? index : nowIndex;
                            
                                        return datas[index]["param"] || {};
                                    }
                            
                                    /**
                                     * 设置data指定item的param参数，有传 data 参数的时候才会有效
                                     *  @method setItemParam
                                     *  @since 1.6.7
                                     *  @param {number|string} [index] [ 获取第几个的数据，默认为当前tab的索引,传字符串会比对id ]
                                     *  @return {object} [返回数组]
                                     *  @chainable
                                     * 
                                     * uiTab.setItemParam(1,{type:"xxx"});
                                     */
                                    uiTab.setItemParam = function (index, opt) {
                                        if (typeof index === "string" && datas) {
                                            let item = ui.array.get(datas, index, "id");
                                            item.param = $.extend(item["param"], opt);
                                            return item.param;
                                        }
                                        var index = typeof index === "number" ? index : nowIndex;
                                        let item = datas[index];
                                        item && (item = $.extend(item, opt));
                                        return item["param"];
                                    }
                            
                            
                            
                                    /**
                                     * 新增修改红点
                                     *  @method addBadge
                                     *  @since 1.6.7
                                     *  @param {string} index 在第几个索引增加红点
                                     *  @param {string} [num] 有多少条未读
                                     *  @chainable
                                     * 
                                     * uiTab.addBadge(1,5);
                                     */
                                    function addBadge(index, num) {
                            
                                        if (index && !(/[0-9]/.test(index))) {
                                            return this;
                                        }
                                        var $item = ui.$(param.id).find(param.menu).children().eq(index);
                                        var hasNum = typeof num !== "undefined" ? num : "";
                                        var $badges = $item.find(".bui-badges");
                                        var $icon = $item.find("i");
                                        if ($badges.size()) {
                                            $badges.text(num);
                                        } else if ($icon.size() && param.iconBadge) {
                                            $icon.append(`<span class="bui-badges">${hasNum}</span>`);
                                        } else {
                                            $item.append(`<span class="bui-badges">${hasNum}</span>`);
                                        }
                                        return this;
                            
                                    }
                            
                                    /**
                                     * 隐藏内容
                                     *  @method hideContent
                                     *  @since 1.7.5
                                     *  @param {function} callback 回调
                                     *  @chainable
                                     * 
                                     * uiTab.hideContent();
                                     */
                                    uiTab.hideContent = function (callback) {
                            
                                        let $tab = ui.$(param.id);
                                        let index = uiTab.index();
                                        cacheItem = {
                                            index: index,
                                            parentWidth: $tab.css("width"),
                                            parentHeight: $tab.css("height")
                                        };
                            
                                        $tab.find(param.menu).children().removeClass("active");
                            
                                        $tab.children(".bui-tab-main").css("display", "none");
                            
                                        $tab.css("height", "auto");
                            
                                        callback && callback.call(this, cacheItem);
                            
                                        return this;
                            
                                    }
                            
                                    /**
                                     * 显示内容
                                     *  @method showContent
                                     *  @since 1.7.5
                                     *  @param {function} callback 回调
                                     *  @chainable
                                     * 
                                     * uiTab.showContent();
                                     */
                                    uiTab.showContent = function (callback) {
                            
                                        let $tab = ui.$(param.id);
                            
                                        $tab.children(".bui-tab-main").css("display", "");
                            
                                        let hasActive = $tab.find(param.menu + " .active");
                                        cacheItem && !hasActive.length && $tab.find(param.menu).children().eq(cacheItem.index).addClass("active");
                            
                                        cacheItem && $tab.css("height", cacheItem.parentHeight);
                            
                                        callback && callback.call(this, cacheItem);
                            
                            
                                        return this;
                            
                                    }
                            
                                    /**
                                     * 切换内容
                                     *  @method toggleContent
                                     *  @since 1.7.5
                                     *  @param {function} showCallback 显示的回调
                                     *  @param {function} hideCallback 隐藏的回调
                                     *  @chainable
                                     * 
                                     * uiTab.toggleContent();
                                     */
                                    uiTab.toggleContent = function (showCallback, hideCallback) {
                                        let $tab = ui.$(param.id);
                            
                                        let displays = $tab.children(".bui-tab-main").css("display");
                                        let isShowContent = true;
                            
                                        if (displays == "none") {
                                            this.showContent(showCallback);
                                            isShowContent = true;
                                        } else {
                                            this.hideContent(hideCallback);
                                            isShowContent = false;
                                        }
                            
                                        return isShowContent;
                            
                                    }
                                    /**
                                     * 当前tab的显示状态
                                     *  @method isShow
                                     *  @since 1.7.5
                                     *  @chainable
                                     * 
                                     * uiTab.isShow();
                                     */
                                    uiTab.isShow = function () {
                                        let $tab = ui.$(param.id);
                            
                                        let displays = $tab.children(".bui-tab-main").css("display");
                            
                                        let showstatus = true;
                                        if (displays == "none") {
                                            showstatus = false;
                                        }
                            
                                        return showstatus;
                            
                                    }
                            
                                    /**
                                     * 移除红点
                                     *  @method removeBadge
                                     *  @since 1.6.7
                                     *  @param {string} index 在第几个索引
                                     *  @chainable
                                     * 
                                     * uiTab.removeBadge(1);
                                     */
                                    function removeBadge(index) {
                                        var $menu = ui.$(param.id).find(param.menu).children();
                                        if (typeof index !== "undefined") {
                                            $menu.eq(index).find(".bui-badges").remove();
                                        } else {
                                            $menu.find(".bui-badges").remove();
                                        }
                                        return this;
                                    }
                                    /**
                                     * 刷新加载当前组件
                                     *  @method refresh
                                     *  @since 1.6.7
                                     *  @param {object} option 
                                     *  @param {string} index 在第几个索引
                                     *  @param {object} param 修改第几个并传参
                                     *  @param {object} beforeLoad 刷新前执行
                                     *  @param {object} loaded 刷新后执行
                                     *  @chainable
                                     * 
                                     * uiTab.refresh(1);
                                     */
                                    uiTab.refresh = function (opt) {
                            
                                        let index = typeof opt === "object" ? opt.index : opt;
                                        let nowindex = typeof index === "number" ? index : uiTab.index();
                            
                                        if (ui.typeof(datas) === "array") {
                            
                                            let item = datas[nowindex];
                            
                                            item.url = item.name + ".html";
                                            item.param = $.extend(false, {}, item.param, opt && opt.param || {});
                            
                                            loader.load(item);
                            
                                        } else {
                                            console.warn("刷新方法仅在data是数组的方式使用")
                                        }
                                        return this;
                                    }
                            
                                    function render(data) {
                            
                                        templates = param.template || template;
                            
                                        var html = templates(data);
                            
                                        ui.$(param.id).html(html);
                                    }
                            
                                    function renderNav(id, data) {
                                        var html = templatesNav(data || datas);
                            
                                        ui.$(id).html(html);
                                    }
                            
                                    // 头部结构
                                    function templateHead(data) {
                                        var classHead = param.position == "bottom" ? "bui-tab-foot" : "";
                            
                                        var headHtml = `<div class="bui-tab-head ${classHead}">`;
                                        headHtml += templateNav(data);
                                        headHtml += `</div>`;
                            
                                        return headHtml;
                                    }
                            
                                    function templateNav(data) {
                                        var sideClass = param.position == "left" || param.position == "right" ? " bui-tab-sidenav" : "";
                                        var navbarHtml = param.navscroll ? `<div class="bui-navbar">` : "";
                                        var headHtml = `${navbarHtml}<ul class="bui-nav ${menuGuid}-menu ${sideClass}">`;
                                        var tplItem = param.templateMenuItem ? param.templateMenuItem : templateMenuItem;
                                        data && data.forEach(function (item, index) {
                            
                                            headHtml += tplItem(item, index);
                                        })
                                        headHtml += `</ul>
                                            ${param.navscroll ? "</div>" : ""}`;
                                        return headHtml;
                                    }
                                    // 菜单结构
                                    function templateMenuItem(item, index) {
                            
                                        var html = "";
                                        var hasIcon = templateIcon(item, index) || templateImage(item, index);
                                        var hasRemoveIcon = item.needRemove ? templateIconRemove(item, index) : "";
                                        var className = item.className ? `${item.className}` : "";
                                        switch (param.iconPosition) {
                                            case "left":
                                                html += `<li class="bui-btn bui-align-${param.align} ${className}"><span class="bui-text-wrap">${hasIcon}${item.title || ""}${hasRemoveIcon}</span></li>`;
                                                break;
                                            case "right":
                                                html += `<li class="bui-btn bui-align-${param.align} bui-box ${className}"><div class="span1">${item.title || ""}</div>${hasIcon}${hasRemoveIcon}</li>`;
                                                break;
                                            case "bottom":
                                                html += `<li class="bui-btn bui-align-${param.align} ${hasIcon ? `bui-box-vertical` : "bui-box"} ${className}"><div class="span1">${item.title || ""}</div>${hasIcon}${hasRemoveIcon}</li>`;
                                                break;
                                            default:
                                                html += `<li class="bui-btn bui-align-${param.align} ${hasIcon ? `bui-box-vertical` : "bui-box"} ${className}">${hasIcon}<div class="span1">${item.title || ""}${hasRemoveIcon}</div></li>`;
                                                break;
                                        }
                            
                                        return html
                                    }
                            
                                    function templateIcon(item, index) {
                            
                                        var html = item.icon ? (item.icon.indexOf("&#") > -1 || item.icon.indexOf("svg") > -1 ? `<i class="icon">${item.icon}</i>` : `<i class="${item.icon}"></i>`) : "";
                                        return html;
                                    }
                            
                                    // 移除的图标
                                    function templateIconRemove(item, index) {
                            
                                        var html = `<i class="icon-close"></i>`;
                                        return html;
                                    }
                            
                                    function templateImage(item, index) {
                            
                                        var html = item.image ? `<i class="icon"><img src="${item.image}" /></i>` : "";
                                        return html;
                                    }
                            
                                    // 主要结构
                                    function templateMain(data) {
                                        var leftHtml = param.position == "left" || param.position == "right" ? `<div class="span1">` : "";
                                        var mainHtml = `${leftHtml}<div class="bui-tab-main"><ul class="${menuGuid}-main">`;
                            
                                        data && data.forEach(function (item, index) {
                                            mainHtml += item.content ? `<li class="bui-tab-main-item">${item.content}</li>` : `<li class="bui-tab-main-item">
                                                        <component id="${item.id}" class="bui-tab-item-component delay-item" name="${item.name && item.name.indexOf("/") === 0 ? item.name.substr(1) : item.name}" delay="true"></component>
                                                    </li>`;
                                        })
                                        mainHtml += `</ul></div>`;
                                        mainHtml += param.position == "left" || param.position == "right" ? "</div>" : "";
                                        return mainHtml
                                    }
                            
                                    function template(data) {
                                        var html = "";
                                        // 菜单用id代表分离结构
                                        var headHtml = param.showNav ? templateHead(data) : "";
                                        var mainHtml = templateMain(data);
                            
                                        switch (param.position) {
                                            case "top":
                                                html = headHtml + mainHtml;
                                                break;
                                            case "bottom":
                                                html = mainHtml + headHtml;
                                                break;
                                            default:
                                                html = headHtml + mainHtml;
                                                break;
                                        }
                            
                                        return html;
                                    }
                            
                                    return uiTab;
                                }
                            
                                return ui;
                            
                            })(window.bui || {}, window.libs);
