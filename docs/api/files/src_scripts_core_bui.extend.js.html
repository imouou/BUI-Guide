                                // 插件原型
                                class PluginsCenter {
                                    constructor(opt) {
                            
                                        this._default = {
                                            log: false,
                                            hasLife: true,
                                            onBeforeInit: null,
                                            onInited: null
                                        };
                            
                                        // 存储插件的一些配置信息
                                        this._plugin = opt || {
                                            name: "", // 插件名称
                                            depend: null, // 复制多个插件的所有方法
                                            config: null,
                                            param: null, // 用户传的初始化参数
                                            callback: null
                                        };
                            
                                        // 全局配置
                                        let globalConfig = ui.config.hasOwnProperty(this._plugin.name) ? ui.config[this._plugin.name] : null;
                            
                                        // 插件的配置参数, 合并默认的参数
                                        this.config = $.extend(false, {}, this._default, this._plugin.config, globalConfig, this._plugin.param);
                            
                                        this.handle = {};
                                        // 是否初始化过,可以用来判断事件重复初始化
                                        this.initStatus = false;
                                        // 是否是重新初始化
                                        this.reinitStatus = false;
                                        // 挂载内部依赖控件
                                        this.widgets = Object.create(null);
                            
                                        // 初始化
                                        this.init();
                            
                                    }
                            
                                    /**
                                     * 实例的公共方法, 获取插件里面的实例, 
                                     *  @method widget
                                     *  @param {string} [name] [ 获取依赖的时候 ]
                                     *  @example
                                        
                                        插件内部使用 that.widgets.loading = bui.loading(); 初始化, 实例才能通过widget方法获取得到内部依赖.
                            
                                        比方初始化了插件: 
                                        var accordion = bui.accordion();
                            
                                        // 实例获取依赖
                                        accordion.widget("loading");
                            
                                    */
                                    widget(name) {
                                        return name && name in this.widgets ? this["widgets"][name] : this["widgets"];
                                    }
                            
                                    /**
                                    * [控件的初始化]
                                    *  @method init
                                    *  @since 1.5.3
                                    *  @example
                            
                                        //初始化生命周期,默认已经初始化1次
                                        var accordion = bui.accordion();
                            
                                        // 执行多一次初始化
                                        accordion.init();
                            
                                    */
                                    init(opt) {
                            
                                        let that = this;
                                        let opts = opt || null;
                                        // 合并初始化参数
                                        that.config = $.extend(true, this.config, opts);
                            
                                        let config = that.config;
                                        // 统一的声明周期
                                        let canInit = config.hasLife && config.onBeforeInit && config.onBeforeInit.call(that, config);
                                        config.log && console.log(`${that._plugin.name} before init`);
                            
                                        if (canInit === false) {
                                            return that;
                                        }
                                        // 初始化用户自定义的插件
                                        let distance = that._plugin.callback && that._plugin.callback.call(that, config);
                                        config.log && console.log(`${that._plugin.name} init`);
                            
                                        // 统一的声明周期
                                        config.hasLife && config.onInited && config.onInited.call(that, config);
                                        config.log && console.log(`${that._plugin.name} after inited`);
                            
                                        // 已经初始化过
                                        this.initStatus = true;
                            
                            
                                        // 存储实例
                                        that.config.uid && ui.history.setUI && ui.history.setUI({
                                            uid: that.config.uid,
                                            ui: that
                                        })
                            
                                        return that;
                                    }
                            
                                    /**
                                     * 插件的公共方法,重新初始化.
                                     *  @method reinit
                                     *  @example
                                        
                                        
                                        实例:
                                        var accordion = bui.accordion();
                                        // 重新初始化
                                        accordion.reinit();
                            
                                    */
                                    reinit() {
                            
                                        // 小销毁
                                        this.destroy();
                            
                                        this.init();
                            
                                        this.config.log && console.log(`${this._plugin.name} reinit`);
                                        return this;
                                    }
                            
                                    /**
                                     * 插件的公共方法,销毁控件.
                                     *  @method destroy
                                     *  @param {function} [callback] [ 销毁后做什么 ]
                                     *  @example
                                        
                                        插件内部使用: 
                                        // 插件内部要销毁的东西, 比方 定时器,事件绑定, 外部是不知道的, 所以要内部处理
                                        that.beforeDestroy = function(){
                            
                                        }
                            
                                        实例的销毁:
                                        var accordion = bui.accordion();
                                        // 调用这个方法前, 会先执行 beforeDestroy. 
                                        accordion.destroy();
                            
                                    */
                                    destroy(callback) {
                                        // 先执行内部销毁,比方定时器,事件
                                        this.beforeDestroy && this.beforeDestroy();
                                        // 释放
                                        this.beforeDestroy = null;
                            
                                        // 清空依赖
                                        this.widgets = {};
                                        // 清空订阅事件
                                        this.off();
                            
                                        // 去掉初始化状态
                                        this.initStatus = false;
                            
                                        callback && callback.call(this);
                            
                                        let uid = this.config.uid;
                                        if (uid) {
                                            var currentHistory = ui.history.getLast();
                                            currentHistory["ui"][uid] = null;
                                        }
                            
                                        this.config.log && console.log(`${this._plugin.name} destroyed`);
                                        return this;
                                    }
                            
                                    /**
                                     * 处理样式的前缀.
                                     *  @method prefix
                                     *  @param {string} [str] [ 去除前缀以后的值 ]
                                     *  @return {string} [ 出现问题的位置 ]
                                     *  @example
                                        
                                        var className = that.prefix("accordion");
                            
                                        // bui-accordion
                            
                                    */
                                    prefix(className) {
                                        return ui.config.classNamePrefix + className;
                                    }
                            
                                    /**
                                     * 插件内部统一的抛出日志方法.
                                     *  @method showLog
                                     *  @param {string|object} [str] [ 日志说明, try{}catche(e){ that.showLog(e) } ]
                                     *  @param {string} [where] [ 出现问题的位置 ]
                                     *  @example
                                        
                                        // 抛出异常, 在bui.accordion init 方法
                                        that.showLog("初始化错误","bui.accordion init")
                            
                                        // 支持对象
                                        try{}catche(e){ that.showLog(e) }
                            
                                    */
                                    showLog(e, where) {
                                        where = where || "";
                                        let message = "";
                                        if (typeof e == "object" && "message" in e && "name" in e) {
                                            message = e.message + ':' + e.name + '&&stack:' + e.stack;
                                        } else if (typeof e == "string") {
                                            message = e;
                                        }
                                        ui.log && console.error(where + " " + message);
                                        ui.trace && console.trace && console.trace();
                                        return this;
                                    }
                            
                                    /**
                                     * 外部统一的修改获取参数, 修改this.config 的参数, 并且重新调用 init 方法.
                                     *  @method option
                                     *  @param {string|object} [key] [ 获取依赖的时候 ]
                                     *  @param {string} [value] [ 获取依赖的时候 ]
                                     *  @example
                                        
                                        
                                        比方初始化了插件: 
                                        var accordion = bui.accordion();
                            
                                        // 实例获取参数
                                        accordion.option("id");
                            
                                        // 修改一个参数
                                        accordion.option("height",500);
                            
                                        // 修改多个参数
                                        accordion.option({"height":500,handle:".bui-btn-test"});
                            
                                    */
                                    option(key, value) {
                            
                                        if (ui.typeof(key) !== "object" && typeof value === "undefined") {
                                            // 读取
                                            if (typeof key === "string") {
                                                return this.config[key];
                                            } else {
                                                return this.config;
                                            }
                                        } else {
                                            let instanceUI = this;
                                            // 修改
                                            if (key == "id") {
                                                ui.showLog("不允许修改控件的ID参数");
                                                return this;
                                            }
                            
                                            if (key && ui.typeof(key) === "object") {
                                                let opt = $.extend(this.config, key);
                                                // 再初始化
                                                return instanceUI && instanceUI.init(opt);
                                            } else if (this.config.hasOwnProperty(key)) {
                            
                                                let obj = {};
                                                obj[key] = value;
                            
                                                let opt = $.extend(this.config, obj);
                            
                                                // 再初始化
                                                return instance && instance.init(opt);
                                            }
                                        }
                                        return this;
                                    }

                                        return ui.emitter();
                                    }
                            
                                    /**
                                     * 为控件绑定事件
                                     *  @event on
                                     *  @param {string} [type] [ 事件类型: "show"(显示目标时) | "hide"(隐藏目标时)  ]
                                     *  @param {function} [callback] [ 绑定的事件, this 为当前点击的菜单 ]

                                            console.log(this);
                                        });
                            
                            
                                        */
                                    get on() {
                                        return this.emitter.on;
                                    }
                            
                                    /**
                                     * 为控件取消绑定事件
                                     *  @event off
                                     *  @param {string} [type] [ 事件类型: "show"(显示目标时) | "hide"(隐藏目标时)  ]
                                     *  @param {function} [callback] [ 绑定的事件, this 为当前点击的菜单 ]
                                     *  @example
                            
                                        that.off("show");
                            
                                    */
                                    get off() {
                                        return this.emitter.off;
                                    }

                                     * 触发自定义事件
                                     *  @event trigger

                                     *  @param {string|object|number} [arguments] [ 需要传给自定义事件的参数,可以有多个 ]
                                     *  @example

                                ui.extend = function (option) {
                            
                                    // 安装插件
                                    class Install {
                                        constructor(opt) {
                                            this.init(opt)
                                        }
                                        init(opt) {
                            
                                            let name = opt.name;
                                            // 检测是否重名
                                            let canRegister = this.checkName(name);
                                            if (!canRegister) {
                                                return ui[name];
                                            }
                            
                                            // 放在第1层调用
                                            ui[name] = function (opts) {
                            
                                                option.param = opts;
                                                return new PluginsCenter(option);
                                            }
                                            return this;
                                        }
                                        checkName(name) {
                                            // true则可以注册
                                            if (!name) {
                                                console.warn(`插件名字不能为空`);
                                                return false;
                                            }
                                            if (ui.hasOwnProperty(name)) {
                                                // console.warn(`已有相同名为${name}的插件或方法,请重新换个名称注册.`);
                                                return false;
                                            }
                                            this.name = name;
                                            return true;
                                        }
                                    }
                            
                                    // // 每一个插件就一个实例, 订阅事件上才不会冲突
                                    var pluginsInstall = new Install(option);
                                }
                                return ui;
