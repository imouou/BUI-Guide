                                    function transformTime(d, formate) {
                                        if (d && d.constructor == Date) {
                                            return d;
                                        }
                                        // 如果是时间戳,只能是数字
                                        if (typeof d === "number" && String(d).length === 13) {
                                            return new Date(d);
                                        }
                            
                                        d = String(d);
                            
                                        var date = new Date();
                                        var year = date.getFullYear(),
                                            month = date.getMonth() + 1,
                                            today = date.getDate();
                                        /* 方法 一 */
                                        // 在Chrome和Safari中,存在当值为'2017-9'或'2017/09'时,new Date(value)会返回非法值,
                                        // 故在统一使用斜杠/格式的时候,还须为时间值补充日期
                                        d = d.replace(/[-|年|月]/img, '/').replace(/[时|分|秒|小时|分钟]/img, ':')
                                            .replace(/^(\d{4}\/\d+?)($|\s)/, function ($0, $1) {
                                                return $1 + '/1'
                                            })
                                            .replace(/[日|号]/img, '')
                                            .replace(/^(\d{1,2}\/\d{1,2}?)($|\s)/, function ($0, $1) {
                                                return date.getFullYear() + '/' + $1
                                            })
                            
                                        // 修复只有时间的时候无法格式化问题
                                        if (d.indexOf("/") < 0 && d.indexOf(":") > -1) {
                                            d = year + "/" + month + "/" + today + " " + d;
                                        }
                            
                                        // 修复只有日期的时候无法格式化问题
                                        if (d.length === 4) {
                                            d = d + "/1/1 00:00:00";
                                        } else if (formate === "MM" || formate === "M") {
                                            d = year + "/" + d + "/1 00:00:00";
                                        } else if (formate === "dd" || formate === "d") {
                                            d = year + "/" + month + "/" + d + " 00:00:00";
                                        }
                                        // else if( formate && formate.indexOf("y") < 0 && formate.indexOf("M") > -1 ){
                                        //     d = year + "/" + d;
                                        // }
                                        return new Date(d);
                                    }

                                        var fM = {
                                            y: function (d, l) {
                                                return d.getFullYear().toString().slice(-l);
                                            },
                                            M: function (d, l) {
                                                var s = l > 1 ? '0' : '';
                                                return (s + (d.getMonth() + 1)).slice(-2);
                                            },
                                            d: function (d, l) {
                                                var s = l > 1 ? '0' : '';
                                                return (s + d.getDate()).slice(-2);
                                            },
                                            h: function (d, l) {
                                                var s = l > 1 ? '0' : '';
                                                return (s + d.getHours()).slice(-2);
                                            },
                                            m: function (d, l) {
                                                var s = l > 1 ? '0' : '';
                                                return (s + d.getMinutes()).slice(-2);
                                            },
                                            s: function (d, l) {
                                                var s = l > 1 ? '0' : '';
                                                return (s + d.getSeconds()).slice(-2);
                                            }

                                        /**
                                         * 格式化后回调
                                         *  @method formatDate
                                         *  @param {string} [date]  "2019-11-01" 
                                         *  @param {string} [pattern]  "yyyy-MM-dd hh:00:00" 
                                         *  @example
                            
                                            //设置某个日期
                                            uiPickedate.formatDate("yyyy-MM-dd hh:00:00");
                            
                                         */
                                        this.formatDate = function (str, pattern) {
                                            if (str == null) {
                                                return '';
                                            }
                                            var date = transformTime(str);
                                            pattern = pattern || "yyyy-MM-dd";
                            
                                            var o = {
                                                "M+": date.getMonth() + 1, //月份
                                                "d+": date.getDate(), //日
                                                "h+": date.getHours(), //小时
                                                "m+": date.getMinutes(), //分
                                                "s+": date.getSeconds(), //秒
                                                "q+": Math.floor((date.getMonth() + 3) / 3), //季度
                                                "S": date.getMilliseconds() //毫秒
                                            };
                                            if (/(y+)/.test(pattern)) {
                                                pattern = pattern.replace(RegExp.$1, (date.getFullYear() + "").substr(4 - RegExp.$1.length));
                                            }
                                            for (var k in o) {
                                                if (new RegExp("(" + k + ")").test(pattern))
                                                    pattern = pattern.replace(RegExp.$1, (RegExp.$1.length == 1) ? (o[k]) : (("00" + o[k]).substr(("" + o[k]).length)));
                                            }
                            
                                            return pattern;
                                        }

                                        this.modifyTitle = function (str) {
                            
                                            ui.obj(guid).find(".bui-dialog-head").html(str);
                            
                                            return this;
                                        }
                            
                                        /**
                                         * 允许触发
                                         *  @method enabled
                                         *  @chainable
                                         *  @since 1.4
                                         *  @example
                            
                                            uiPickerdate.enabled();
                            
                                         */
                                        this.enabled = function () {
                                            var $handleItem = ui.obj(opt_handle);
                            
                                            $handleItem && $handleItem.removeClass("disabled");
                                            return this;
                                        }
                            
                                        /**
                                         * [销毁控件]
                                         *  @method destroy
                                         *  @since 1.4.2
                                         *  @param {boolean} [bool]  默认: false 销毁部分 | true 销毁全部   
                                         *  @example
                            
                                            //销毁
                                            uiPickerdate.destroy();
                            
                                         */
                                        this.destroy = function (bool) {
                                            var bool = bool == true ? true : false;
                            
                                            this.off("show");
                                            this.off("hide");
                                            this.off("change");
                            
                                            uiDialog && uiDialog.destroy(bool);
                            
                                            dp = null;
                                        }
                                        /**
                                         * 获取依赖的控件
                                         *  @method widget
                                         *  @param {string} [name]  依赖控件名 dialog 
                                         *  @example
                            
                                            //获取依赖控件
                                            var uiPickerdateWidget = uiPickerdate.widget();
                            
                            
                                         */
                                        this.widget = function (name) {
                                            var control = {
                                                dialog: uiDialog || {}
                                            };
                                            return ui.widget.call(control, name);
                                        };
                            
                                        /**
                                         * 为控件绑定事件
                                         *  @event on
                                         *  @since 1.3.0
                                         *  @param {string} [type]  事件类型: "change"(日期改变的时候) | "show"(日期展示的时候)| "hide"(日期隐藏的时候) 
                                         *  @param {function} [callback]  绑定的事件, this 为当前点击的菜单 
                                         *  @example
                            
                                            uiPickerdate.on("show",function () {
                                                // 点击的菜单
                                                console.log(this);
                                            });
                            
                            
                                         */
                                        this.on = function (type, callback) {
                                            ui.on.apply(dp, arguments);
                                            return this;
                                        }
                            
                                        /**
                                         * 为控件取消绑定事件
                                         *  @event off
                                         *  @since 1.3.0
                                         *  @param {string} [type]  事件类型: "change"(日期改变的时候) | "show"(日期展示的时候)| "hide"(日期隐藏的时候) 
                                         *  @param {function} [callback]  绑定的事件, this 为当前点击的菜单 
                                         *  @example
                            
                                            uiPickerdate.off("show");
                            
                            
                                         */
                                        this.off = function (type, callback) {
                                            ui.off.apply(dp, arguments);
                                            return this;
                            
                                        }
                                        /*
                                         * 触发自定义事件
                                         */
                                        function trigger(type) {
                                            //点击事件本身,或者为空,避免循环引用
                                            dp.self = (this == window || this == dp) ? null : this;
                            
                                            ui.trigger.apply(dp, arguments);
                                        }
                            
                                    }
                            
                                    return function (opt) {
                                        return new Datepicker(opt);
                                    }
                                })();
                            
                                return ui;
