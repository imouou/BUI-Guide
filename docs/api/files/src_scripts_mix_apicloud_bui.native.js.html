                             * <h5>页面交互</h5>
                             * {{#crossLink "bui.load"}}{{/crossLink}}: 页面跳转及传参 <br>

                                /*
                                 * 数据请求
                                 * @namespace  bui
                                 * @class  ajax
                                 * @constructor 
                                 * @param {object} option 
                                 *   @param {string} option.url [请求的地址]
                                 *   @param {object} option.data [传输的数据对象]
                                 *   @param {string} [option.method] [传输的方法 GET | POST]
                                 *   @param {string} [option.dataType] [默认 json | jsonp | script | xml | html | text]
                                 *   @param {string} [option.contentType] [默认 'text/html;charset=UTF-8' | 'application/x-www-form-urlencoded']'
                                 *   @param {number} [option.timeout] [超出这个时间则认为请求失败]
                                 * @example
                                 *     bui.ajax({
                                 *         url: "",
                                 *         data: {}
                                 *     }).then(function(res){
                                 *         console.log(res)
                                 *     },function(res,status){
                                 *         console.log(status);
                                           // status = "timeout" || "error" || "abort", "parsererror"
                                 *     })
                                 * 
                                 */
                                ui.ajax = function (params) {
                            
                                    //延时处理
                                    var def = $.Deferred();
                            
                                    var ajaxHandle,
                                        tag = ui.guid();
                                    params = params || {};
                                    params.data = params.data || {};
                                    params.method = params.method || "GET";
                                    params.dataType = params.dataType || "json";
                                    params.timeout = params.timeout || 60000;
                                    params.headers = params.headers || {};
                                    params.processData = params.processData == false ? false : true;
                                    params.mimeType = params.mimeType || "none";
                                    params.cache = params.cache == false ? false : true;
                            
                            
                                    if (typeof (params.async) == "undefined") {
                                        params.async = true;
                                    }
                                    // 如果没传contentType 就根据不同的请求给予默认的参数值
                                    if (params.method == "GET" && (params.contentType === undefined || params.contentType === "")) {
                                        params.contentType = "text/html;charset=UTF-8";
                                    } else if (params.method == "POST" && (params.contentType === undefined || params.contentType === "")) {
                            
                                        params.contentType = "application/x-www-form-urlencoded";
                                    } else {
                                        params.contentType = params.contentType;
                                    }
                            
                                    // 采用web方式请求
                                    if (this.isWebapp || params.native == false) {
                                        ajaxHandle = $.ajax({
                                            async: params.async,
                                            url: params.url,
                                            data: params.data,
                                            type: params.method,
                                            dataType: params.dataType,
                                            headers: params.headers,
                                            contentType: params.contentType,
                                            timeout: params.timeout,
                                            processData: params.processData,
                                            mimeType: params.mimeType,
                                            cache: params.cache,
                                            success: function (res, status, xhr) {
                                                var obj;
                                                if (typeof res == 'string') {
                                                    try {
                                                        obj = JSON.parse(res);
                                                    } catch (e) {
                                                        obj = res;
                                                    }
                                                } else {
                                                    obj = res || {};
                                                }
                                                def.resolve(obj, status, xhr);
                                            },
                                            error: function (res, status, xhr) {
                                                // status: "timeout", "error", "abort", "parsererror"
                                                var obj;
                                                if (typeof res == 'string') {
                                                    try {
                                                        obj = JSON.parse(res);
                                                    } catch (e) {
                                                        obj = res;
                                                    }
                                                } else {
                                                    obj = res || {};
                                                }
                                                def.reject(obj, status, xhr);
                                            }
                                        })
                            
                                    } else {
                            
                                        // 原生请求暂无 abort 阻止请求方法,所以会导致list 多次重复请求问题. 
                                        var successCallback = function (res, status) {
                            
                                            var obj;
                                            if (typeof res == 'string') {
                                                try {
                                                    obj = JSON.parse(res);
                                                } catch (e) {
                                                    obj = res;
                                                }
                                            } else {
                                                obj = res || {};
                                            }
                                            def.resolve(obj, status);
                                        };
                                        var failCallback = function (res, status) {
                                            var obj;
                                            if (typeof res == 'string') {
                                                try {
                                                    obj = JSON.parse(res);
                                                } catch (e) {
                                                    obj = res;
                                                }
                                            } else {
                                                obj = res || {};
                                            }
                                            def.reject(obj, status);
                                        };
                            
                                        ajaxHandle = api.ajax({
                                            "url": params.url,
                                            "tag": tag,
                                            "data": {
                                                stream: "",        //文件路径（字符串类型）
                                                body: "",            //以纯文本的方式提交数据（字符串类型）
                                                values: params.data,        //以表单方式提交参数（JSON对象）, 如 {"field1": "value1", "field1": "value2"} (直接传JSON对像.)
                                                files: {}            //以表单方式提交文件，支持多文件上传（JSON对象）,如 {"file": "path"}，也支持同一字段对应多文件：{"file":["path1","path2"]}。路径支持绝对路径，以及fs://、widget://、cache://等文件路径协议.
                                            },
                                            "async": params.async,
                                            "method": params.method,
                                            "headers": params.headers,
                                            "contentType": params.contentType,
                                            "dataType": params.dataType,
                                            "cache": params.cache,
                                            "timeout": params.timeout,
                                        }, function (ret, err) {
                                            if (ret) {
                                                successCallback(ret, err)
                                            } else {
                                                failCallback(ret, err)
                                            }
                                        });
                            
                                    };
                            
                                    //中断请求
                                    def.abort = function () {
                            
                                        if (this.isWebapp || params.native == false) {
                                            ajaxHandle && ajaxHandle.abort();
                                        } else {
                                            api.cancelAjax({
                                                tag: tag
                                            });
                                        }
                            
                                    };
                            
                                    return def;
                            
                                };
                            
                                /**
                                 * 页面跳转可以传参 获取参数请看 {{#crossLink "bui.getPageParams"}}{{/crossLink}} 方法
                                 * @namespace  bui
                                 * @class  load
                                 * @constructor 
                                 * @param {object} option 
                                 *   @param {string} option.url 请求的地址
                                 *   @param {object} option.param 传过去另外页面的参数
                                 * @example
                                 *     bui.load({ 
                                 *         url: "",
                                 *         param:{}
                                 *     })
                                 * 
                                 */
                                ui.load = function (params) {
                            
                                    var newurl, param,
                                        url = params.url,
                                        localUrl = window.location.href;
                            
                                    if (!params) {
                                        ui.showLog("should be object like {url:'http://domain',params:{....}}", "bui.load:apicloud");
                                        return;
                                    }
                                    if (!url) {
                                        ui.showLog("url is necessary!", "bui.load:apicloud");
                                        return;
                                    }
                            
                            
                                    //传过去目标页面的值
                                    param = params.param;
                                    //如果开启调试,使用url的方式传值
                                    if (this.isWebapp || params.native == false) {
                            
                                        newurl = ui.setUrlParams(params.url, param);
                            
                                        window.location.href = newurl;
                            
                                    } else {
                                        if (url.indexOf("http://") < 0 || url.indexOf("https://") < 0 || url.indexOf("file://") < 0 || url.indexOf("/") < 0) {
                                            var lastIndex1 = localUrl.lastIndexOf("/"),
                                                lastIndex2 = localUrl.lastIndexOf(".");
                                            url = localUrl.substring(0, lastIndex1) + "/" + url;
                                            params.url = url;
                            
                                            var urlname = url.slice(url.lastIndexOf("/") + 1, (url.lastIndexOf(".") > -1 ? url.lastIndexOf(".") : undefined));
                            
                                            api.openWin({
                                                name: urlname,
                                                url: params.url,
                                                pageParam: param,
                                                slidBackEnabled: false,
                                                vScrollBarEnabled: false,
                                                animation: {
                                                    type: "none",                //动画类型（详见动画类型常量）
                                                    subType: "from_right",       //动画子类型（详见动画子类型常量）
                                                    duration: 250                //动画过渡时间，默认300毫秒
                                                },
                                                hScrollBarEnabled: false
                                            });
                                        }
                                    }
                                };
                            
                                /**
                                 * 获取页面参数,配合 {{#crossLink "bui.load"}}{{/crossLink}} 使用
                                 * @namespace  bui
                                 * @class  getPageParams
                                 * @constructor 
                                 * @param {function} callback 数据都在回调里面
                                 * @example
                                 *     bui.getPageParams(function(res){
                                 *         var urlparam = res;
                                 *
                                 *          console.log(urlparam) //{}
                                 *     });
                                 * 
                                 */
                                ui.getPageParams = function (option) {
                            
                                    var callback,
                                        native;
                                    if (typeof option === "function") {
                                        callback = option || function () { };
                                        native = true;
                                    } else if (option && option.callback) {
                                        callback = option.callback || function () { };
                                        native = option.native;
                                    }
                            
                                    //如果开启调试,使用url的方式取值
                                    if (this.isWebapp || native == false) {
                            
                                        var params = this.getUrlParams();
                            
                                        callback && callback(params);
                            
                                    } else {
                            
                                        var params = api.pageParam;
                            
                                        callback && callback(params);
                                    }
                                };
                                /**
                                 * 后退, 原生可以使用回调,pc调试只能在当前页面触发回调
                                 * @namespace  bui
                                 * @class  back
                                 * @constructor 
                                 * @param {function} callback 在后退的页面触发
                                 * @example

                                ui.back = function (option) {
                            
                                    var callback,
                                        native;

                                        callback = option || function () { };
                                        native = true;
                                    } else if (option && option.callback) {
                                        callback = option.callback || function () { };
                                        native = option.native;
                                    }
                                    if (this.isWebapp || native == false) {
                            
                                        //pc不能在后退以后执行操作
                                        window.history.back();
                            
                                    } else {
                                        var localUrl = window.location.href,
                                            lastIndex1 = localUrl.lastIndexOf("/"),
                                            lastIndex2 = localUrl.lastIndexOf(".");
                                        var urlname = localUrl.slice(lastIndex1 + 1, (lastIndex2 > -1 ? lastIndex2 : undefined));
                            
                                        //后退以后的页面执行
                                        api.closeWin({
                                            name: api.winName
                                        });
                            
                                        callback && callback(api.winName);
                                        // 后退的时候可以发送事件
                                        //api.sendEvent({
                                        //     name: 'myEvent',
                                        //     extra: {
                                        //         key1: 'value1',
                                        //         key2: 'value2'
                                        //     }
                                        // });
                                    }
                            
                                };

                                 * 刷新页面
                                 * @namespace  bui
                                 * @class  refresh
                                 * @constructor 
                                 * @example
                                 *     bui.refresh();
                                 *

                                 * 打开文件
                                 * @namespace  bui
                                 * @class  run
                                 * @param {id} string [应用的id,如果是"http://"则打开浏览器]
                                 * @param {data} {} [启动参数]
                                 * @constructor 
                                 * @example
                                 *     bui.run("http://www.easybui.com/demo/api/");
                                 * 
                                 */
                                ui.run = function (id, data) {
                            
                                    var data = data || { 'appParam': '' };
                            
                                    if (this.isWebapp) {
                                        if (String(id).indexOf("http://") > -1) {
                                            var newurl = ui.setUrlParams(id, data);
                                            var open_link = window.open('', '_newtab');
                                            open_link.location = newurl;
                                        }
                                    } else {
                            
                                        //Android中的使用方法如下：
                                        api.openApp({
                                            iosUrl: id, //'weixin://', //打开微信的，其中weixin为微信的URL Scheme
                                            appParam: data,
                                            androidPkg: 'android.intent.action.VIEW',
                                            mimeType: 'text/html',
                                            uri: id
                                        }, function (ret, err) {
                                            // if (ret) {

                                            // } else {
                                            //     alert(JSON.stringify(err));
                                            // }
                                        });
                                    }
                                }
                            
                            
                                return ui;
