                                            // 重新加载

                                        that.reload = function (opts) {
                            
                                            opts = opts || {};
                                            var opt = options = $.extend(false, that.config, options, opts);
                                            // 参数以传进来的参数为主, 不然数组会被合并
                                            opt.param = opts.param || {};
                            
                                            if (opt.iframe) {
                                                $iframe && $iframe.attr("src", opt.url)
                                                return this;
                                            }
                            
                                            isReload = true;
                                            // 解析url 不允许修改url
                                            urlModule = parsingUrl(opt.url);
                            
                                            // 可以自定义脚本路径
                                            if (opt.script) {
                                                var index = opt.script.indexOf(".js");
                                                urlModule.module = opt.script.substr(0, index);
                                            }
                            
                                            // 加载前执行,可以拦截处理
                                            var canLoad = opt.beforeLoad && opt.beforeLoad.call(that, urlModule);
                            
                                            // 不允许跳转
                                            if (canLoad === false) {
                                                return this;
                                            }
                                            if (typeof urlModule.url === "function") {
                                                res = urlModule.url();
                            
                                                handleContent(res);
                                            } else if (typeof urlModule.url === "string") {
                                                // 动态加载模板
                                                loader.import(urlModule.url, function (res) {
                                                    // 如果有id,加载到指定位置.
                                                    if (typeof res === "object") {
                                                        opt.onError && opt.onError.call(that, res);
                                                        return;
                                                    }
                            
                                                    handleContent(res);
                                                    // 自动加载
                                                    // opt.autoload && uiDialog.open();
                            
                                                });
                                            }
                            
                                            var handleContent = function (res) {
                                                // 处理成局部样式
                                                res = ui.unit.handleContent(res) || "";
                                                res = res.replace(/#module\.id/g, "#" + (uiDialog.config.id || opt.dialogid));
                            
                                                var path = urlModule.url.substr(0, urlModule.url.lastIndexOf("/") + 1);
                                                res = res.replace(/#module\.path\//g, path);
                            
                                                cacheHtml = templateDialog(res, opt);
                            
                                                $dom("#" + (uiDialog.config.id || opt.dialogid)).html(cacheHtml);
                            
                                                // 重选计算宽高
                                                if (opt.resize) {
                                                    uiDialog.resize(opt);
                                                }
                            
                            
                                                if (!opt.scroll) {
                                                    uiDialog.scroll(false);
                                                } else {
                                                    uiDialog.scroll(true);
                                                }
                            
                            
                                                // 加载脚本
                                                requireModule(urlModule.module);
                            
                                                opt.loaded && opt.loaded.call(that, urlModule);
                            
                                            }
                            
                            
                                        }

                                            if (canClose === false) {
                                                return this;
                                            }
                            
                                            // 同步历史记录才会删除
                                            ui.hasRouter && options.syncHistory && ui.history.removeLast(options.syncHistory);
                                            // 动画优先关闭以后再执行方法; 例如图表等方法会比较耗性能.
                                            if (options.closeFirst) {
                            
                                                uiDialog.close(function () {
                                                    that.trigger("close");
                                                    moduleHide(callback);
                                                    hasOpenInit = false;
                                                });
                                                return that;
                                            }
                            
                                            moduleHide(callback);
                                            // 执行关闭
                                            uiDialog.close();
                                            that.trigger("close");
                            
                                            hasOpenInit = false;
                                            return that;
                                        };;
                            
                                        /**
                                         * 移除对话框
                                         *  @method remove
                                         *  @param {function} [callback] 关闭后执行
                                         *  @chainable
                                         *  @example
                             
                                            // 关闭
                                            uiPage.remove();
                                            
                                         *
                                         */
                                        that.remove = function (callback) {
                                            uiDialog.remove(callback);
                                            that.trigger("remove");
                                            hasOpenInit = false;
                                            return that;
                                        }
                            
                                        /**
                                         * 打开
                                         *  @method open
                                         *  @param {function} [callback] 打开后执行
                                         *  @chainable
                                         *  @example
                             
                                            // 打开
                                            uiPage.open();
                                            
                                         *
                                         */
                                        that.open = function (callback) {
                            
                                            var canOpen = options.beforeOpen && options.beforeOpen.call(that);
                            
                                            if (canOpen === false) {
                                                return this;
                                            }
                            
                                            if (options.iframe) {
                                                uiDialog.open(callback);
                                                return that;
                                            }
                            
                                            // 增加历史
                                            // 交给路由去操作
                                            var toggle = {
                                                hide: that.close,
                                                show: that.open
                                            };
                            
                                            // 同步历史记录
                                            ui.hasRouter && options.syncHistory && ui.history.add({
                                                id: uiDialog.config.id,
                                                name: urlModule.module,
                                                url: options.url,
                                                replace: false,
                                                param: options.param || {},
                                                part: {},
                                                page: {},
                                                effect: "push",
                                                toggle: toggle,
                                                syncHistory: options.syncHistory
                                            })
                            
                                            // 把id放到模块里
                                            loader.map({
                                                moduleName: urlModule.module,
                                                id: uiDialog.config.id
                                            })
                                            // 动画优先打开以后再执行方法; 例如图表等方法会比较耗性能.
                            
                                            // 修改最后一个历史记录的页面加载,用于模块内部的参数获取
                                            ui.history.getLast()["currentPage"] = uiDialog.config.id;
                            
                                            // 先执行动画再加载脚本
                                            if (options.openFirst) {
                            
                                                // 默认自动计算main高度
                                                uiDialog.open(function () {
                                                    that.trigger("open");
                                                    // 只打开执行一次
                                                    requireModule(urlModule.module, callback);
                            
                                                })
                            
                                                return that;
                                            }
                            
                            
                                            // 打开前先执行加载再处理打开动画
                                            requireModule(urlModule.module, callback);
                            
                                            uiDialog.open();
                            
                                            that.trigger("open");
                            
                                            return that;
                                        };
                            
                                        // 关闭时的模块处理
                                        function moduleHide(callback) {
                                            // 执行脚本
                                            var mods = loader.map()["modules"][urlModule.module] || {};
                            
                                            // 生命周期隐藏, 单页才有生命周期
                                            if (mods.hide && options.life) {
                                                var lastPage = ui.hasRouter ? ui.history.getLast() : {};
                                                mods.hide({
                                                    type: "page",
                                                    target: lastPage
                                                })
                                            }
                            
                                            typeof callback === "function" && callback.call(that, mods["exports"]);
                            
                                            options.closed && options.closed.call(that, mods);
                                        }
                            
                                        // 传参
                                        that.setParam = function (opt) {
                                            var opts = $.extend(false, {}, options.param, opt);
                            
                                            // 参数需要先传
                                            ui.history.setPage({
                                                id: uiDialog.config.id,
                                                param: opts
                                            })
                                        }
                            
                                        that.minimize = uiDialog.minimize;
                                        that.maxmize = uiDialog.maxmize;
                                        that.revert = uiDialog.revert;
                                        that.resize = uiDialog.resize;
                                        that.top = uiDialog.top;
                                        that.fullscreen = uiDialog.fullscreen;
                                        that.isFullscreen = uiDialog.isFullscreen;
                                        that.scroll = uiDialog.scroll;
                            
                                        // 加载模块并执行回调
                                        function requireModule(urlmod, callback) {
                                            var lastPage = ui.history.getLast();
                            
                                            _modules = loader.map()["modules"] || {};
                            
                                            var _mod = _modules[urlmod];
                            
                                            // 如果有缓存, 并且已经打开过一次, 执行回调就好. 可以在回调里面编译或者做其它事情
                            
                                            if (options.cache && hasModuleInit && !isReload) {
                            
                                                !options.openFirst && uiDialog.open();
                            
                                                // 如果有回调处理回调再进行编译
                                                // loader.require(urlmod,callback)
                                                typeof callback === "function" && callback.call(that, _mod["exports"]);
                                                // 修改打开状态
                                                hasOpenInit = true;
                                                return that;
                                            }
                            
                                            // 修改打开状态
                                            hasOpenInit = true;
                                            // 参数需要先传
                                            ui.history.setPage({
                                                id: uiDialog.config.id,
                                                name: urlmod,
                                                url: options.url,
                                                dialog: that,
                                                param: options.param
                                            })
                                            // 修改模块的module.props 
                                            loader.set(urlmod, {
                                                // 模块选择器
                                                "props": options.param,
                                                "$": function (el) {
                                                    return $("#" + uiDialog.config.id).find(el);
                                                }
                                            });
                            
                                            // 重新渲染结构
                                            // !options.cache && ui.$("#" + uiDialog.config.id).html(cacheHtml)
                                            // 修改加载的状态
                                            isReload = false;
                            
                                            // 如果已经执行过,并且不需要缓存才每次都执行.
                                            if (_mod && typeof _mod["loaded"] === "function") {
                                                // 每次执行loaded
                                                var extportModule = _mod["loaded"] && _mod["loaded"].call(module, _mod["exports"]) || _mod["exports"];
                            
                                                // 修改参数
                                                ui.history.setPage({
                                                    id: uiDialog.config.id,
                                                    name: urlmod,
                                                    url: options.url,
                                                    dialog: uiDialog,
                                                    exports: extportModule,
                                                    param: options.param
                                                })
                                                var mods = loader.map()["modules"][urlmod] || {};
                                                // 生命周期的显示
                                                if (mods.show && options.life) {
                                                    mods.show({
                                                        type: "page",
                                                        target: lastPage
                                                    });
                                                }
                                                // 如果有回调处理回调再进行编译
                                                typeof callback === "function" && callback.call(that, extportModule);
                                                // 编译组件
                                                that.compile({
                                                    id: uiDialog.config.id,
                                                });
                            
                                                options.opened && options.opened.call(that, extportModule);
                                                // 先加载完组件再执行打开
                                                !options.openFirst && uiDialog.open();
                            
                                                hasModuleInit = true;
                            
                                                return this;
                                            }
                            
                                            // 执行脚本
                                            loader.require(urlmod, function (mod) {
                                                var mods = loader.map()["modules"][urlmod] || {};
                            
                                                ui.history.setPage({
                                                    id: uiDialog.config.id,
                                                    name: urlmod,
                                                    url: options.url,
                                                    dialog: uiDialog,
                                                    exports: mod,
                                                    param: options.param
                                                })
                            
                                                // 生命周期的显示
                                                if (mods.show && options.life) {
                            
                                                    mods.show({
                                                        type: "page",
                                                        target: lastPage
                                                    })
                                                }
                                                typeof callback === "function" && callback.call(that, mod);
                            
                                                // 编译组件
                                                that.compile({
                                                    id: uiDialog.config.id,
                                                });
                                                options.opened && options.opened.call(that, mod);
                            
                                                // 模块已经加载完成
                                                hasModuleInit = true;
                                            })
                            
                            
                                        }
                            
                                        // 解析module
                                        function parsingUrl(url) {
                                            var modules = loader.map()["modules"];
                                            var mod = {};
                                            var index = url.indexOf(options.pageSuffix);
                                            if (index > -1) {
                                                mod.url = url;
                                                mod.module = url.substr(0, index);
                                            } else {
                                                // 如果模块,要从这里找到模块对应的地址
                                                mod.url = modules[url] && modules[url]["template"] ? modules[url]["template"] : url + options.pageSuffix
                                                mod.module = url;
                                            }
                            
                                            return mod;
                                        }
                            
                                        // 执行
                                        that.init(options);
                            
                                        return that;
                            
                                    }
                                })
                            
                            })(window.bui || {}, window.libs);
