                            /**
                             * @module Native
                             */
                            ;
                            (function (ui, $) {
                            
                            
                                /**
                                 * 页面跳转可以传参 获取参数请看 {{#crossLink "bui.getPageParams"}}{{/crossLink}} 方法
                                 *  <h3>预览地址: <a href="http://www.easybui.com/demo/index.html#pages/ui_method/bui.load.html" target="_blank">demo</a></h3>
                                 * @namespace  bui
                                 * @class  load
                                 * @constructor
                                 * @param {object} option
                                 *   @param {string} option.url 请求的地址
                                 *   @param {object} option.param 传过去另外页面的参数
                                 *  @param {boolean} option.firstAnimate [是否动画优先，跳转的页面如果有很多接口及渲染，可以开启，优先跳转再请求 默认 false | true]
                                 *   @param {boolean} option.replace 替换当前url (1.3.0 新增),不新增历史记录
                                 *   @param {boolean} option.reload 1.4.2新增,默认:false, 重新加载,如果单页应用要跳转到单页应用,则需要这个选项为true
                                 *   @param {boolean} option.iframe 1.4.2新增,默认:false, 单页支持iframe加载,其它开发方式无效
                                 * @example
                                 *     // 执行页面跳转
                                 *     bui.load({
                                 *         url: "",
                                 *         param:{}
                                 *     })
                                 *
                                 */
                                ui.load = function (option) {
                                    var needReload = ui.hasRouter ? router.config.reload : false;
                                    var config = {
                                        url: "",
                                        param: {},
                                        reload: needReload,
                                        firstAnimate: false,
                                        replace: false,
                                        native: true,
                                        needNative: false
                                    }
                                    var params = $.extend(true, {}, config, ui.config.load, option),
                                        newurl,
                                        querryParam = {};
                                    var url = params.url;
                                    if (!params.url) {
                                        ui.showLog("url 不能为空!", "bui.load:web");
                                        return;
                                    }
                            
                                    // 取消手机的键盘,防止键盘占位
                                    document.activeElement.blur();
                                    // 使用外部打开
                                    if (url.indexOf("tel:") >= 0 || url.indexOf("mailto:") >= 0 || url.indexOf("sms:") >= 0) {
                                        ui.unit.openExtral(url);
                                        return;
                                    }
                                    if (url && url.indexOf("?") > -1) {
                                        var paramArr = url.split("?");
                                        querryParam = ui.unit.keyStringToObject(paramArr[1]);
                                        params.url = paramArr[0];
                                    }
                            
                                    try {
                                        var paramObj = typeof params.param === "string" && JSON.parse(params.param) || params.param || {};
                            
                                        params.param = $.extend(true, {}, querryParam, paramObj);
                                    } catch (e) {
                                        ui.showLog("param 参数值必须是一个{}对象", "bui.load:web")
                                        return;
                                    }
                            
                            
                                    if (option && option.needNative || ui.nativeRouter) {
                            
                                        ui.native.load && ui.native.load(params);
                                        return;
                                    }
                            
                                    // 生成一个新的url地址
                                    newurl = ui.setUrlParams(params.url, params.param);
                            
                                    if (params.target === "_blank") {
                                        // 异步处理前
                                        var tab = window.open('about:blank');
                                        tab.location.href = ui.hasRouter && newurl.indexOf("http") !== 0 ? "index.html#" + newurl : newurl;
                                        return;
                                    }
                            
                                    // 重新加载,跟是否单页无关, 用于单页应用跟单页应用之间的跳转
                                    if (params.reload) {
                                        // # 改为 ？ 才能实现重新加载的效果
                                        window.location.href = ui.hasRouter ? "index.html?module=" + newurl : newurl;
                                        return;
                                    }
                                    //  else if (params.reload && !ui.isWebapp) {
                                    //     // 原生跳转
                                    //     ui.native.load && ui.native.load(params);
                                    //     return;
                                    // }
                                    // 采用替换的方式
                                    if (params.replace && !(ui.hasRouter)) {
                                        window.location.replace(newurl);
                                        return;
                                    }
                            
                                    // 优先使用单页,然后才是根据平台切换
                                    if (ui.hasRouter) {
                                        window.router.load(params)
                                    } else {
                                        window.location.href = newurl;
                                    }
                                    //  else {
                            
                                    //     // 如果没有原生版本,使用web
                                    //     if (typeof ui.native.load === "undefined") {
                                    //         window.location.href = newurl;
                                    //         return;
                                    //     }
                                    //     // 原生跳转
                                    //     ui.native.load && ui.native.load(params);
                                    // }
                            
                                };
                            
                            
                                return ui;
                            
                            })(window.bui || {}, window.libs);
