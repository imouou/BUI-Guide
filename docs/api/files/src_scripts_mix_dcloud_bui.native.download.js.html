                            /**
                             * @module Native
                             */
                            ;
                            (function(ui, $) {
                                /*
                                 * <h3>下载控件, 支持安卓, ios未测, web不支持</h3>
                                 * <a href="../../ui_controls/bui.download.html" target="_blank">demo</a>
                                 * <p>下载文件在本地,web不支持</p>
                                 * {{#crossLink "bui.download/getFile"}}{{/crossLink}}: 获取文件,获取不到则下载文件 <br>
                                 * {{#crossLink "bui.download/start"}}{{/crossLink}}: 下载文件 <br>
                                 * {{#crossLink "bui.download/stop"}}{{/crossLink}}: 停止下载 <br>
                                 * {{#crossLink "bui.download/toBase64"}}{{/crossLink}}: 下载的图片转换成可以插入页面的图片 <br>
                                 * {{#crossLink "bui.download/widget"}}{{/crossLink}}: 获取依赖的控件 <br>
                                 *  @namespace bui
                                 *  @class download
                                 *  @constructor
                                 *  @param {object} option
                                 *  @param {string} [option.url] [ 下载的地址 ]
                                 *  @param {object} [option.data] [ 请求下载需要传的参数 ]
                                 *  @param {number} [option.timeout] [ 触发timeout的时间默认60000 ]
                                 *  @param {string} [option.fileName] [ 自定义文件名称,包括后缀, 1.3.0 以上  ]
                                 *  @param {string} [option.folderName] [ 文件夹名称,默认是download ]
                                 *  @param {boolean} [option.showProgress] [ 是否显示进度条 默认true  ]
                                 *  @param {boolean} [option.native] [ 默认true 由debug状态决定 false则强制采用web方式 ]
                                 *  @param {function} [option.onProgress] [ 自定义进度条回调,接收百分比值 ]
                                 *  @param {function} [option.onSuccess] [ 成功的回调 ]
                                 *  @param {function} [option.onFail] [ 失败的回调 ]
                                 *  @example
                                 *
                                 *   js:
                                 *
                                        // 1. 初始化 这里如果传url初始化,则马上下载
                                        var uidownload = bui.download();
                            
                                        // 2. 开始下载
                                        uiUpload.start({
                                            data: null
                                        })
                                 *
                                 */
                                ui.native.download = function(option, depend) {
                                    var config = {
                                            url: "",
                                            data: {},
                                            headers: {},
                                            method: "GET",
                                            showProgress: true,
                                            timeout: 60000,
                                            fileName: "",
                                            folderName: "download",
                                            tips: {
                                                start: "开始下载...",
                                                process: "链接服务器...",
                                                finished: "下载完成"
                                            },
                                            native: true,
                                            onProgress: null,
                                            onSuccess: null,
                                            onFail: null
                                        }
                                        //方法
                                    var module = {
                                        start: start,
                                        stop: stop,
                                        pause: pause,
                                        resume: resume,
                                        toBase64: toBase64,
                                        init: init
                                    };
                            
                                    //用于option方法的设置参数
                                    var options = module.config = $.extend(true, {}, config, ui.config.download, option),
                                        depend = depend || {};
                            
                                    var uiAjax = null,
                                        uiFile = depend.file,
                                        uiLoading = depend.loading,
                                        webapp = ui.isWebapp || options.native == false,
                                        targetUrl,
                                        targetEncodeUrl,
                                        folderPath,
                                        filePath,
                                        showProgress = options.showProgress,
                                        onProgress = options.onProgress,
                                        onFail = options.onFail,
                                        progressPercent;
                            
                            
                                    // 初始化
                                    init(options);
                            
                                    //初始化
                                    function init(option) {
                            
                                        folderPath = "_doc/" + option.folderName + "/";
                            
                                        // 获取文件夹
                                        // uiFile.getFolder({
                                        //     folderName: option.folderName,
                                        //     create: false,
                                        //     onSuccess: function (folder,file) {
                                        //         //文件夹路径
                                        //         folderPath = folder.fullPath;
                                        //     },
                                        //     onFail: function (err) {
                                        //         // 创建文件夹
                                        //         uiFile.getFolder({
                                        //             folderName: option.folderName,
                                        //             create: true,
                                        //             onSuccess: function (folder,file) {
                                        //                 //文件夹路径
                                        //                 folderPath = folder.fullPath;
                                        //             },
                                        //             onFail: function (err) {
                                        //                 onFail && onFail(err);
                                        //             }
                                        //         });
                                        //     }
                                        // });
                                        if (!option.url) {
                                            return this;
                                        }
                            
                                        //开始下载
                                        start(option);
                            
                                        return this;
                                    }
                            
                                    /*
                                     * 开始下载
                                     *  @method start
                                     *  @param {object} [option] [参数同初始化一样,查看最顶部的参数]
                                     *  @chainable
                                     *  @example
                                     *
                                        uiDownload.start({
                                            url: "http://"
                                        })
                            
                                     */
                            
                                    function getFileName(url) {
                                        // 获取下载地址的文件名
                                        var index = url.lastIndexOf("/") + 1;
                                        var name = url.substr(index);
                                        var hasParam = name.indexOf("?") > -1;
                            
                                        return hasParam ? name.substr(0, hasParam) : name;
                                    }
                            
                                    function start(param) {
                            
                                        var params = param;
                            
                                        targetUrl = params.url;
                                        targetEncodeUrl = encodeURI(targetUrl);
                            
                                        var onProgress = params.onProgress;
                            
                                        params.showProgress && uiLoading.show({
                                            text: "0%"
                                        });
                            
                                        //文件存储路径
                                        var filePath = folderPath + (params.fileName || getFileName(targetUrl));
                            
                                        // 原生的下载
                            
                                        uiAjax = plus.downloader.createDownload(
                                            targetEncodeUrl, {
                                                method: params.method || 'GET',
                                                data: params.data || {},
                                                filename: filePath,
                                                timeout: params.timeout || 30000,
                                                headers: params.headers
                                            },
                                            function(download, status) {
                                                if (status == 200) {
                                                    // hint("下载成功: " + download.filename);
                                                    // that.openFile(filename);
                                                    // 转换成平台绝对路径
                            
                                                    params.onSuccess && params.onSuccess.call(module, download.filename, download, uiLoading);
                                                } else {
                                                    // hint({ content: "下载失败: " + status });
                                                    stopLoading();
                                                    params.onFail && params.onFail.call(module, status, uiLoading);
                                                }
                                                // 清空任务
                                                uiAjax = null;
                                            });
                            
                            
                                        uiAjax.addEventListener("statechanged", function(task, status) {
                                            if (!uiAjax) { return; }
                                            switch (task.state) {
                                                case 1:
                                                    params.showProgress && uiLoading.text(options.tips.start);
                                                    break; //开始
                                                case 2:
                                                    params.showProgress && uiLoading.text(options.tips.process);
                                                    break; //链接到服务器
                                                case 3:
                                                    var progressVal = (task.downloadedSize / task.totalSize) * 100;
                                                    var percent = parseInt(progressVal);
                                                    params.showProgress && uiLoading.text("正在下载:" + percent + "%");
                                                    if (percent == "100") {
                                                        params.showProgress && uiLoading.stop();
                                                        uiAjax = null;
                                                    }
                                                    break;
                                                case 4:
                                                    params.showProgress && uiLoading.text(options.tips.finished);
                                                    break;
                                            }
                                        });
                                        uiAjax.start();
                                    }
                            
                                    /*
                                     * 停止下载
                                     *  @method stop
                                     *  @chainable
                                     *  @example
                                     *
                                        uiDownload.stop()
                            
                                     */
                                    function stop() {
                            
                                        stopLoading();
                            
                                        uiAjax && uiAjax.abort();
                                    }
                            
                                    /*
                                     * 暂停下载
                                     *  @method pause
                                     *  @chainable
                                     *  @example
                                     *
                                        uiDownload.pause()
                            
                                     */
                                    function pause() {
                            
                                        stopLoading();
                            
                                        uiAjax && uiAjax.pause();
                                    }
                            
                                    /*
                                     * 恢复下载
                                     *  @method resume
                                     *  @chainable
                                     *  @example
                                     *
                                        uiDownload.resume()
                            
                                     */
                                    function resume() {
                            
                                        stopLoading();
                            
                                        uiAjax && uiAjax.resume();
                                    }
                            
                            
                                    // 开启进度
                                    function startLoading(percent) {
                                        uiLoading && uiLoading.show({
                                            text: percent
                                        });
                                    }
                            
                                    // 关闭进度
                                    function stopLoading() {
                                        uiLoading && uiLoading.stop();
                            
                                    }
                            
                                    /*
                                     * 把选择的图片文件转换成base64地址,可以直接展示在页面上
                                     *  @method toBase64
                                     *  @param {object} option
                                     *  @param {string} [option.data] [ 通过getFile得到的文件  ]
                                     *  @param {function} [option.onSuccess] [成功的回调]
                                     *  @param {function} [option.onFail] [失败的回调]
                                     *  @chainable
                                     *  @example
                                     *
                                        uiDownload.toBase64({
                                            data: "file:///", //本地图片路径
                                            onSuccess: function (imgurl) {
                                                $("#id").append('<img src="'+imgurl+'" />')
                                            }
                                        });
                            
                                     */
                                    function toBase64(param) {
                            
                                        uiFile.toBase64(param);
                                        return this;
                                    }
                            
                                    return module;
                                };
                            
                                return ui;
                            
                            })(window.bui || {}, window.libs);
