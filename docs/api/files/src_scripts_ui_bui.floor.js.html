                            /**
                             * UI组件
                             * @module UI
                             */
                            
                            ;
                            (function (ui, $) {
                                /**
                                   * <div class="oui-fluid">
                                   *   <div class="span8">
                                   *     <h2>楼层控件</h2>
                                   *   </div>
                                   *   <div class="span4"><a href="http://www.easybui.com/demo/index.html#pages/ui_controls/bui.floor.html" target="_blank"><img src="http://www.easybui.com/static/images/controls/bui-floor_low.gif" alt="控件预览"/></a></div>
                                   * </div>
                                   *  @namespace bui
                                   *  @class floor
                                   *  @since 1.5.6
                                   *  @constructor
                                   *  @param {object} option
                                   *  @param {string} option.id 控件的dom
                                 *  @param {string} [option.uid] 1.6.6新增，实例的自定义唯一名字，用于命令式调用
                                   *  @param {string} [option.menu] 点击操控的菜单在哪里
                                   *  @param {string} [option.handle] 菜单的循环元素是哪个,默认是.bui-btn
                                   *  @param {string} [option.floorItem] 内容的循环元素是哪个,默认是.bui-floor-item
                                   *  @param {function} option.onScroll 1.7.0 滚动触发
                                   *  @param {function} option.onScrollTop 1.7.0 滚动到顶部触发
                                   *  @param {function} option.onScrollBottom 1.7.0 滚动到底部触发
                                   *  @param {number} option.area 默认:0.5  当滚动内容到一半的距离的时候,就会跳转菜单的激活效果, 0.1则接近顶部 
                                   *  @param {number} option.height 默认:"inherit"(由外部样式定义) | "auto"(自动计算) | 300(计算以后传入) 
                                   *  @example
                                   *
                            
                                    *   // 示例1: 静态结构初始化, 固定结构, li里面的内容自定义
                                    *   html:
                                    *
                                             // 菜单可以分离
                                             <ul id="floorNav" class="bui-nav">
                                                 <li class="bui-btn active">表单</li>
                                                 <li class="bui-btn">正文</li>
                                                 <li class="bui-btn">附件(2)</li>
                                             </ul>
                                             <div id="floor" class="bui-floor">
                                                 <div class="bui-floor-main">
                                                    <!-- 关联 bui-floor-item 选择器 -->
                                                    <div class="bui-panel bui-floor-item">
                                                        <div class="bui-panel-head">表单</div>
                                                        <div class="bui-panel-main">表单内容</div>
                                                    </div>
                                                    <div class="bui-panel bui-floor-item">
                                                        <div class="bui-panel-head">正文</div>
                                                        <div class="bui-panel-main">正文内容</div>
                                                    </div>
                                                    <div class="bui-panel bui-floor-item">
                                                        <div class="bui-panel-head">附件</div>
                                                        <div class="bui-panel-main">附件内容</div>
                                                    </div>
                                                 </div>
                                                 <div class="bui-floor-foot"></div>
                                             </div>
                                            
                                            // 初始化
                                            var uiFloor = bui.floor({
                                                id: "#floor",
                                                menu: "#floorNav",          // id两个单词必须驼峰氏写法,不能中横杠
                                                floorItem: ".bui-floor-item",  // 默认,可不填
                                            })
                                    * 
                                    *
                                    */
                            
                                ui.extend({
                                    name: "floor",
                                    config: {
                                        id: "",
                                        menu: "",
                                        handle: ".bui-btn",
                                        floorItem: ".bui-floor-item",
                                        footer: ".bui-floor-footer",
                                        area: 0.5,
                                        delay: 200,
                                        height: "inherit",
                                        distance: 1,
                                        onScroll: null,
                                        onScrollTop: null,
                                        onScrollBottom: null,
                                        onLoaded: null,
                                    },
                                    callback: function (option) {
                                        var that = this,
                                            $id = null,
                                            $menu = null,
                                            $handle = null,
                                            $floorItem = null,
                                            $footer = null,
                                            flag = true, // 防止点击的时候也触发滚动
                                            hasEvent = false,
                                            scrollFirst = true, // 第一次滚动
                                            clickFirst = true, // 第一次点击
                                            timeout = null,
                                            areaHeight = 0,
                                            floorTops = []; // 存储id下有多少个
                                        // console.log(winTop)
                            
                            
                                        /**
                                         * 初始化
                                         *  @method init
                                         *  @example
                                         *
                            
                                            uiFloor.init();
                                         *
                                         */
                            
                                        that.init = function (opts) {
                                            // 合并参数
                                            var opt = $.extend(true, that.config, opts);
                                            that.config = opt;
                            
                                            if (!opt.id) {
                                                ui.showLog("id 不能为空");
                                            }
                            
                                            // 获取计算可视高度
                                            this.resize(opt);
                            
                                            // // 计算高度,默认通过样式处理
                                            // if (typeof opt.height === "string" && opt.height !== "inherit") {
                                            //     var height = $(window).height() - ui.$("header").height() - ui.$("footer").height() - $id[0].offsetTop;
                                            //     // 是否传百分比
                                            //     height = opt.height.indexOf("%") > -1 ? ($(window).height() * parseInt(opt.height, 10) / 100) : height;
                                            //     $id.height(height).css("position", "relative");
                                            //     areaHeight = height;
                                            // } else if (typeof opt.height === "number" && opt.height > 0) {
                                            //     $id.height(opt.height).css("position", "relative");
                                            //     areaHeight = opt.height;
                                            // } else {
                                            //     // 如果不是absolute 可能存在部分手机不能滚动的情况
                                            //     areaHeight = $id.height();
                                            // }
                            
                                            // 获取有多少层
                                            getFloorTops();
                            
                                            // 绑定事件
                                            if (!hasEvent) {
                                                bind(opt)
                                            }
                            
                                            opt.onLoaded && opt.onLoaded.call(that, opt);
                                        }
                            
                                        that.getSelector = function (opt) {
                            
                                            $id = ui.$(opt.id);
                                            $menu = ui.$(opt.menu);
                                            $handle = $menu && $menu.find(opt.handle);
                            
                                            $floorItem = $id.find(opt.floorItem);
                                            $footer = $id.find(opt.footer);
                            
                                            return this;
                                        }
                            
                                        that.resize = function (opt) {
                            
                                            this.getSelector(opt);
                            
                                            // 计算高度,默认通过样式处理
                                            if (typeof opt.height === "string" && opt.height !== "inherit") {
                                                var height = $(window).height() - ui.$("header").height() - ui.$("footer").height() - $id[0].offsetTop;
                                                // 是否传百分比
                                                height = opt.height.indexOf("%") > -1 ? ($(window).height() * parseInt(opt.height, 10) / 100) : height;
                                                $id.height(height).css("position", "relative");
                                                areaHeight = height;
                                            } else if (typeof opt.height === "number" && opt.height > 0) {
                                                $id.height(opt.height).css("position", "relative");
                                                areaHeight = opt.height;
                                            } else {
                                                // 如果不是absolute 可能存在部分手机不能滚动的情况
                                                areaHeight = $id.height();
                                            }
                                        }
                            
                                        function handleActive(index) {
                                            if ($menu.length === 1) {
                                                // 第一个的时候处理
                                                $handle && $handle.removeClass("active");
                                                $handle && $handle.eq(index).addClass("active");
                                            } else if ($menu.length > 1) {
                                                $menu.each((i, item) => {
                                                    let $item = $(item);
                                                    let $itemHandle = $item.find(that.config.handle);
                                                    $itemHandle.removeClass("active");
                                                    $itemHandle.eq(index).addClass("active");
                                                })
                                            }
                                        }
                                        function bind(opt) {
                                            var scrollCount = function (e) {
                                                if (flag) {
                            
                                                    var winTop = $(this).scrollTop();
                                                    if (winTop === 0) {
                                                        e.index = 0;
                                                        that.trigger("scrolltop", e);
                                                        opt.onScrollTop && opt.onScrollTop.call(that, e);
                                                    }
                                                    // 滚动时执行一次; 避免store延迟渲染导致不正确
                                                    if (winTop > 5 && scrollFirst) {
                                                        // 重新执行一次
                                                        that.getSelector(opt);
                                                        getFloorTops();
                                                        scrollFirst = false;
                                                    }
                                                    // 默认到达可视区域50% 
                                                    var winHeight = areaHeight * parseFloat(opt.area);
                                                    var index = 0;
                                                    for (index = 0; index < floorTops.length; index++) {
                                                        var item = floorTops[index];
                                                        // -1 代表该元素被隐藏
                                                        if (item.top > -1) {
                                                            var scrollItemTop = item.top - winTop;
                            
                                                            // 第一个为0等于滚动条的滚动高度则跳出循环
                                                            if (item.top == winTop && winTop == 0) {
                                                                // 第一个的时候处理
                                                                handleActive(index);
                                                                // 滚动到第几层
                                                                e.index = index;
                                                                that.trigger("scrollto", e);
                                                                return;
                                                            }
                                                            // 如果进入区域内也会操作
                                                            if (scrollItemTop < winHeight) {
                                                                handleActive(index);
                                                                // 滚动到第几层
                                                                e.index = index;
                                                                that.trigger("scrollto", e);
                                                            }
                                                        }
                                                    }
                                                    // 到底了
                                                    if (parseInt(winTop + this.offsetHeight, 10) == parseInt(this.scrollHeight, 10)) {
                                                        e.index = floorTops.length - 1;
                                                        that.trigger("scrollbottom", e);
                                                        opt.onScrollBottom && opt.onScrollBottom.call(that, e);
                                                    }
                                                }
                            
                                                that.trigger("scroll", e);
                                                opt.onScroll && opt.onScroll.call(that, e);
                            
                                            }
                                            // 绑定滚动事件处理
                                            $id.scroll(scrollCount)
                            
                                            $menu && $menu.off("click.bui", opt.handle).on("click.bui", opt.handle, function (e) {
                                                flag = false;
                                                // 第一次点击执行一次; 避免store延迟渲染导致不正确
                                                if (clickFirst) {
                                                    that.getSelector(opt);
                                                    getFloorTops();
                                                    clickFirst = false;
                                                }
                                                var index = $(this).index();
                            
                                                that.to(index);
                            
                                                that.trigger("click", e);
                                            })
                            
                                            $id.parents("main").css("position", "relative");
                            
                                            hasEvent = true;
                                        }
                            
                                        /**
                                         * 跳转
                                         *  @method to
                                         *  @param {number} index [跳转到第几个]
                                         *  @param {function} callback [跳转以后执行]
                                         *  @example
                                         *
                            
                                            uiFloor.to();
                                         *
                                         */
                            
                                        that.to = function (index, callback) {
                                            // 有可能初始化在前面,菜单动态渲染在后面,这里重新获取一下$handle
                                            $handle = $menu && $menu.find(that.config.handle);
                                            var srcollTop = floorTops[index] && floorTops[index].top || 0;
                                            $id[0]["scrollTop"] = srcollTop;
                            
                                            handleActive(index);
                                            // 防止点击时的滚动触发到滚动区域的处理
                                            if (timeout) {
                                                clearTimeout(timeout);
                                            }
                                            timeout = setTimeout(function () {
                                                flag = true;
                                            }, that.config.delay)
                            
                                            callback && callback.call(that, index);
                                            that.trigger("to", index);
                                            var e = {
                                                target: $id[0],
                                                index: index,
                                            }
                                            that.trigger("scrollto", e);
                                        }
                            
                                        /**
                                        * [销毁控件]
                                        *  @method destroy
                                        *  @example
                               
                                           //销毁
                                           uiDropdown.destroy();
                               
                                        */
                                        that.destroy = function (bool) {
                                            var bool = bool == true ? true : false;
                            
                                            if ($id) {
                                                $id.off("click.bui");
                                                $id.off("scroll");
                                                bool && $id.remove();
                                                $id = null;
                                            }
                                            if ($menu) {
                                                $menu.off("click.bui");
                                                $menu = null;
                                            }
                            
                                            that.off("click");
                                            that.off("to");
                                            that.off("scroll");
                                            that.off("scrollto");
                                            that.off("scrolltop");
                                            that.off("scrollbottom");
                            
                                            hasEvent = false;
                                        }
                            
                                        // 把楼层选择器的操作转为数组操作
                                        function getFloorTops() {
                                            // 先清空再重新获取
                                            floorTops = [];
                                            $floorItem.each(function (item, index) {
                                                var items = {
                                                    top: this.style.display === "none" ? -1 : this.offsetTop,
                                                    display: this.style.display
                                                }
                                                floorTops.push(items)
                                            });
                            
                                        }
                            
                                        // 初始化
                                        that.init(that.config);
                            
                                    }
                                })
                            
                            
                            })(window.bui || {}, window.libs);
