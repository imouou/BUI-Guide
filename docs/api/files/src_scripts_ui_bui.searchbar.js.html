                            /**
                             * UI控件库 
                             * @module UI
                             */
                            
                            ;
                            (function (ui, $) {
                            
                                /**
                                 * <div class="oui-fluid">
                                 *   <div class="span8">
                                 *     <h2>搜索条</h2>
                                 *     <p>输入时会延迟执行.示例是 searchbar 跟 list 一起使用的场景.</p>
                                 *     <h3>预览地址: <a href="http://www.easybui.com/demo/index.html#pages/ui_controls/bui.searchbar.html" target="_blank">demo</a></h3>
                                 *     <h3>方法说明:</h3>
                                 * {{#crossLink "bui.searchbar/search"}}{{/crossLink}}: 单独调用搜索方法 <br>
                                 * {{#crossLink "bui.searchbar/option"}}{{/crossLink}}: 获取设置参数 <br>
                                 * {{#crossLink "bui.searchbar/widget"}}{{/crossLink}}: 获取依赖控件 <br>
                                 *   </div>
                                 *   <div class="span4"><a href="http://www.easybui.com/demo/index.html#pages/ui_controls/bui.searchbar.html" target="_blank"><img src="http://www.easybui.com/static/images/controls/bui-searchbar_low.gif" alt="控件预览"/></a></div>
                                 * </div>
                                 *  @namespace bui
                                 *  @class searchbar
                                 *  @constructor 
                                 *  @param {object} option  
                                 *  @param {string} option.id 控件id
                                 *  @param {string} [option.uid] 1.6.6新增，实例的自定义唯一名字，用于命令式调用
                                 *  @param {string} [option.handle] 点击的搜索按钮
                                 *  @param {boolean} [option.readonly] 1.6.7 是否只读
                                 *  @param {string} [option.placeholder] 1.6.7 占位的字符
                                 *  @param {function} [option.onInput]  实时搜索的回调
                                 *  @param {function} [option.onChange]  1.5.4新增, 搜索改变的时候触发,在失去焦点后
                                 *  @param {function} [option.onKeydown]  1.7.5新增, 监听键盘触发，onKeydown(e){ console.log(e.keyCode) }  keyCode=13 为回车键
                                 *  @param {function} [option.onRemove]  移除关键词以后的回调
                                 *  @param {function} [option.onBeforeInit]  1.5.1新增 初始化前触发
                                 *  @param {function} [option.onInited]  1.5.1新增 初始化以后触发
                                 *  @param {function} [option.callback]  点击按钮的回调
                                 *  @example
                                 * 
                                 *   html:
                                 *
                                        <div id="searchbar" class="bui-searchbar">
                                            <div class="bui-input">
                                                <i class="icon-search"></i>
                                                <input type="text" value="" placeholder="搜索"/>
                                            </div>
                                        </div>
                                 *      
                                 *   js: 
                                 *   
                                        // 初始化
                                        var uiSearchbar = bui.searchbar({ 
                                            id: "#searchbar",
                                            callback: function(e,keyword) {
                                                // 点击以后做什么事情
                                            }
                                          });
                                  
                                        
                                 *
                                 */
                                // onInput 需要使用函数节流来控制输入的关键词的判断
                                ui.searchbar = function (option) {
                            
                                    var config = {
                                        id: "",
                                        uid: "",
                                        handle: ".icon-search,.btn-search",
                                        placeholder: "",
                                        handleRemove: "." + ui.config.icon.remove,
                                        delayTime: 400,
                                        readonly: false,
                                        onInput: null,
                                        onChange: null,
                                        onRemove: null,
                                        onKeydown: null,
                                        autoinit: true,
                                        onBeforeInit: null,
                                        onInited: null,
                                        callback: null
                                    }
                                    //方法
                                    var module = {
                                        name: "searchbar",
                                        handle: {},
                                        on: on,
                                        off: off,
                                        search: search,
                                        placeholder: placeholder,
                                        readonly: readonly,
                                        reset: reset,
                                        value: value,
                                        destroy: destroy,
                                        widget: widget,
                                        option: options,
                                        config: param,
                                        init: init
                                    };
                                    //用于option方法的设置参数
                                    var param = module.config = $.extend(true, {}, config, ui.config.searchbar, option);
                            
                                    var $id,
                                        id,
                                        el,
                                        keywordCache,
                                        hasEventInit = false,
                                        $input,
                                        $remove,
                                        $target;
                            
                                    //执行初始化
                                    param.autoinit && init(param);
                            
                            
                                    /**
                                     * 初始化方法,用于重新初始化结构,事件只初始化一次
                                     *  @method init
                                     *  @param {object} [option] 参数控件本身
                                     *  @chainable
                                     */
                                    function init(opt) {
                                        var option = $.extend(true, param, opt);
                            
                                        option.onBeforeInit && option.onBeforeInit.call(module, option);
                                        //验证id
                                        $id = ui.obj(option.id);
                            
                                        module.$el = $id;
                            
                                        //option获取新参数使用
                                        param = module.config = option;
                            
                                        $input = $id.find("input");
                            
                                        //增加删除按钮
                                        $remove = $id.find(option.handleRemove);
                                        if ($remove.length < 1) {
                                            $input.after('<i class="' + option.handleRemove.substr(1) + '"></i>');
                                            $remove = $id.find(option.handleRemove);
                                        }
                                        $remove.hide();
                            
                                        if (!hasEventInit) {
                                            //绑定事件
                                            bind(option);
                                        }
                            
                                        // 只读
                                        if (option.readonly) {
                                            readonly(option.readonly);
                                        }
                            
                                        // 修改占位符，默认静态输入
                                        if (option.placeholder) {
                                            placeholder(option.placeholder);
                                        }
                            
                                        // 存储实例
                                        option.uid && ui.history.setUI && ui.history.setUI({
                                            uid: option.uid,
                                            ui: module
                                        })
                            
                                        option.onInited && option.onInited.call(module, option);
                                        return this;
                                    }
                            
                                    function bind(option) {
                                        //绑定搜索
                                        $id.off("click.bui", option.handle).on("click.bui", option.handle, function (e) {
                            
                                            document.activeElement.blur();
                            
                                            var keyword = $input.val();
                                            keywordCache = keyword;
                            
                                            trigger.call(module, "search", e, keyword);
                            
                                            option.callback && option.callback.call(module, e, keyword);
                            
                                        })
                            
                                        //绑定删除
                            
                                        $id.off("click.bui", option.handleRemove).on("click.bui", option.handleRemove, function (e) {
                            
                                            document.activeElement.blur();
                            
                                            $input.val('');
                            
                                            var keyword = $input.val();
                                            keywordCache = keyword;
                            
                                            $(this).hide();
                            
                                            trigger.call(module, "remove", e, keyword);
                                            option.onRemove && option.onRemove.call(module, e, keyword);
                            
                                        })
                                        // IOS输入拼音的处理
                                        var zhFlag = false;
                                        var timeout = null;
                            
                                        //监测搜索框有没有内容, 使用延迟执行优化
                                        option.onInput && $id.off("input", "input").on('compositionstart', function () {
                                            zhFlag = true;
                                            // console.log('中文输入法，录入开始');
                                        }).on('compositionend', function (e) {
                                            zhFlag = false;
                                            var word = $input.val();
                                            // console.log('中文输入法，输入结束');
                                            // 移动端 这里不会执行, PC会多次执行
                                            option.onInput.call(module, e, word);
                                        }).on("input", "input", function (e) {
                            
                                            var keyword = $input.val();
                                            keywordCache = keyword;
                            
                                            if (keyword) {
                                                $remove.show();
                                            } else {
                                                $remove.hide();
                                            }
                                            // 中文输入法不走这里
                                            trigger.call(module, "input", e, keyword);
                                            if (zhFlag) {
                                                return;
                                            }
                            
                                            if (timeout) {
                                                clearTimeout(timeout);
                                            }
                                            timeout = setTimeout(function () {
                                                option.onInput.call(module, e, keyword);
                                                timeout = null;
                                            }, option.delayTime)
                            
                                        });
                            
                                        //失去焦点的时候触发
                                        option.onChange && $id.off("change", "input").on("change", "input", function (e) {
                                            var keyword = $input.val();
                                            keywordCache = keyword;
                            
                                            if (keyword) {
                                                $remove.show();
                                            } else {
                                                $remove.hide();
                                            }
                                            trigger.call(module, "change", e, keyword);
                                            option.onChange.call(module, e, keyword);
                                        });
                            
                                        //失去焦点的时候触发
                                        option.onKeydown && $id.off("keydown", "input").on("keydown", "input", function (e) {
                                            var keyword = $input.val();
                            
                                            trigger.call(module, "keydown", e, keyword);
                                            option.onKeydown.call(module, e, keyword);
                                        });
                            
                                        hasEventInit = true;
                                        return this;
                                    }
                            
                                    /**
                                     * 调用搜索的方法
                                     *  @method search
                                     *  @since 1.3.0
                                     *  @param {string} [name]  依赖控件名 
                                     *  @example 
                                        
                                        uiSearchbar.on("search",function(e,keyword){
                            
                                        })
                                        uiSearchbar.search();
                                            
                                     */
                                    function search(word) {
                                        var keyword = word || keywordCache;
                                        keywordCache = keyword;
                                        trigger.call(this, "search", {}, keyword);
                            
                                        param.callback && param.callback.call(this, {}, keyword);
                            
                                        return this;
                                    }
                            
                                    /**
                                     * 修改占位符
                                     *  @method placeholder
                                     *  @since 1.6.7
                                     *  @param {string} [word] 占位符 
                                     *  @example 
                                        
                                        uiSearchbar.placeholder("请输入关键词");
                                            
                                     */
                                    function placeholder(word) {
                            
                                        $input.attr('placeholder', word);
                            
                                        return this;
                                    }
                            
                                    /**
                                     * 只读，禁止输入
                                     *  @method readonly
                                     *  @since 1.6.7
                                     *  @param {boolean} [bool] true 禁止输入 false 允许输入
                                     *  @example 
                                        
                                        // 禁止输入
                                        uiSearchbar.readonly(true);
                                        // 允许输入
                                        uiSearchbar.readonly(false);
                                            
                                     */
                                    function readonly(bool) {
                            
                                        Boolean(bool) && $input.attr('readonly', bool);
                                        !Boolean(bool) && $input.removeAttr('readonly');
                            
                                        return this;
                                    }
                            
                                    /**
                                     * 获取当前输入框的值
                                     *  @method value
                                     *  @since 1.5.3
                                     *  @param {string} [name]  依赖控件名 
                                     *  @example 
                                        
                                       var val = uiSearchbar.value();
                                            
                                     */
                                    function value(val) {
                            
                                        if (typeof val === "undefined") {
                                            return $input.val();
                            
                                        } else {
                                            $input.val(val);
                                            val && $remove.show();
                                        }
                                        return module;
                                    }
                                    /**
                                     * 清空当前输入框的值
                                     *  @method reset
                                     *  @since 1.5.4
                                     *  @param {string} [name]  依赖控件名 
                                     *  @example 
                                        
                                       uiSearchbar.reset();
                                            
                                     */
                                    function reset() {
                                        $input.val('');
                                        $remove.hide();
                                        return this;
                                    }
                            
                                    /**
                                     * [销毁控件]
                                     *  @method destroy
                                     *  @since 1.4.2
                                     *  @param {boolean} [bool]  默认: false 销毁部分 | true 销毁全部   
                                     *  @example 
                                        
                                        //销毁
                                        uiSearchbar.destroy();
                                        
                                     */
                                    function destroy(bool) {
                                        var bool = bool == true ? true : false;
                            
                                        if ($id) {
                                            $id.off();
                                            bool && $id.remove();
                                        }
                            
                                        off("search");
                                        off("remove");
                                        off("input");
                            
                                    }
                            
                                    /**
                                     * 获取依赖的控件
                                     *  @method widget
                                     *  @param {string} [name]  依赖控件名 
                                     *  @example 
                                        
                                        //获取依赖控件
                                        var uiSearchbarWidget = uiSearchbar.widget();
                                        
                                            
                                     */
                                    function widget(name) {
                                        var control = {};
                                        return ui.widget.call(control, name);
                                    };
                                    /**
                                     * 获取设置参数
                                     *  @method option
                                     *  @param { string | object } [key]  不传则获取所有参数, 类型为string,没有第2个参数则获取某个参数 
                                     *  @param { string | number | boolean | function } [value]  设置参数的时候要传,设置多个参数不用传,获取参数的时候也不用传 
                                     *  @chainable
                                     *  @example 
                                        
                                        
                                        //获取所有参数
                                        var option = uiSearchbar.option();
                            
                                        //获取某个参数
                                        var id = uiSearchbar.option( "id" );
                            
                                        //修改一个参数
                                        uiSearchbar.option( "handle",".btn-search" );
                            
                                        //修改多个参数
                                        uiSearchbar.option( {"handle":".btn-search"} );
                                            
                                     */
                                    function options(key, value) {
                            
                                        return ui.option.call(module, key, value);
                                    }
                            
                            
                                    /**
                                     * 为控件绑定事件
                                     *  @event on
                                     *  @since 1.3.0
                                     *  @param {string} [type]  事件类型: "search"(搜索时触发) | "remove"(移除关键字时触发) | "input"(每次输入时触发) 
                                     *  @param {function} [callback]  绑定的事件, this 为当前点击的菜单 
                                     *  @example 
                                        
                                        uiSearchbar.on("show",function () {
                                            // 点击的菜单
                                            console.log(this);
                                        });
                                        
                                            
                                     */
                                    function on(type, callback) {
                                        ui.on.apply(module, arguments);
                                        return this;
                                    }
                            
                                    /**
                                     * 为控件取消绑定事件
                                     *  @event off
                                     *  @since 1.3.0
                                     *  @param {string} [type]  事件类型: "search"(搜索时触发) | "remove"(移除关键字时触发) | "input"(每次输入时触发) 
                                     *  @param {function} [callback]  绑定的事件, this 为当前点击的菜单 
                                     *  @example 
                                        
                                        uiSearchbar.off("show");
                                        
                                            
                                     */
                                    function off(type, callback) {
                                        ui.off.apply(module, arguments);
                                        return this;
                            
                                    }
                                    /*
                                     * 触发自定义事件
                                     */
                                    function trigger(type) {
                                        //点击事件本身,或者为空,避免循环引用
                                        module.self = (this == window || this == module) ? null : this;
                            
                                        ui.trigger.apply(module, arguments);
                                    }
                                    return module;
                                };
                            
                            
                                return ui;
                            
                            })(bui || {}, libs);
