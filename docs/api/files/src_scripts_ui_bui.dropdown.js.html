                                "use strict";
                                /**
                                 *
                                 * <div class="oui-fluid">
                                 *   <div class="span8">
                                 *     <h2>下拉菜单</h2>
                                 *     <p>下拉的菜单默认是相对于页面宽度的,当你需要相对按钮宽度时,relative改为false;下拉菜单可以选中不修改值,一般在点击的时候触发,也可以控制小三角的显示,具体可以查看demo.</p>
                                 *     <h3>预览地址: <a href="http://www.easybui.com/demo/index.html#pages/ui_controls/bui.dropdown.html" target="_blank">demo</a></h3>
                                 *     <h3>方法说明:</h3>
                                 * {{#crossLink "bui.dropdown/active"}}{{/crossLink}}: 初始化显示第几个二级菜单的值 <br>
                                 * {{#crossLink "bui.dropdown/value"}}{{/crossLink}}: 不传参则获取,传参则设置 <br>
                                 * {{#crossLink "bui.dropdown/text"}}{{/crossLink}}: 不传参则获取文本,传参则设置文本 <br>
                                 * {{#crossLink "bui.dropdown/hide"}}{{/crossLink}}: 隐藏二级菜单 <br>
                                 * {{#crossLink "bui.dropdown/show"}}{{/crossLink}}: 显示二级菜单 <br>
                                 * {{#crossLink "bui.dropdown/hideAll"}}{{/crossLink}}: 隐藏所有二级菜单 <br>
                                 * {{#crossLink "bui.dropdown/option"}}{{/crossLink}}: 获取设置参数 <br>
                                 * {{#crossLink "bui.dropdown/widget"}}{{/crossLink}}: 获取依赖控件 <br>
                                 *   </div>
                                 *   <div class="span4"><a href="http://www.easybui.com/demo/index.html#pages/ui_controls/bui.dropdown.html" target="_blank"><img src="http://www.easybui.com/static/images/controls/bui-dropdown-scenes_low.gif" alt="控件预览"/></a></div>
                                 * </div>
                                 *  @namespace bui
                                 *  @class dropdown
                                 *  @constructor
                                 *  @param {object} option
                                 *  @param {string} option.id [控件id]
                                 *  @param {array} option.data 1.5新增, [{name:"",value:""}]  
                                 *  @param {string} [option.uid] 1.6.6新增，实例的自定义唯一名字，用于命令式调用
                                 *  @param {string} [option.event] 1.7.4 默认 "click" | "hover"
                                 *  @param {string} [option.handle] 一级菜单的按钮
                                 *  @param {string} [option.handleChildren] 文本值的位置
                                 *  @param {string} [option.placeholder] 1.6.6新增, 占位文本，默认为空，以静态结构为主
                                 *  @param {string} [option.value] 1.5新增, 初始化第一个值  
                                 *  @param {string} [option.target] 二级菜单,默认是按钮的下一个
                                 *  @param {string} [option.targetHandle] 二级菜单的按钮,点击触发callback
                                 *  @param {number} [option.width]  二级菜单的宽度,一级自适应宽度 
                                 *  @param {boolean} [option.showArrow]  是否显示箭头 
                                 *  @param {boolean} [option.showActive]  是否显示选中的高亮效果 
                                 *  @param {boolean} [option.relative]  true相对于屏幕两边 false相对于父层 
                                 *  @param {boolean} [option.change]  true修改文本, false只是做选择操作,可以通过change属性定义 
                                 *  @param {boolean} [option.autoClose]  1.4.5新增,点击二级菜单的时候自动关闭, 默认:true | false 
                                 *  @param {boolean} [option.stopPropagation]  1.4.5新增,点击二级菜单的时候自动关闭, 默认:true | false 
                                 *  @param {string} [option.position]  显示的位置 bottom | top | left | right 也可以通过position属性设置,权限会更高 
                                 *  @param {function} [option.onBeforeHide]  1.7.5新增 隐藏前触发 
                                 *  @param {function} [option.onBeforeShow]  1.7.5新增 展示前触发 
                                 *  @param {function} [option.onHide]  1.7.5新增 隐藏后触发 
                                 *  @param {function} [option.onShow]  1.7.5新增 展示后触发 
                                 *  @param {function} [option.onBeforeInit]  1.5.1新增 初始化前触发 
                                 *  @param {function} [option.onInited]  1.5.1新增 初始化以后触发 
                                 *  @param {function} [option.onChange]  1.5.4新增 只有数据改变才会触发 
                                 *  @param {function} [option.callback]  点击按钮的回调,第一个参数是自己的引用 
                                 *  @example
                                 *
                                 *   html:
                                 
                                    <div id="dropdown" class="bui-dropdown">
                                        <div class="bui-btn">
                                            下拉菜单
                                        </div>
                                    </div>
                                 
                                 *   js:
                                 
                                    // 初始化
                                    var uiDropdown = bui.dropdown({
                                        id: "#dropdown",
                                        data: [{name:"分享",value:"share"}],
                                    })

                                 *
                                 */
                                ui.dropdown = function (option) {
                            
                                    //默认配置
                                    var config = {
                                        id: ".bui-dropdown",
                                        handle: ".bui-btn",
                                        handleChildren: ".span1",
                                        target: ".bui-list",
                                        targetHandle: ".bui-btn",
                                        data: null,
                                        position: "bottom",
                                        showArrow: false,
                                        showActive: true,
                                        autoClose: true,
                                        stopPropagation: false,
                                        width: 0,
                                        placeholder: "",
                                        field: null,//{ name:"name", value:"value" }
                                        value: "",
                                        relative: true,
                                        // event: "click.bui",
                                        event: "click.bui",
                                        change: true,
                                        autoinit: true,
                                        template: null,
                                        onBeforeInit: null,
                                        onInited: null,
                                        callback: null,
                                        onChange: null,
                                    }
                            
                                    //方法
                                    var module = {
                                        name: "dropdown",
                                        handle: {},
                                        on: on,
                                        off: off,
                                        active: active,
                                        disabled: disabled,
                                        enabled: enabled,
                                        value: value,
                                        values: values,
                                        change: change,
                                        getItem: values,
                                        reset: reset,
                                        text: text,
                                        hide: hide,
                                        show: show,
                                        isShow: isShow,
                                        hideAll: hideAll,
                                        destroy: destroy,
                                        widget: widget,
                                        option: options,
                                        config: param,
                                        init: init
                                    };
                                    //用于option方法的设置参数
                                    var param = module.config = $.extend(true, {}, config, ui.config.dropdown, option);
                            
                                    var id,
                                        $id,
                                        $menu,
                                        //是否使用相对定位,相对定位没有定义宽度的话是继承父层的宽度,如果
                                        winWidth = document.documentElement.clientWidth,
                                        handlers = {},
                                        direction,
                                        relative,
                                        submenuWidth,
                                        hasEventInit = false,
                                        $submenu,
                                        placeholder = "",
                                        selectVal = "",
                                        selectText = "";
                            
                            
                                    //执行初始化
                                    param.autoinit && init(param);
                            
                            
                                    /**
                                     * 初始化方法,用于重新初始化结构,事件只初始化一次
                                     *  @method init
                                     *  @param {object} [option] 参数控件本身
                                     *  @chainable
                                     */
                                    function init(opt) {
                                        var option = $.extend(true, param, opt);
                            
                                        option.onBeforeInit && option.onBeforeInit.call(module, option);
                                        if (option.id) {
                                            $id = ui.obj(option.id);
                                        } else {
                                            ui.showLog("dropdown id不能为空", "bui.dropdown.init");
                                            return;
                                        }
                            
                            
                                        module.$el = $id;
                            
                                        //option获取新参数使用
                                        param = module.config = option;
                                        $menu = $id.children(option.handle);
                                        $submenu = option.target ? $id.find(option.target) : $menu.next();
                            
                                        option.placeholder && text(option.placeholder);
                                        // 占位的文本
                                        placeholder = option.placeholder || $menu.text() || $menu.val();
                            
                                        // 动态渲染
                                        if (option.data && option.data.length) {
                            
                                            option.data = option.field ? ui.unit.filterField(option.data, option.field) : option.data;
                            
                                            var html = typeof option.template === "function" ? option.template.call(module, option.data) : template(option.data);
                                            $submenu.length ? $submenu.remove() && $menu.after(html) : $id.append(html);
                            
                                            $submenu = $id.find(option.target);
                                        }
                            
                                        relative = option.relative;
                            
                                        //下拉菜单的方向
                                        direction = $id.attr("position") || option.position;
                            
                                        var offsetLeft = $id[0] && $id[0].offsetLeft >= document.documentElement.clientWidth ? 0 : $id[0] && $id[0].offsetLeft,
                                            left = option.width ? -option.width + ($id.parent().width()) + 'px' : -offsetLeft + "px",
                            
                                            position = {
                                                "bottom": {
                                                    menuPosition: "bui-menu-bottom",
                                                    arrowPosition: "bui-arrow-up",
                                                    left: left
                                                },
                                                "top": {
                                                    menuPosition: "bui-menu-top",
                                                    arrowPosition: "bui-arrow-down",
                                                    left: left
                                                },
                                                "left": {
                                                    menuPosition: "bui-menu-left",
                                                    arrowPosition: "bui-arrow-right",
                                                    left: "auto"
                                                },
                                                "right": {
                                                    menuPosition: "bui-menu-right",
                                                    arrowPosition: "bui-arrow-left",
                                                    left: "100%"
                                                }
                                            };
                            
                                        //二级菜单需要定宽,如果是上下,显示屏幕宽度,左右继承父层的宽度
                                        submenuWidth = option.width > 0 ? option.width : (relative ? winWidth : option.width || $id.parent().width());
                            
                                        //初始化二级菜单的宽度及隐藏状态
                                        if (parseFloat(submenuWidth) > 0) {
                                            $submenu.width(submenuWidth);
                                        };
                            
                                        var className = option.showArrow ? position[direction]["arrowPosition"] + ' ' + position[direction]["menuPosition"] : position[direction]["menuPosition"];
                                        // 修正菜单左边位置
                                        fixmenu(className, position[direction]["left"]);
                            
                                        let val = option.value || $menu.attr('value') || $menu.children(option.handleChildren).val();
                            
                                        // 如果有传值,直接设置
                                        if (val) {
                                            value(val);
                            
                                            // 第一次也要触发onChange
                                            option.onChange && option.onChange.call(module, { target: null, currentTarget: null, value: selectVal });
                                        } else {
                                            //设置已经选择的二级菜单
                                            var $active = $submenu.find(option.targetHandle + ".active").eq(0);
                            
                                            var index = $active.index();
                            
                                            if (index > -1) {
                                                active(index);
                                            }
                                        }
                            
                                        if (!hasEventInit) {
                                            bind(option);
                                        }
                            
                            
                                        // 存储实例
                                        option.uid && ui.history.setUI && ui.history.setUI({
                                            uid: option.uid,
                                            ui: module
                                        })
                            
                                        option.onInited && option.onInited.call(module, option);
                                        return this;
                            
                                    }
                            
                                    function bind(option) {
                                        var prevValue = "";
                                        // 一级菜单
                                        var controlMenu = function (e) {
                            
                                            if ($(this).hasClass('disabled')) {
                                                return;
                                            }
                            
                                            var isActive = $(this).hasClass('active');
                                            var $target = option.target ? $(this).parent().find(option.target) : $(this).next();
                            
                                            //隐藏所有二级菜单;
                                            hideAll();
                                            if (isActive) {
                                                var canhide = option.onBeforeHide && option.onBeforeHide.call(module, e);
                            
                                                if (canhide === false) {
                                                    return;
                                                }
                                                $(this).removeClass("active")
                                                $target.css("display", "none")
                                                trigger.call(module, "hide");
                            
                                                option.onHide && option.onHide.call(module, e);
                                                // hide();
                                            } else {
                                                // show();
                            
                                                var canshow = option.onBeforeShow && option.onBeforeShow.call(module, e);
                            
                                                if (canshow === false) {
                                                    return;
                                                }
                                                $(this).addClass("active");
                                                $target.css("display", "block");
                                                trigger.call(module, "show");
                            
                            
                                                option.onShow && option.onShow.call(module, e);
                                            }
                            
                                            prevValue = $(this).attr("value") || "";
                            
                                            // 阻止冒泡才不会导致 body 触发关闭事件,这样也会导致外部绑定事件不能从父层绑定
                                            e.stopPropagation();
                                        };
                            
                                        // 二级菜单
                                        var controlSubmenu = function (e) {
                                            var $this = $(this);
                                            var $parent = $this.parent();
                                            var val = $this.attr("value") || "",
                                                title = $.trim($this.text());
                            
                                            //是否修改选中以后的文本值
                                            var change = $parent.attr("change") != undefined ? $parent.attr("change") : option.change;
                            
                                            //显示选中的状态
                                            if (option.showActive) {
                                                $this.addClass("active")
                                                    .siblings().removeClass('active');
                                            }
                            
                                            value.call(this, val);
                            
                                            // 是否有改变事件
                                            prevValue !== val && option.onChange && option.onChange.call(module, e);
                                            //隐藏菜单
                                            param.autoClose && hide();
                                            e.target = this;
                            
                                            option.callback && option.callback.call(module, e);
                            
                                            // 导致btn控件无法跳转页面
                                            param.stopPropagation && e.stopPropagation();
                                        }
                            
                                        //一级菜单
                                        if (option.event.indexOf("hover") > -1) {
                                            $menu.off('mouseenter mouseleave')
                                            $menu.on('mouseenter', controlMenu)
                                            $menu.on('mouseleave', function (e) {
                                                $(".bui-dropdown > .bui-btn").removeClass("active");
                                                $(".bui-dropdown > .bui-list").css("display", "none");
                                            });
                                        } else if (option.event.indexOf("click") > -1) {
                                            $menu.off(option.event).on(option.event, controlMenu)
                                        }
                            
                                        // 二级菜单
                                        $submenu.off('click.bui', option.targetHandle).on('click.bui', option.targetHandle, controlSubmenu);
                            
                                        var hidedropdown = function (e) {
                                            hideAll();
                                            e.stopPropagation();
                                        };
                                        param.autoClose && $("body").off("click.bui").on("click.bui", ":not(.bui-dropdown)", hidedropdown)
                            
                                        hasEventInit = true;
                                    }
                            
                                    //修正菜单
                                    function fixmenu(classname, value) {
                            
                                        $submenu.addClass(classname);
                                        if (relative) {
                                            $submenu.css({
                                                left: value
                                            })
                                        }
                                    }
                            
                            
                                    /**
                                     * 1.5.1 支持 设置值或者文本, 或者获取值
                                     *  @method value
                                     *  @param {string} [text] 设置的值, name 或者 value 的值
                                     *  @example
                            
                                        //设置值
                                        uiDropdown.value("10");
                            
                                        //获取值
                                        var val = uiDropdown.value();
                            
                                     */
                                    function value(val) {
                            
                                        if (typeof val === "undefined") {
                                            // callback 点击的获取
                                            var value = event && event.target ? $(event.target).attr("value") || selectVal : $menu.attr("value") || selectVal;
                            
                                            return value;
                            
                                        } else {
                            
                                            var isHtml = ui.typeof(this) === "htmllielement";
                            
                                            var $targetMenuList = isHtml && $(this).parent();
                                            var $targetMenu = isHtml ? $(this).parents(".bui-dropdown").children(param.handle) : $menu;
                                            var $menuItems = isHtml ? $targetMenuList.find(param.targetHandle) : $submenu.find(param.targetHandle);
                                            $menuItems.removeClass("active");
                            
                                            if (param.data && param.data.length) {
                            
                                                if (val == "") {
                                                    selectVal = "";
                                                    $targetMenu.attr("value", val);
                                                    return;
                                                }
                                                var vals = typeof param.data[0].value === "number" ? Number(val) : String(val);
                                                var names = typeof param.data[0].name === "number" ? Number(val) : String(val);
                                                var valIndex = ui.array.index(param.data, vals, "value"),
                                                    textIndex = ui.array.index(param.data, names, "name");
                            
                                                if (valIndex > -1) {
                                                    $menuItems.eq(valIndex).addClass("active");
                                                    // 设置值
                                                    $targetMenu.attr("value", val);
                            
                                                    // 设置文本
                                                    param.change && text.call(this, param.data[valIndex]["name"]);
                            
                                                    selectVal = val;
                            
                                                } else if (textIndex > -1) {
                            
                                                    $menuItems.eq(textIndex).addClass("active");
                                                    // 设置值
                                                    $targetMenu.attr("value", param.data[textIndex]["value"]);
                                                    // 设置文本
                                                    param.change && text.call(this, val);
                            
                                                    selectVal = param.data[textIndex]["value"];
                            
                                                }
                                            } else {
                            
                                                $menuItems.each(function (index, item) {
                                                    var innerText = item.innerText,
                                                        innerValue = item.getAttribute("value");
                                                    if (innerText == val || innerValue == val) {
                                                        $(this).addClass("active")
                                                        // 设置值
                                                        $targetMenu.attr("value", innerValue);
                                                        param.change && text.call(this, innerText);
                            
                                                        selectVal = innerValue;
                            
                                                    }
                                                })
                                            }
                                        }
                            
                                        // 如果文本值不修改,那修改value也会触发change事件
                                        if (!param.change) {
                                            trigger.call(module, "change");
                                        }
                            
                                        return selectVal;
                                    }
                            
                                    // 改变并触发事件
                                    // 1.7.6
                                    function change(val) {
                            
                                        value(val);
                            
                                        text(val);
                                        param.onChange && param.onChange.call(module, { target: null, currentTarget: null, value: val });
                                    }
                                    /**
                                     * 返回一个选中的对象
                                     *  @method values
                                     *  @since 1.5.1
                                     *  @return {object} {name:"",value:""}
                                     *  @example
                            
                                        //设置值
                                        var vals = uiDropdown.values();
                                        // {name:"",value:""}
                            
                                     */
                                    function values() {
                                        return {
                                            name: selectText,
                                            value: selectVal
                                        }
                                    }
                                    /**
                                     * 重置下拉的选择
                                     *  @method reset
                                     *  @since 1.5.1
                                     *  @example
                            
                                        uiDropdown.reset();
                            
                                     */
                                    function reset() {
                                        selectVal = "";
                                        value("");
                                        text(placeholder);
                                        hide();
                                        return this;
                                    }
                                    /**
                                     * 设置某一个二级菜单的值
                                     *  @method active
                                     *  @since 1.3.0
                                     *  @param {number} [index] 第几个二级菜单
                                     *  @example
                            
                                        //设置值
                                        uiDropdown.active(1);
                            
                                     */
                                    function active(index) {
                                        index = parseInt(index);
                                        //设置已经选择的二级菜单
                                        var $active = $submenu.find(param.targetHandle).eq(index);
                            
                                        if ($active.length >= 0) {
                                            var val = $active.attr("value") || $.trim($active.text()) || "";
                                            //设置并且隐藏菜单
                                            value(val);
                            
                                            param.showActive && $active.addClass("active").siblings().removeClass("active");
                                        }
                                        return this;
                            
                                    }
                            
                                    /**
                                     * 设置或者获取文本
                                     *  @method text
                                     *  @param {string} [text] 设置文本
                                     *  @example
                            
                                        //设置文本
                                        uiDropdown.text("广东省");
                            
                                        //获取文本
                                        var val = uiDropdown.text();
                            
                                     */
                                    function text(text) {
                                        var val = val || "";
                            
                                        if (typeof text === "undefined") {
                            
                                            var value = selectText || $.trim($menu.text());
                                            return value;
                                        } else {
                                            var isHtml = ui.typeof(this) === "htmllielement";
                            
                                            var $targetMenu = isHtml ? $(this).parents(".bui-dropdown").children(param.handle) : $menu;
                            
                                            var $span1 = $targetMenu.children(param.handleChildren);
                            
                                            if ($span1.length) {
                            
                                                if ($span1[0].tagName === "INPUT") {
                                                    $span1.val(text);
                                                    // 触发store改变
                                                    $span1.trigger("change");
                                                } else {
                                                    $span1.text(text);
                                                }
                                            } else {
                                                if ($targetMenu[0].tagName === "INPUT") {
                                                    $targetMenu.val(text);
                                                    // 触发store改变
                                                    $targetMenu.trigger("change");
                                                } else {
                                                    $targetMenu.text(text);
                                                }
                                                // $targetMenu.text(text);
                                            }

                                            trigger.call(module, "change");
                                        }
                                        return this;
                                    }

                                     *  @since 1.4.2
                                     *  @param {boolean} [bool]  默认: false 销毁部分 | true 销毁全部   
                                     *  @example
                            
                                        //销毁
                                        uiDropdown.destroy();
                            
                                     */
                                    function destroy(bool) {
                                        var bool = bool == true ? true : false;
                            
                                        if ($id) {
                                            $id.off("click.bui");
                                            bool && $id.remove();
                                        }
                                        if ($menu) {
                                            $menu.off("click.bui");
                                            bool && $menu.remove();
                                        }
                            
                                        $("body").off("click.bui");
                            
                                        off("show");
                                        off("hide");
                            
                                    }
