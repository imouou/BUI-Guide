                            /**
                             * UI控件库
                             * @module UI
                             */
                            
                            ;
                            (function (ui, $) {
                            
                                /**
                                 * <div class="oui-fluid">
                                 *   <div class="span8">
                                 *     <h2>输入框交互</h2>
                                 *     <h3>预览地址: <a href="http://www.easybui.com/demo/index.html#pages/ui_controls/bui.input.html" target="_blank">demo</a></h3>
                                 *     <h3>方法说明:</h3>
                                 *     {{#crossLink "bui.input/value"}}{{/crossLink}}: 获取值 <br>
                                 *   </div>
                                 *   <div class="span4"></div>
                                 * </div>
                                 *  @namespace bui
                                 *  @class input
                                 *  @constructor
                                 *  @since 1.5.0
                                 *  @param  {object} option [description]
                                 *  @param  {string} option.id [事件的父级]
                                 *  @param {string} [option.uid] 1.6.6新增，实例的自定义唯一名字，用于命令式调用
                                 *  @param  {string} [option.target] 目标是input
                                 *  @param {function} [option.onChange]  1.5.6新增 改变以后触发,无需修改event的参数 
                                 *  @param  {string} [option.event] 类型，默认“input” || "change"
                                 *  @param  {string} [option.iconClass] 1.5.6改为支持多图标样式名,默认:".icon-remove", 多个格式应该为: ".newfont.newfont-remove"
                                 *  @param  {boolean} [option.showIcon] 是否显示图标,会影响到callback回调, 默认: true(显示图标)|false(隐藏图标)
                                 *  @param  {boolean} [option.showLength] 是否显示长度,结合maxLength参数, 默认: false(隐藏长度)|true(显示长度)
                                 *  @param  {number} [option.maxLength] 输入的长度, 0不限制 ,默认:0 
                                 *  @param  {function} [option.onInput] 输入的回调
                                 *  @param  {function} [option.onBlur] 离开的回调, 必须返回 true 的时候, value方法才能拿到值. 如果 return fasle, 则value 方法拿不到值,用于校验
                                 *  @param  {function} [option.onFocus] 聚焦的回调
                                 *  @param {function} [option.onBeforeInit]  1.5.1新增 初始化前触发 
                                 *  @param {function} [option.onInited]  1.5.1新增 初始化以后触发 
                                 *  @param  {function} [option.callback] 点击图标的回调
                                 *  @example
                                 *
                                 * html:
                            
                                    <div class="bui-input password-input">
                                        <input id="password" type="password" placeholder="密码">
                                    </div>
                            
                                   js:
                            
                                    var uiInput = bui.input({
                                        id: ".password-input",
                                        callback: function (e) {
                                            // 点击删除按钮清空
                                            $("#password").val('');
                                            $(e.target).hide();
                                        }
                                    })
                            
                                 *
                                 */
                                ui.input = function (option) {
                            
                                    var config = {
                                        id: "",
                                        uid: "",
                                        target: "input,textarea",
                                        event: "input",
                                        iconClass: ".icon-remove",
                                        showIcon: true,
                                        showLength: false,
                                        maxLength: 0,
                                        value: "",
                                        onInput: null,
                                        onBlur: null,
                                        onFocus: null,
                                        autoinit: true,
                                        onBeforeInit: null,
                                        onChange: null,
                                        onInited: null,
                                        callback: null
                                    }
                            
                                    //方法
                                    var module = {
                                        name: "input",
                                        handle: {},
                                        empty: empty,
                                        value: value,
                                        values: values,
                                        reset: reset,
                                        toggleType: toggleType,
                                        on: on,
                                        off: off,
                                        widget: widget,
                                        option: options,
                                        config: param,
                                        init: init
                                    };
                                    //用于option方法的设置参数
                                    var param = module.config = $.extend(true, {}, config, ui.config.searchbar, option);
                            
                                    var $id,
                                        id,
                                        keywordCache,
                                        hasEventInit = false,
                                        valCache,
                                        $input,
                                        iconClass,
                                        $valLength,
                                        cacheVals = [],
                                        $target;
                            
                                    //执行初始化
                                    param.autoinit && init(param);
                            
                            
                                    /**
                                     * 初始化方法,用于重新初始化结构,事件只初始化一次
                                     *  @method init
                                     *  @param {object} [option] 参数控件本身
                                     *  @chainable
                                     */
                                    function init(opt) {
                                        var option = $.extend(true, param, opt);
                            
                                        option.onBeforeInit && option.onBeforeInit.call(module, option);
                            
                                        //option获取新参数使用
                                        param = module.config = option;
                            
                                        if (option.id == "" || option.id === null) {
                                            return;
                                        }
                            
                                        //验证id
                                        $id = ui.obj(option.id);
                            
                                        module.$el = $id;
                            
                                        $target = $id.find(option.target);
                            
                                        $input = $target;
                            
                                        iconClass = ui.unit.startWithClass(option.iconClass) ? option.iconClass : '.' + option.iconClass;
                            
                                        if (!hasEventInit) {
                                            //绑定事件
                                            bind(option);
                            
                                            if (option.showLength) {
                            
                                                var html = templateLength(option);
                                                $id.append(html);
                            
                                            }
                                            option.maxLength > 0 && $target.attr("maxlength", option.maxLength)
                            
                                        }
                            
                                        option.value && value(option.value);
                            
                                        // 存储实例
                                        option.uid && ui.history.setUI && ui.history.setUI({
                                            uid: option.uid,
                                            ui: module
                                        })
                            
                                        // 触发剩余字数的计算
                                        option.showLength && $target.trigger(option.event);
                                        option.onInited && option.onInited.call(module, option);
                                        return this;
                                    }
                            
                                    function bind(option) {
                                        // IOS输入拼音的处理
                                        var zhFlag = false;
                                        // 输入框监听延迟执行
                                        $target.on(option.event, function (e) {
                            
                                            var val = this.value || e.data && e.data.value || "",// trigger 传的值在 e.data ,this.value 获取不到
                                                $parent = $(this).parent(),
                                                $btnRemove = $parent.find(iconClass);
                            
                                            valCache = val;
                            
                                            // 可能是纯数字，数字没有长度
                                            val = String(val);
                            
                                            // bug:在数字类型，输入123e会导致 val = "" 获取到0的长度
                                            if (val !== "" && option.showIcon) {
                            
                                                if ($btnRemove && $btnRemove.length) {
                                                    $btnRemove.css("display", "block");
                                                } else {
                            
                                                    $parent.append('<i class="' + iconClass.replace(/\./g, ' ').substr(1) + '"></i>');
                                                    $btnRemove = $target.find(iconClass);
                                                }
                                            } else {
                                                $btnRemove && $btnRemove.css("display", "none");
                                            }
                                            // trigger 触发的为空，也要触发变更
                                            if (e.data && e.data.value === "") {
                                                $btnRemove && $btnRemove.css("display", "none");
                                            }
                            
                                            if (option.showLength) {
                            
                                                let len = val == $(this).attr("placeholder") ? 0 : val.length;
                            
                                                $valLength = $parent.find("em");
                                                $valLength.text(val.length);
                                            }
                                            // 中文输入法不走这里
                                            if (zhFlag) {
                                                return;
                                            }
                                            option.onInput && option.onInput.call(module, e);
                            
                                            trigger.call(module, "input", e);
                            
                                        }).on('compositionstart', function () {
                                            zhFlag = true;
                                            // console.log('中文输入法，录入开始');
                                        }).on('compositionend', function (e) {
                                            zhFlag = false;
                                            var word = $input.val();
                                            // console.log('中文输入法，输入结束');
                                            option.onInput && option.onInput.call(module, e, word);
                                        });
                            
                                        option.onChange && $target.on("change", function (e) {
                                            option.onChange.call(module, e);
                                            trigger.call(module, "change", e);
                                        })
                            
                                        option.onFocus && $target.on("focus", function (e) {
                            
                                            var val = this.value;
                            
                                            valCache = val;
                            
                                            $input = $(this);
                            
                                            if (option.showIcon) {
                            
                                                $id.find(iconClass).css("display", "none");
                            
                                                val && $(this).next().css("display", "block");
                                            }
                            
                                            option.onFocus.call(module, e);
                            
                                            trigger.call(module, "focus", e);
                            
                                        })
                            
                                        option.onBlur && $target.off("blur").on("blur", function (e) {
                            
                                            var isRight = option.onBlur && option.onBlur.call(module, e);
                                            // 如果有 onBlur 回调,则必须返回true 才会有值
                                            if (isRight == true || isRight === null) {
                                                valCache = this.value;
                                            } else {
                                                valCache = '';
                                            }
                            
                                            trigger.call(module, "blur", e);
                            
                                        })
                            
                                        // 图标点击事件
                                        option.showIcon && $id.off("click.bui", iconClass).on("click.bui", iconClass, function (e) {
                            
                                            option.callback && option.callback.call(module, e);
                            
                                            trigger.call(module, "click", e);
                                        })
                            
                            
                                        hasEventInit = true;
                            
                                        return this;
                                    }
                            
                                    function templateLength(option) {
                                        var $parent = $input.parent();
                            
                                        var html = '<span class="bui-input-length"><em>0</em>/' + option.maxLength + '</span>';
                                        return html;
                                    }
                            
                                    /**
                                     * 获取并设置值, 只能单个操作
                                     *  @method value
                                     *  @example
                            
                                        var val = uiInput.value();
                            
                                     */
                                    function value(val) {
                            
                                        if (typeof val !== "undefined") {
                            
                                            $input.val(val);
                                            // 触发剩余字数的计算，触发store的数据变更
                                            $target.trigger(param.event);
                                            return this;
                                        }
                            
                                        return $input.val();
                                    }
                            
                                    /**
                                     * 批量获取设置值
                                     *  @method values
                                     *  @since 1.8.0
                                     *  @param {array} [vals] 设置值
                                     *  @example
                            
                            
                                        // 取值
                                        var vals = uiInput.values();
                            
                            
                                        // 设置值
                                        对应的html:
                                        <div class="bui-input"><input id="user" type="text" name="username"></div>
                            
                                        uiInput.values([{
                                            id:"user",  // id 或者 name 进行区分，name也是唯一值
                                            name:"username",
                                            value:"easybui"
                                        }]);
                            
                                     */
                                    function values(val) {
                                        var vals = [];
                                        if (typeof val === "undefined") {
                                            $input.each((index, item) => {
                                                let val = {};
                            
                                                val["id"] = item.id || "";
                                                val["name"] = item.name || "";
                                                val["value"] = item.value || "";
                            
                                                vals.push(val);
                                            })
                            
                                            return vals;
                                        } else {
                                            let data = val && ui.typeof(val) === "array" ? val : [val];
                            
                                            data.forEach((item, index) => {
                                                if (item && typeof item === "object") {
                                                    // value id name
                                                    let $el = item.id ? ui.$(`#${item.id}`) : ui.$(`[name="${item.name}"`);
                            
                                                    $el && $el.val(item.value || "");
                            
                                                    // 触发store变更
                                                    $el.trigger("change");
                                                    // 自身图标的展示
                                                    $el.trigger(param.event);
                                                } else {
                                                    // 字符数字
                                                    $input.val(item || "");
                            
                                                    // 触发store变更
                                                    // 触发剩余字数的计算，触发store的数据变更
                                                    $target.trigger(param.event);
                                                }
                                            })
                                        }
                            
                                        return this;
                                    }
                                    /**
                                     * 重置
                                     *  @method reset
                                     *  @since 1.6.2
                                     *  @example
                            
                                        uiInput.reset();
                            
                                     */
                                    function reset() {
                            
                                        empty();
                                        return this;
                                    }
                            
                                    /**
                                     * 清空值, 在一组目标下，在 callback 里面，调用 this.empty(e), 会只单独清除一个
                                     *  @method empty
                                     *  @since 1.8.0
                                     *  @example
                            
                                        uiInput.epmty();
                            
                                     */
                                    function empty(e) {
                            
                                        if (e && e.target) {
                                            $(e.target).css("display", "none");
                                            $(e.target).prev().val('');
                                            // 触发store同步
                                            $(e.target).prev().trigger("change");
                                        } else {
                            
                                            $input.val('');
                                            $input.next().css("display", "none");
                            
                                            // 触发store的事件
                                            $target.trigger('change');
                                        }
                            
                                        param.showLength && $valLength.text(0);
                            
                                        return this;
                                    }
                            
                                    /**
                                     * 切换text跟password, 用于密码的展示
                                     *  @method toggleType
                                     *  @example
                            
                                        uiInput.toggleType();
                            
                                     */
                                    function toggleType() {
                            
                                        // 点击删除按钮清空
                                        var type = $input.attr("type");
                                        if (type == "text") {
                                            $input.attr('type', "password");
                                        } else {
                                            $input.attr('type', "text");
                                        }
                            
                                        return this;
                                    }
                            
                                    /**
                                     * 获取依赖的控件
                                     *  @method widget
                                     *  @param {string} [name]  依赖控件名 
                                     *  @example
                            
                                        //获取依赖控件
                                        var uiInputWidget = uiInput.widget();
                            
                            
                                     */
                                    function widget(name) {
                                        var control = {};
                                        return ui.widget.call(control, name);
                                    };
                                    /**
                                     * 获取设置参数
                                     *  @method option
                                     *  @param { string | object } [key]  不传则获取所有参数, 类型为string,没有第2个参数则获取某个参数 
                                     *  @param { string | number | boolean | function } [value]  设置参数的时候要传,设置多个参数不用传,获取参数的时候也不用传 
                                     *  @chainable
                                     *  @example
                            
                            
                                        //获取所有参数
                                        var option = uiInput.option();
                            
                                        //获取某个参数
                                        var id = uiInput.option( "id" );
                            
                            
                                     */
                                    function options(key, value) {
                            
                                        return ui.option.call(module, key, value);
                                    }
                            
                            
                                    /**
                                     * 为控件绑定事件
                                     *  @event on
                                     *  @param {string} [type]  事件类型: "click"(搜索时触发) | "focus"(移除关键字时触发) | "input"(每次输入时触发) 
                                     *  @param {function} [callback]  绑定的事件, this 为当前点击的菜单 
                                     *  @example
                            
                                        uiInput.on("input",function () {
                                            // 点击的菜单
                                            console.log(this);
                                        });
                            
                            
                                     */
                                    function on(type, callback) {
                                        ui.on.apply(module, arguments);
                                        return this;
                                    }
                            
                                    /**
                                     * 为控件取消绑定事件
                                     *  @event off
                                     *  @param {string} [type]  事件类型: "click"(搜索时触发) | "focus"(移除关键字时触发) | "input"(每次输入时触发) 
                                     *  @param {function} [callback]  绑定的事件, this 为当前点击的菜单 
                                     *  @example
                            
                                        uiInput.off("input");
                            
                            
                                     */
                                    function off(type, callback) {
                                        ui.off.apply(module, arguments);
                                        return this;
                            
                                    }
                                    /*
                                     * 触发自定义事件
                                     */
                                    function trigger(type) {
                                        //点击事件本身,或者为空,避免循环引用
                                        module.self = (this == window || this == module) ? null : this;
                            
                                        ui.trigger.apply(module, arguments);
                                    }
                                    return module;
                                };
                            
                            
                                return ui;
                            
                            })(bui || {}, libs);
