                                /**
                                 * 本地数据存储,支持字符串跟对象存储,注意:取的时候是返回一个数组
                                 *  <h3>预览地址: <a href="http://www.easybui.com/demo/index.html#pages/ui_method/bui.storage.html" target="_blank">存字符demo</a> | <a href="http://www.easybui.com/demo/index.html#pages/ui_method/bui.storage_json.html" target="_blank">存对象demo</a></h3>
                                 * @namespace  bui
                                 * @class  storage
                                 * @constructor
                                 * @param {object} [option]  
                                 * @param {number} [option.size]  存储多少条数据,默认为1 | 0 为不限制 
                                 * @param {boolean} [option.reverse]  1.4新增 默认true(最新的数据在后面),false(最新的数据在前面,常用于最近历史记录) 
                                 * @param {boolean} [option.deep]  1.5.4新增 默认false || true; 是否采用逐层取值，部分数据在存数据的时候，存的是一个多维数组，那就需要开启为true; 
                                 * @param {boolean} [option.local]  默认true 永久存储 | false 临时存储,切换页面则消失 
                                 * @param {boolean} [option.prefix]  1.5.5新增, 默认 "" , 用于区分不同数据段,clear才能指定清除 
                                 * @example
                                 *
                                    存储使用示例:
                                 *
                                    // 存储1条数据
                                    var storage = bui.storage();
                                        storage.set("username","hello");
                                        storage.set("username","bui");
                                        // 第2个会覆盖第1个
                            
                                   // 获取所有数据
                                   var data = storage.get("username");
                                   //console.log( data ) ["bui"]
                            
                                   // 拿到第一条数据的值
                                   var val = storage.get("username",0);
                                   //console.log( val ) "bui"
                            
                                 * 多条数据存储示例:
                                 *
                                    // 保存数据
                                    var storage = bui.storage({size:2});
                                    //存储第1条数据
                                    storage.set("userinfo",{"id":"n1","name":"hello"},"id");
                                    //存储第2条数据
                                    storage.set("userinfo",{"id":"n2","name":"bui"},"id");
                                    //删除1条数据,通过name值比对
                                    storage.remove("userinfo","hello","name");
                                    // 获取数据,返回一个数组
                                    var data = storage.get("userinfo");
                            
                                     //console.log( data ) [{"id":"n2","name":"bui"}]
                                 *
                                 */
                                ui.storage = function (s) {
                            
                                    var size = 1,
                                        commond = "",
                                        prefix = "",
                                        param = null,
                                        storage = null;
                            
                                    var _config = {
                                        size: size,
                                        reverse: true,
                                        deep: false,
                                        local: true,
                                        uid: "",
                                        prefix: ""
                                    }
                            
                                    var module = {
                                        init: init,
                                        get: get,
                                        getItem: get,
                                        getAll: getAll,
                                        setAll: setAll,
                                        set: set,
                                        setItem: set,
                                        remove: remove,
                                        removeItem: remove,
                                        clear: clear
                                    }
                            
                                    // 初始化
                                    init(s);
                            
                                    function init(opt) {
                                        var option = null;
                            
                                        if (typeof opt === "number" || typeof opt === "string") {
                                            //全局配置，如果没有就默认1
                                            // size = parseInt(opt) == 0 ? 0 : parseInt(opt);
                                            opt = { size: opt };
                                        }
                            
                                        option = param = $.extend(true, _config, opt);
                                        // 大小
                                        size = +option.size;
                                        prefix = option.prefix || "";
                            
                                        // 最新的数据是在数组前面还是后面, 默认是前面( true ) 后面(false)
                                        commond = option.reverse ? "push" : "unshift";
                            
                                        // 是临时存储还是永久存储
                                        storage = option.local ? localStorage : sessionStorage;
                            
                                        // 存储实例
                                        option.uid && ui.history.setUI && ui.history.setUI({
                                            uid: option.uid,
                                            ui: module
                                        })
                            
                                        return this;
                                    }
                            
                            
                                    /**
                                     * 以数组保存数据,如果数据是一个对象,需要传keyname,防止数据重复增加.
                                     *  @method set
                                     *  @chainable
                                     *  @param {string} key [存储的变量名]
                                     *  @param {object} value [存储的数据,可以是string,也可以是一个对象]
                                     *  @param {string} [keyname] 存储的数据的比对唯一键值,不传会导致数据重复,size为1 则都会替换
                                     *  @example
                                        var storage = bui.storage();
                                            //普通数组无需传字段,不传最后的值默认只存储1条数据
                                            storage.set("username","Fast");
                                            //存储对象示例
                                            storage.set("userinfo",{"id":"n1","name":"Fast"},"id");
                            
                                     */
                                    function set(name, value, keyname) {
                            
                                        name = prefix + name;
                                        //获取已有的历史记录
                                        var data = size == 1 ? null : storage["getItem"](name);
                            
                                        var datas = [];
                                        var datastring = '';
                            
                                        var items = keyname ? value[keyname] : value;
                            
                                        //添加第一条数据
                                        if (data === null) {
                            
                                            datas.push(value);
                            
                                            // 逐层转换成字符串
                                            // datastring = ui.unit.jsonToString(datas);//JSON.stringify(datas);
                                            try {
                                                datastring = JSON.stringify(datas);
                                                storage["setItem"](name, datastring);
                                            } catch (oException) {
                                                if (oException.name == 'QuotaExceededError') {
                                                    console.log('超出本地存储限额！');
                                                    //如果历史信息不重要了，可清空后再设置
                                                    // storage["removeItem"](name)
                                                    // storage["setItem"](name,datastring);
                                                }
                                                if (e.message.indexOf("JSON.strinify") > -1) {
                                                    ui.showLog('数据格式错误', "bui.storage.set");
                                                }
                                            }
                            
                                        } else {
                            
                                            var dataobj = JSON.parse(data);
                                            var bool = ui.array.compare(dataobj, items, keyname) // : ui.array.compare(dataobj, items);
                            
                                            //如果存在就覆盖,不存在就增加
                                            if (!bool) {
                            
                                                dataobj[commond](value);
                            
                                                //如果超过配置的记录，就删除最后一条
                                                if (dataobj.length > size && size != 0) {
                                                    //删除最后一个元素
                                                    commond == "unshift" ? dataobj.pop() : dataobj.shift();
                                                }
                            
                                                try {
                                                    datastring = JSON.stringify(dataobj);
                                                    storage["setItem"](name, datastring);
                                                } catch (oException) {
                                                    if (oException.name == 'QuotaExceededError') {
                                                        console.log('超出本地存储限额！');
                                                    }
                                                    if (e.message.indexOf("JSON.strinify") > -1) {
                                                        ui.showLog('数据格式错误', "bui.storage.set");
                                                    }
                                                }
                            
                                            } else {
                            
                                                // 覆盖记录
                                                ui.array.remove(dataobj, items, keyname);
                            
                                                dataobj[commond](value);
                            
                                                try {
                                                    datastring = JSON.stringify(dataobj);
                                                    storage["setItem"](name, datastring);
                                                } catch (oException) {
                                                    if (oException.name == 'QuotaExceededError') {
                                                        console.log('超出本地存储限额！');
                                                    }
                                                    if (e.message.indexOf("JSON.strinify") > -1) {
                                                        ui.showLog('数据格式错误', "bui.storage.setAll");
                                                    }
                                                }
                            
                                            }
                            
                                        };
                            
                                        return this;
                                    }
                            
                                    /**
                                     * 获取存储的数组,或者某条数据
                                     *  @method get
                                     *  @param {string} key [存储的变量名]
                                     *  @param {string|number} [value] 获取某一条数据,为数字时是在第几条数据,0为第一条数据
                                     *  @param {string} keyname [在某个字段里面]
                                     *  @return {array} 返回数组或者某条数据或者undefined
                                     *  @example
                            
                                        var storage = bui.storage();
                            
                                        // 获取数组
                                        var data = storage.get("username");
                                        //console.log( data ) // ["BUI"]
                            
                                        // 获取数组里的第1条数据,才是自己存进去的内容
                                        var data = storage.get("username",0);
                                        //console.log( data ) // BUI
                            
                                        // 获取数组里的某条数据在某个字段 本地存储的内容为: [{id:"wangws","name":"Fast"},{id:"oubr","name":"BUI"}]
                                        var data = storage.get("username","oubr","id");
                                        //console.log( data ) // {id:"oubr","name":"BUI"}
                            
                                     */
                                    function get(name, value, keyname) {
                                        name = prefix + name;
                                        var datastr = storage["getItem"](name) || "";
                                        var data;
                                        try {
                                            if (!param.deep) {
                                                data = datastr && JSON.parse(datastr);
                                            } else {
                                                // size比较大时，取值较慢，影响页面切换, 有些存数据的时候，存的是多维数组，就需要开启
                                                data = datastr && ui.unit.stringToJson(datastr);
                                            }
                            
                                        } catch (e) {
                                            data = datastr;
                                            ui.showLog(e.name + ": " + e.message, "bui.storage.get");
                                        }
                            
                                        //如果有传参数就只获取记录的某一条
                                        if (typeof value === "number" && keyname) {
                            
                                            data = data && data[value] && data[value][keyname] || undefined;
                                        } else if (typeof value === "string") {
                                            var index = ui.array.index(data, value, keyname);
                            
                                            data = data && data[index];
                                        } else if (typeof value === "number") {
                                            data = data && data[value] || undefined;
                                        } else {
                                            data = data;
                                        };
                            
                                        return data;
                                    }
                            
                            
                                    /**
                                     * 删除字段的某条信息
                                     *  @method remove
                                     *  @chainable
                                     *  @param {string} key [存储的变量名]
                                     *  @param {string|number} [value]  要删的内容 | 索引(在第几个)  
                                     *  @param {string|number} [keyname]  字段名 | 长度(1.4新增,配合value是数字时使用) 
                                     *  @example
                            
                                        var storage = bui.storage();
                            
                                            // 删除第几条数据
                                            storage.remove("username",0);
                            
                                            // 删除第1条包含之后的2条数据
                                            storage.remove("username",1,2);
                                            // 删除某条数据
                                            storage.remove("username","Fast");
                            
                                            // 删除某条数据,通过id比对, 如果存储的对象是JSON
                                            storage.remove("userinfo","n1","id");
                            
                                     */
                                    function remove(name, value, keyname) {
                                        name = prefix + name;
                                        //检查字段是否为空
                                        if (typeof name !== "string") {
                                            ui.showLog("要删除的字段名只能是字符串", "bui.storage.remove");
                                            return
                                        }
                                        //获取已有的历史记录
                                        var data = get(name) || [];
                                        //如果有传参数就只删除记录的某一条
                                        if (typeof value === "number") {
                            
                                            var len = typeof keyname === "number" ? keyname : 1;
                                            // 删除第几条数据
                                            data.splice(value, len);
                            
                                            try {
                                                var datastring = JSON.stringify(data) || "";
                            
                                                storage["setItem"](name, datastring);
                            
                                            } catch (e) {
                                                ui.showLog(e.name + ": " + e.message, "bui.storage.remove");
                            
                                            }
                                        } else if (typeof value === "string") {
                                            var datanow = ui.array.remove(data, value, keyname);
                            
                                            try {
                                                var datastring = JSON.stringify(datanow) || "";
                            
                                                storage["setItem"](name, datastring);
                            
                                            } catch (e) {
                                                ui.showLog(e.name + ": " + e.message, "bui.storage.remove");
                                            }
                            
                                        } else {
                            
                                            //删除整个记录
                                            storage["removeItem"](name);
                                        };
                            
                                        return this;
                            
                                    };
                            
                                    /**
                                     * 清空本地存储
                                     *  @method clear
                                     *  @chainable
                                     *  @example
                                        var storage = bui.storage();
                                            storage.clear();
                            
                                     */
                                    function clear() {
                                        // 只清除相同前缀的数据
                                        if (prefix) {
                                            for (var i = 0; i < localStorage.length; i++) {
                                                var key = localStorage.key(i); //获取本地存储的Key
                            
                                                if (key.indexOf(prefix) === 0) {
                                                    storage["removeItem"](key);
                                                }
                                            }
                                        } else {
                                            storage["clear"]();
                                        }
                                        return this;
                                    };
                                    /**
                                     * 获取所有本地缓存数据
                                     *  @method getAll
                                     *  @since 1.5.5
                                     *  @param index [number] 有prefix参数的时候,支持传索引,返回存进去之前的值, 不传则每个keyname返回一个数组]
                                     *  @chainable
                                     *  @example
                                        var storage = bui.storage();
                                        var localObj = storage.getAll();
                            
                                     */
                                    function getAll(index) {
                                        var obj = {};
                                        var num = parseInt(index) || 0;
                                        for (var i = 0; i < localStorage.length; i++) {
                                            var key = localStorage.key(i); //获取本地存储的Key
                                            var keyname = key.substr(prefix.length);
                                            if (prefix && key.indexOf(prefix) === 0) {
                                                // obj[key] = storage["getItem"](key);
                                                try {
                                                    var valobj = JSON.parse(storage["getItem"](key));
                                                    obj[key] = typeof index !== "undefined" ? valobj[num] : valobj;
                                                } catch (e) {
                                                    obj[key] = storage["getItem"](key);
                                                }
                                            }
                                            if (prefix === "") {
                                                try {
                                                    obj[key] = JSON.parse(storage["getItem"](key));
                                                } catch (e) {
                                                    obj[key] = storage["getItem"](key);
                                                }
                                            }
                                        }
                                        return obj;
                                    };
                                    /**
                                     * 重新设置整个缓存数组，storage是单条存的，如果整个数组要替换，set方法会设置为 [[]] , 通过setAll，可以让数组变回原来的格式 []
                                     *  @method setAll
                                     *  @since 1.8.3
                                     *  @param name {string} 存在哪个字段
                                     *  @param value {array} 设置整个数组
                                     *  @chainable
                                     *  @example
                                        var storage = bui.storage();
                            
                                        storage.setAll("keyname",[{name:"bui"}]);
                            
                                     */
                                    function setAll(name, value, keyname) {
                                        if (value && ui.typeof(value) === "array") {
                                            try {
                                                let valstr = JSON.stringify(value);
                                                storage["setItem"](name, valstr);
                                            } catch (oException) {
                                                if (oException.name == 'QuotaExceededError') {
                                                    console.log('超出本地存储限额！');
                                                    //如果历史信息不重要了，可清空后再设置
                                                    // storage["removeItem"](name)
                                                    // storage["setItem"](name,datastring);
                                                }
                                                if (e.message.indexOf("JSON.strinify") > -1) {
                                                    ui.showLog('数据格式错误', "bui.storage.setAll");
                                                }
                                            }
                                        } else {
                                            set(name, value, keyname);
                                        }
                                        return value;
                                    };
                                    return module;
                                };
                            
                                return ui;
                            
                            })(window.bui || {}, window.libs);
