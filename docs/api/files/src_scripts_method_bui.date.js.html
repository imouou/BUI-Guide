                            (function(ui, $) {
                            
                                /**
                                 * 日期常用方法
                                 * @namespace  bui
                                 * @class  date
                                 */
                                ui.date = {};
                            
                                /**
                                 * 日期格式化, 兼容IOS的日期格式转换
                                 * @namespace  bui
                                 * @method  format
                                 * @constructor
                                 * @param {string} datestr [ 日期格式或者时间戳 ]
                                 * @param {string} pattern [ 格式: yyyy-MM-dd hh:mm:ss  分别代表 年-月-日 时:分:秒 ]
                                 * @example
                                 *      
                                        // 时间戳转日期
                                 *      var date = bui.date.format(1584781609309,"yyyy-MM-dd hh:mm:ss");
                                        // 2020-03-21 17:06:49
                                        // 日期转日期
                                 *      var date = bui.date.format("2020-3-20","MM/dd");
                                        // 03/20
                                 *
                                 */
                                ui.date.format = function(str, pattern) {
                                    if (str == null) {
                                        return '';
                                    }
                                    var date = this.convert(str);
                                    pattern = pattern || "yyyy-MM-dd";
                                    var o = {
                                        "M+": date.getMonth() + 1, //月份
                                        "d+": date.getDate(), //日
                                        "h+": date.getHours(), //小时
                                        "m+": date.getMinutes(), //分
                                        "s+": date.getSeconds(), //秒
                                        "q+": Math.floor((date.getMonth() + 3) / 3), //季度
                                        "S": date.getMilliseconds() //毫秒
                                    };
                                    if (/(y+)/.test(pattern)) {
                                        pattern = pattern.replace(RegExp.$1, (date.getFullYear() + "").substr(4 - RegExp.$1.length));
                                    }
                                    for (var k in o) {
                                        if (new RegExp("(" + k + ")").test(pattern))
                                            pattern = pattern.replace(RegExp.$1, (RegExp.$1.length == 1) ? (o[k]) : (("00" + o[k]).substr(("" + o[k]).length)));
                                    }
                            
                                    return pattern;
                                };
                            
                                /**
                                 * 转日期对象
                                 * @namespace  bui
                                 * @method  convert
                                 * @constructor
                                 * @param {string} date [ 日期或者时间戳 ]
                                 * @return {object} [返回Date对象]
                                 * @example
                                 
                                 *      var date = bui.date.convert(1584781609309);
                                 *  // 返回Date对象
                            
                                 *
                                 */
                                ui.date.convert = function(d, formate) {
                                    if (d && d.constructor == Date) {
                                        return d;
                                    }
                                    // 如果是时间戳,只能是数字
                                    if (typeof d === "number" && String(d).length === 13) {
                                        return new Date(d);
                                    }
                            
                                    d = String(d);
                            
                                    var date = new Date();
                                    var year = date.getFullYear(),
                                        month = date.getMonth() + 1,
                                        today = date.getDate();
                                    /* 方法 一 */
                                    // 在Chrome和Safari中,存在当值为'2017-9'或'2017/09'时,new Date(value)会返回非法值,
                                    // 故在统一使用斜杠/格式的时候,还须为时间值补充日期
                                    d = d.replace(/[-|年|月]/img, '/').replace(/[时|分|秒|小时|分钟]/img, ':')
                                        .replace(/^(\d{4}\/\d+?)($|\s)/, function($0, $1) {
                                            return $1 + '/1'
                                        })
                                        .replace(/[日|号]/img, '')
                                        .replace(/^(\d{1,2}\/\d{1,2}?)($|\s)/, function($0, $1) {
                                            return date.getFullYear() + '/' + $1
                                        })
                                        // JAVA日期转换 2020-05-18T09:23:15.123 
                                    if (d.indexOf("T") > -1 && d.indexOf("GMT") == -1) {
                                        d = d.replace("T", " ");
                                        var ds = d.indexOf(".") > -1 ? d.split(".") : [d];
                                        return new Date(ds[0]);
                                    }
                                    // Mon May 18 2020 09:23:15 GMT+0800
                                    if (d.indexOf("GMT") > -1) {
                                        var ds = d.split(" ");
                                        var enmonth = {
                                            "Jan": "1",
                                            "Feb": "2",
                                            "Mar": "3",
                                            "Apr": "4",
                                            "May": "5",
                                            "Jun": "6",
                                            "Jul": "7",
                                            "Aug": "8",
                                            "Sep": "9",
                                            "Oct": "10",
                                            "Nov": "11",
                                            "Dec": "12"
                                        }
                                        var newDateTime = ds[3] + "/" + enmonth[ds[1]] + "/" + ds[2] + " " + ds[4];
                                        return new Date(newDateTime);
                                    }
                                    // 修复只有时间的时候无法格式化问题
                                    if (d.indexOf("/") < 0 && d.indexOf(":") > -1) {
                                        d = year + "/" + month + "/" + today + " " + d;
                                    }
                            
                                    // 修复只有日期的时候无法格式化问题
                                    if (d.length === 4) {
                                        d = d + "/1/1 00:00:00";
                                    } else if (formate === "MM" || formate === "M") {
                                        d = year + "/" + d + "/1 00:00:00";
                                    } else if (formate === "dd" || formate === "d") {
                                        d = year + "/" + month + "/" + d + " 00:00:00";
                                    }
                                    return new Date(d);
                            
                                };
                                /** 
                                 * 某个日期后多少天
                                 * @namespace  bui
                                 * @method  after
                                 * @constructor
                                 * @param {string} datestr [ 日期格式或者时间戳 ]
                                 * @param {number} n [ 默认1, 正数为日期后多少天, 负数为日期之前多少天 ]
                                 * @param {string} pattern [ 默认: yyyy-MM-dd , 格式: yyyy-MM-dd hh:mm:ss  分别代表 年-月-日 时:分:秒 ]
                                 * @example
                                 *      
                                        var date = bui.date.after("2020-3-20",5);
                                        // 2020-3-25
                                 *
                                 */
                                ui.date.after = function(str, n, pattern) {
                                    // 取多少天之后的日期
                                    var n = n === 0 ? 0 : (n || 1);
                            
                                    var date = this.convert(str);
                            
                                    // 当前时间戳
                                    var dateTime = date.getTime();
                                    var dateAfter = parseInt(n) * 1000 * 3600 * 24;
                                    var left = dateTime + dateAfter;
                            
                                    return this.format(left, pattern || "yyyy-MM-dd")
                                };
                                /** 
                                 * 某个时间后多少小时
                                 * @namespace  bui
                                 * @method  afterTime
                                 * @constructor
                                 * @param {string} timestr [ 时间 ]
                                 * @param {number} n [ 正数为时间后多少分钟, 负数为时间之前多少分钟 ]
                                 * @param {string} pattern [ 默认: hh:mm , 格式: yyyy-MM-dd hh:mm:ss  分别代表 年-月-日 时:分:秒 ]
                                 * @example
                                 *      
                                        var time = bui.date.afterTime("15:00",30);
                                        // 15:30
                            
                                 *
                                 */
                                ui.date.afterTime = function(str, n, pattern) {
                                    //提前30分钟, 提前为 负数
                                    var afterMin = n === 0 ? 0 : (n || 30);
                                    // 当前日期
                                    var date = this.format(new Date(), "yyyy/MM/dd") + " " + str;
                                    var dateObj = new Date(date);
                                    // 当前时间戳
                                    var dateTime = dateObj.getTime();
                                    var nextTime = afterMin * 1000 * 60;
                            
                                    var nextValue = dateTime + nextTime;
                            
                                    return this.format(nextValue, pattern || "hh:mm");
                                };
                            
                                /** 
                                 * 差多少, 不用区分谁先谁后
                                 * @namespace  bui
                                 * @method  count
                                 * @since  1.6.2
                                 * @constructor
                                 * @param {string} after [ 日期或时间,或数字 ]
                                 * @param {string} before [ 日期或时间,或数字 ]
                                 * @param {string} type [ 默认: date | hour | number ]
                                 * @example
                                 *      
                                        var time = bui.date.count("17:00","12:00","hour");
                                        {
                                            day: 0
                                            hour: 5
                                            minute: 0
                                            second: 0
                                            tip: "0天 05:00:00"
                                            value: 18000000
                                        }
                            
                                 *
                                 */
                                ui.date.count = function(after, before, type) {
                                    var value = null;
                                    var afterdate, beforedate;
                                    switch (type) {
                                        case "number":
                                            afterdate = after;
                                            beforedate = before;
                                            break;
                                        case "hour":
                                            let today = this.format(new Date(), "yyyy/MM/dd");
                                            let afterdatestr = today + " " + after;
                                            let afterdateObj = this.convert(afterdatestr);
                            
                                            let beforedatestr = today + " " + before;
                                            let beforedateObj = this.convert(beforedatestr);
                                            afterdate = afterdateObj.getTime();
                                            beforedate = beforedateObj.getTime();
                                            break;
                            
                                        case "date":
                                        default:
                                            afterdate = this.convert(after).getTime();
                                            beforedate = this.convert(before).getTime();
                                            break;
                                    }
                            
                                    var count = (afterdate - beforedate);
                                    count = Math.abs(count);
                            
                                    var day = parseInt(count / 1000 / 3600 / 24, 10);
                                    var hour = parseInt(count / 1000 / 3600 % 24, 10);
                                    var minute = parseInt(count / 1000 / 60 % 60, 10);
                                    var second = parseInt(count / 1000 % 60, 10);
                            
                                    var val = {
                                        value: count,
                                        day: day,
                                        hour: hour,
                                        minute: minute,
                                        second: second,
                                        tip: `${day}天 ${ui.unit.fixZero(hour)}:${ui.unit.fixZero(minute)}:${ui.unit.fixZero(second)}`
                                    };
                            
                                    return val;
                                };
                            
                                /** 
                                 * 某个月有多少天
                                 * @namespace  bui
                                 * @method  countMonth
                                 * @since  1.6.2
                                 * @constructor
                                 * @param {string} date [ 日期 ]
                                 * @example
                                 *      
                                        var howmanyday = bui.date.countMonth("2020-04");
                                        // 30天
                            
                                 *
                                 */
                                ui.date.countMonth = function(date) {
                                    var dateobj = this.convert(date);
                                    var selectedMonth = dateobj.getMonth() + 1;
                                    // 设置当前月份为下个月
                                    dateobj.setMonth(selectedMonth);
                                    // 设置当前日期为上个月的最后一天
                                    dateobj.setDate(0);
                            
                                    var val = dateobj.getDate();
                            
                                    return val;
                                };
                                /** 
                                 * 某个日期为星期几, 或者周几
                                 * @namespace  bui
                                 * @method  toWeek
                                 * @constructor
                                 * @param {string} datestr [ 日期 例如: 2020-03-20 ]
                                 * @param {number|array} n [ 默认不填返回 "星期几", 1为返回 "周几" ]
                                 * @example
                                 * 
                                 *      var week = bui.date.toWeek("2020-3-20");
                                        // 星期六
                                        var week = bui.date.toWeek("2020-3-20",1);
                                        // 周六
                                 *
                                 */
                                ui.date.toWeek = function(str, weekText) {
                                    var date = this.convert(str);
                                    var weeks = null;
                                    switch (weekText) {
                                        case 1:
                                            weeks = ["周日", "周一", "周二", "周三", "周四", "周五", "周六"];
                                            break;
                                        default:
                                            weeks = weekText || ["星期天", "星期一", "星期二", "星期三", "星期四", "星期五", "星期六"];
                                            break;
                                    }
                            
                                    return weeks[date.getDay()];
                                }
                            
                                /** 
                                 * 返回今天日期
                                 * @namespace  bui
                                 * @method  today
                                 * @since  1.6.2
                                 * @constructor
                                 * @param {string} formate [ 输出的格式 默认: M-d ]
                                 * @example
                                 * 
                                 *      var today = bui.date.today();
                                 *
                                 */
                                ui.date.today = function(formate) {
                                    formate = formate || "M-d";
                                    return this.format(new Date(), formate);
                                }
                            
                                /** 
                                 * 离现在多久,小于现在(秒,分钟,小时,天)前, 大于现在则返回 (秒,分钟,小时,天)后. 一般用于新闻的日期
                                 * @namespace  bui
                                 * @method  fromnow
                                 * @since  1.6.2
                                 * @constructor
                                 * @param {string} date [ 标准日期 ]
                                 * @example
                                 * 
                                 *      var tips = bui.date.fromnow("2020-6-18 10:00")
                                        // "32分钟前"
                                 *
                                 */
                                ui.date.fromnow = function(date) {
                                    var nowdatetime = +new Date();
                                    var target = this.convert(date).getTime();
                                    var tips = "";
                                    var left = this.count(nowdatetime, target, "date");
                                    var unit = nowdatetime > target ? "前" : "后";
                                    if (left.day) {
                                        tips = left.day + "天" + unit;
                                    } else if (left.hour) {
                                        tips = left.hour + "小时" + unit;
                                    } else if (left.minute) {
                                        tips = left.minute + "分钟" + unit;
                                    } else if (left.second) {
                                        tips = left.second + "秒" + unit;
                                    }
                            
                                    return tips;
                                }
                            
                                return ui;
                            
                            })(window.bui || {}, window.libs);
